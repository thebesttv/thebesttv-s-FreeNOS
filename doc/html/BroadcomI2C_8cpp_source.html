26 BroadcomI2C::Result BroadcomI2C::initialize()
27 {
28  // Map I2C registers
29  if (m_io.map(IO_BASE + BSC1, PAGESIZE,
30  Memory::User|Memory::Readable|Memory::Writable|Memory::Device) != IO::Success)
31  return IOError;
32 
33  // Initialize GPIO controller
34  if (m_gpio.initialize() != BroadcomGPIO::Success)
35  return IOError;
36 
37  // Set GPIO pins to ALT0 function for SDA, SCL pins
38  m_gpio.setAltFunction(2, BroadcomGPIO::Function0);
39  m_gpio.setAltFunction(3, BroadcomGPIO::Function0);
40 
41  // Detect I2C controller. In Qemu, the ClockDivider is always zero.
42  if (m_io.read(ClockDivider) == 0)
43  {
44  return NotFound;
45  }
46 
47  // Set a slow clock to attempt workaround the I2C bug in Broadcom 2835
48  setClockDivider(0x5dc * 3);
49  DEBUG("I2C GPIO pins set");
50  DEBUG("ClockDivider is " << m_io.read(ClockDivider));
51  DEBUG("Status is " << m_io.read(Status));
52 
53  // Done
54  return Success;
55 }

70 BroadcomI2C::Result BroadcomI2C::write(u8 *buf, Size size)
71 {
72  if (size > FIFOSize)
73  return RangeError;
74 
75  // Clear FIFO and Status
76  m_io.write(Control, ClearFIFO);
77  m_io.write(Status, ClockStretchTimeout | AcknowledgeError | TransferDone);
78 
79  // Set data length
80  m_io.write(DataLength, size);
81 
82  DEBUG("Writing " << size << " bytes");
83 
84  // Fill the FIFO
85  for (Size i = 0; i < size; i++)
86  {
87  DEBUG("buf[" << i << "] = " << buf[i]);
88  m_io.write(FIFO, buf[i]);
89  }
90 
91  // Begin transfer
92  m_io.write(Control, Enable | Transfer);
93 
94  // Wait until transfer done is set
95  while (!(m_io.read(Status) & TransferDone));
96 
97  // Check status
98  if (m_io.read(Status) & AcknowledgeError)
99  {
100  ERROR("Acknowledge error");
101  return IOError;
102  }
103  else if (m_io.read(Status) & ClockStretchTimeout)
104  {
105  ERROR("ClockStretchTimeout error");
106  return IOError;
107  }
108  // TMP: flag done? This is not documented but used in libbcm
109  m_io.write(Control, m_io.read(Control) | 0x2);
110 
111  // Done
112  return Success;
113 }

115 BroadcomI2C::Result BroadcomI2C::read(u8 *buf, Size size)
116 {
117  Size count = 0;
118 
119  if (size > FIFOSize)
120  return RangeError;
121 
122  // Clear FIFO and Status
123  m_io.write(Control, ClearFIFO);
124  m_io.write(Status, ClockStretchTimeout | AcknowledgeError | TransferDone);
125 
126  // Set data length
127  m_io.write(DataLength, size);
128 
129  // Begin transfer
130  m_io.write(Control, Enable | ReadMode | Transfer);
131  DEBUG("start read " << size << " bytes");
132  DEBUG("status: " << m_io.read(Status));
133 
134  // Wait until transfer done is set
135  while (!(m_io.read(Status) & TransferDone));
136 
137  // Read out the whole FIFO
138  while (m_io.read(Status) & RxFIFOHasData)
139  {
140  buf[count++] = m_io.read(FIFO);
141  DEBUG("buf[" << count-1 << "]=" << buf[count-1]);
142  }
143 
144  // Check status
145  if (m_io.read(Status) & AcknowledgeError)
146  {
147  ERROR("Acknowledge error");
148  return IOError;
149  }
150  else if (m_io.read(Status) & ClockStretchTimeout)
151  {
152  ERROR("ClockStretchTimeout error");
153  return IOError;
154  }
155  // TMP: flag done? This is not documented but used in libbcm
156  m_io.write(Control, m_io.read(Control) | 0x2);
157 
158  // Done
159  return Success;
160 }
