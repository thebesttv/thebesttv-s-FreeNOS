89 Error UDP::sendPacket(NetworkClient::SocketInfo *src, IOBuffer & buffer, Size size)
90 {
91  NetworkClient::SocketInfo dest;
92  NetworkQueue::Packet *pkt;
93  Header *hdr;
94  Error r;
95 
96  DEBUG("");
97 
98  // Read destination
99  buffer.read(&dest, sizeof(dest));
100  DEBUG("send payload to: " << dest.address << " port: " << dest.port << " size: " << size);
101 
102  // Get a fresh IP packet
103  r = m_ipv4->getTransmitPacket(
104  &pkt, dest.address, IPV4::UDP, sizeof(Header) + size - sizeof(dest)
105  );
106  if (r != ESUCCESS)
107  return r;
108 
109  // Fill UDP header
110  hdr = (Header *) (pkt->data + pkt->size);
111  hdr->sourcePort = cpu_to_be16(src->port);
112  hdr->destPort = cpu_to_be16(dest.port);
113  hdr->length = cpu_to_be16(size - sizeof(dest) + sizeof(Header));
114  hdr->checksum = 0;
115 
116  // Insert payload. The payload is just after the 'dest' struct in the IOBuffer.
117  buffer.read(pkt->data + pkt->size + sizeof(Header), size - sizeof(dest), sizeof(dest));
118 
119  // Calculate final checksum
120  hdr->checksum = checksum((IPV4::Header *)(pkt->data + pkt->size - sizeof(IPV4::Header)),
121  hdr, size - sizeof(dest));
122  DEBUG("checksum = " << (uint) hdr->checksum);
123 
124  // Increment packet size
125  pkt->size += sizeof(Header) + size - sizeof(dest);
126 
127  // Transmit now
128  return m_device->transmit(pkt);
129 }

158 const u16 UDP::checksum(const IPV4::Header *ip,
159  const UDP::Header *udp,
160  const Size datalen)
161 {
162  IPV4::PseudoHeader phr;
163  ulong sum = 0;
164 
165  // Setup a pseudo header
166  phr.reserved = 0;
167  phr.source = ip->source;
168  phr.destination = ip->destination;
169  phr.protocol = IPV4::UDP;
170  phr.length = cpu_to_be16((sizeof(Header) + datalen));
171  DEBUG("ip src = " << phr.source << " dst = " << phr.destination);
172 
173  // Sum the pseudo header
174  sum += calculateSum((u16 *) &phr, sizeof(phr));
175  sum += calculateSum((u16 *) udp, sizeof(*udp) + datalen);
176 
177  // Keep only last 16 bits and add carry bits
178  sum = (sum >> 16) + (sum & 0xffff);
179  sum += (sum >> 16);
180 
181  // Take one's complement
182  sum = ~sum;
183  return (u16) sum;
184 }
