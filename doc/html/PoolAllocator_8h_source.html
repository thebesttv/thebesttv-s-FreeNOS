67  {

69  Size num = count / sizeof(Address);
70 
71  // At least one
72  if (!num) num++;
73 
74  // Scan bitmap as fast as possible
75  for (Size i = 0; i < num; i++)
76  {
77  // Point to the correct offset
78  ptr = (Address *) (&blocks) + i;
79 
80  // Any blocks free?
81  if (*ptr != (Address) ~ZERO)
82  {
83  // Find the first free bit
84  for (Size bit = 0; bit < sizeof(Address) * 8; bit++)
85  {
86  if (!(*ptr & 1 << bit))
87  {
88  *ptr |= (1 << bit);
89  free--;
90  return addr + (((i * sizeof(Address) * 8) + bit) * size);
91  }
92  }
93  }
94  }
95  // Out of memory
96  return ZERO;
