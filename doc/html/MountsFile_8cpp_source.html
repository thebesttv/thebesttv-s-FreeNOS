44 Error MountsFile::write(IOBuffer & buffer, Size size, Size offset)
45 {
46  FileSystemMount fs;
47  FileSystemMount *mounts = getMounts();
48  Error r;
49 
50  // Input must be exactly one FileSystemMount struct
51  if (size != sizeof(fs))
52  return EIO;
53 
54  // Copy the input mount struct
55  if ((r = buffer.read(&fs, sizeof(fs))) <= 0)
56  return r;
57 
58  // Append to our filesystem mounts table
59  for (Size i = 0; i < FILESYSTEM_MAXMOUNTS; i++)
60  {
61  if (!mounts[i].path[0])
62  {
63  memcpy((void *)&mounts[i], &fs, sizeof(fs));
64  NOTICE("mounted " << mounts[i].path);
65  return size;
66  }
67  }
68 
69  // Mounts table is full
70  return ENOBUFS;
71 }
