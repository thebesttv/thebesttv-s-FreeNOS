64 FileStatus::Result FileStatus::printStatus(const String & file) const
65 {
66  struct stat st;
67 
68  // Try to stat the file
69  if ((stat(*file, &st)) < 0)
70  {
71  ERROR("failed to stat `" << *file << "': " << strerror(errno));
72  return IOError;
73  }
74 
75  // Output file statistics
76  printf("File: %s\r\n", *file);
77  printf("Type: ");
78 
79  // Print the right file type
80  if (S_ISREG(st.st_mode))
81  printf("Regular File\r\n");
82 
83  else if (S_ISDIR(st.st_mode))
84  printf("Directory\r\n");
85 
86  else if (S_ISCHR(st.st_mode))
87  {
88  printf("Character Device\r\n");
89  printf("Major ID: %u\r\n", st.st_dev.major);
90  printf("Minor ID: %u\r\n", st.st_dev.minor);
91  }
92  else if (S_ISBLK(st.st_mode))
93  {
94  printf("Block Device\r\n");
95  printf("Major ID: %u\r\n", st.st_dev.major);
96  printf("Minor ID: %u\r\n", st.st_dev.minor);
97  }
98  else
99  printf("Unknown\r\n");
100 
101  // Print additional file information fields
102  printf("Mode: %u\r\n", st.st_mode);
103  printf("Size: %u\r\n", st.st_size);
104  printf("Uid: %u\r\n", st.st_uid);
105  printf("Gid: %u\r\n", st.st_gid);
106 
107  // Done
108  return Success;
109 }
