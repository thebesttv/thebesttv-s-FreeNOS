99 void setupHeap()
100 {
101  PageAllocator *pageAlloc;
102  PoolAllocator *poolAlloc;
103  Arch::MemoryMap map;
104  Memory::Range heap = map.range(MemoryMap::UserHeap);
105 
106  // Allocate one page to store the allocators themselves
107  Memory::Range range;
108  range.size = PAGESIZE;
109  range.access = Memory::User | Memory::Readable | Memory::Writable;
110  range.virt = heap.virt;
111  range.phys = ZERO;
112  VMCtl(SELF, Map, &range);
113 
114  // Allocate instance copy on vm pages itself
115  pageAlloc = new (heap.virt) PageAllocator(heap.virt, heap.size);
116  poolAlloc = new (heap.virt + sizeof(PageAllocator)) PoolAllocator();
117  poolAlloc->setParent(pageAlloc);
118 
119  // Set default allocator
120  Allocator::setDefault(poolAlloc);
121 }

140 void setupMappings()
141 {
142  // Fill the mounts table
143  memset(mounts, 0, sizeof(FileSystemMount) * FILESYSTEM_MAXMOUNTS);
144  strlcpy(mounts[0].path, "/sys", PATH_MAX);
145  strlcpy(mounts[1].path, "/", PATH_MAX);
146  mounts[0].procID = SYSFS_PID;
147  mounts[0].options = ZERO;
148  mounts[1].procID = ROOTFS_PID;
149  mounts[1].options = ZERO;
150 
151  // Map user program arguments
152  Arch::MemoryMap map;
153  Memory::Range argRange = map.range(MemoryMap::UserArgs);
154 
155  // First page is the argc+argv (skip here)
156  // Second page is the current working directory
157  currentDirectory = new String((char *) argRange.virt + PAGESIZE, false);
158 
159  // Third page and above contain the file descriptors table
160  files = (FileDescriptor *) (argRange.virt + (PAGESIZE * 2));
161 
162  // Inherit file descriptors table from parent (if any).
163  // Without a parent, just clear the file descriptors
164  if (getppid() == 0)
165  {
166  memset(files, 0, argRange.size - (PAGESIZE * 2));
167  (*currentDirectory) = "/";
168  }
169 }

171 ProcessID findMount(const char *path)
172 {
173  FileSystemMount *m = ZERO;
174  Size length = 0, len;
175  char tmp[PATH_MAX];
176 
177  // Is the path relative?
178  if (path[0] != '/')
179  {
180  getcwd(tmp, sizeof(tmp));
181  snprintf(tmp, sizeof(tmp), "%s/%s", tmp, path);
182  }
183  else
184  strlcpy(tmp, path, PATH_MAX);
185 
186  // Find the longest match
187  for (Size i = 0; i < FILESYSTEM_MAXMOUNTS; i++)
188  {
189  if (mounts[i].path[0])
190  {
191  len = strlen(mounts[i].path);
192 
193  // Only choose this mount, if it matches,
194  // and is longer than the last match.
195  if (strncmp(tmp, mounts[i].path, len) == 0 && len > length)
196  {
197  length = len;
198  m = &mounts[i];
199  }
200  }
201  }
202 
203  // All done
204  return m ? m->procID : ZERO;
205 }

207 void refreshMounts(const char *path)
208 {
209  FileSystemMessage msg;
210  pid_t pid = getpid();
211 
212  // Skip for rootfs and sysfs
213  if (pid == ROOTFS_PID || pid == SYSFS_PID)
214  return;
215 
216  // Clear mounts table
217  MemoryBlock::set(&mounts[2], 0, sizeof(FileSystemMount) * (FILESYSTEM_MAXMOUNTS-2));
218 
219  // Re-read the mounts table from SysFS.
220  msg.type = ChannelMessage::Request;
221  msg.action = ReadFile;
222  msg.path = "/sys/mounts";
223  msg.buffer = (char *) &mounts;
224  msg.size = sizeof(FileSystemMount) * FILESYSTEM_MAXMOUNTS;
225  msg.offset = 0;
226  msg.from = SELF;
227  ChannelClient::instance->syncSendReceive(&msg, SYSFS_PID);
228 }

289  arguments = (char *) map.range(MemoryMap::UserArgs).virt;
290 
291  // Fill in arguments list
292  while (argc < ARGV_COUNT && *arguments)
293  {
294  argv[argc] = arguments;
295  arguments += ARGV_SIZE;
296  argc++;
297  }
298 
299  // Pass control to the program
300  ret = main(argc, argv);
301 
302  // Terminate execution
303  runDestructors();
304  exit(ret);
