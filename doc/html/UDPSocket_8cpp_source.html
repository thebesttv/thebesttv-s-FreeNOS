40 Error UDPSocket::read(IOBuffer & buffer, Size size, Size offset)
41 {
42  DEBUG("");
43 
44  NetworkQueue::Packet *pkt = m_queue.pop();
45  if (!pkt)
46  return EAGAIN;
47 
48  IPV4::Header *ipHdr = (IPV4::Header *)(pkt->data + sizeof(Ethernet::Header));
49  UDP::Header *udpHdr = (UDP::Header *)(pkt->data + sizeof(Ethernet::Header) + sizeof(IPV4::Header));
50  NetworkClient::SocketInfo info;
51  Size payloadSize = pkt->size - sizeof(Ethernet::Header)
52  - sizeof(IPV4::Header)
53  - sizeof(UDP::Header);
54 
55  // Fill socket info
56  info.address = ipHdr->source;
57  info.port = udpHdr->sourcePort;
58  buffer.write(&info, sizeof(info));
59 
60  // Fill payload
61  Size sz = size > payloadSize ? payloadSize : size;
62  buffer.write(udpHdr+1, sz, sizeof(info));
63  m_queue.release(pkt);
64  return sz + sizeof(info);
65 }

67 Error UDPSocket::write(IOBuffer & buffer, Size size, Size offset)
68 {
69  DEBUG("");

72  if (!m_info.port)
73  {
