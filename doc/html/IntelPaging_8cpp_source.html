23 IntelPaging::IntelPaging(MemoryMap *map, SplitAllocator *alloc)
24  : MemoryContext(map, alloc)
25  , m_pageDirectory(0)
26  , m_pageDirectoryAddr(0)
27  , m_pageDirectoryAllocated(false)
28 {
29  IntelCore core;
30 
31  // Allocate page directory from low physical memory.
32  if (alloc->allocateLow(sizeof(IntelPageDirectory),
33  &m_pageDirectoryAddr) == Allocator::Success)
34  {
35  m_pageDirectoryAllocated = true;
36  m_pageDirectory = (IntelPageDirectory *) alloc->toVirtual(m_pageDirectoryAddr);
37 
38  // Initialize the page directory
39  MemoryBlock::set(m_pageDirectory, 0, sizeof(IntelPageDirectory));
40 
41  // Lookup the currently active page directory
42  IntelPageDirectory *currentDirectory =
43  (IntelPageDirectory *) alloc->toVirtual(core.readCR3());
44 
45  // Inherit kernel mappings. The kernel has permanently mapped 1GB of
46  // physical memory (i.e. the "low memory" in SplitAllocator). The low
47  // memory starts at its physical base address offset (varies per core).
48  Memory::Range kdata = m_map->range(MemoryMap::KernelData);
49  m_pageDirectory->copy(currentDirectory,
50  kdata.virt,
51  kdata.virt + kdata.size);
52 
53  // Also inherit kernel private mappings, such as APIC mappings.
54  kdata = m_map->range(MemoryMap::KernelPrivate);
55  m_pageDirectory->copy(currentDirectory,
56  kdata.virt,
57  kdata.virt + kdata.size);
58  }
59 }
