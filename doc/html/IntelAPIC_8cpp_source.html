54 Timer::Result IntelAPIC::start(IntelPIT *pit)
55 {
56  u32 t1, t2, ic, loops = 20;
57 
58  // Start the APIC timer
59  m_io.write(DivideConfig, Divide16);
60  m_io.write(InitialCount, 0xffffffff);
61  m_io.write(Timer, TimerVector | PeriodicMode);
62 
63  // Measure the speed of the APIC timer using the
64  // known absolute frequency of the PIT timer. First
65  // wait for the next PIT trigger.
66  pit->waitTrigger();
67 
68  // Collect the current APIC timer counter
69  t1 = m_io.read(CurrentCount);
70 
71  // Wait for several PIT triggers
72  for (uint i = 0; i < loops; i++)
73  pit->waitTrigger();
74 
75  // Measure the current APIC timer counter again.
76  t2 = m_io.read(CurrentCount);
77 
78  // Configure the APIC timer to run at the same frequency as the PIT.
79  ic = (t1 - t2) / loops;
80  m_frequency = pit->getFrequency();
81  m_io.write(InitialCount, ic);
82 
83  // Calculate APIC bus frequency in hertz using the known PIT
84  // frequency as reference for diagnostics.
85  u32 busFreq = ic * 16 * pit->getFrequency();
86  NOTICE("Detected " << busFreq / 1000000 << "."
87  << busFreq % 1000000 << " Mhz APIC bus");
88  NOTICE("APIC counter set at " << ic);
89  return Timer::Success;
90 }

92 Timer::Result IntelAPIC::wait(u32 microseconds) const
93 {
94  if (!isKernel)
95  {
96  Timer::Info info;
97 
98  // Get current kernel timer ticks
99  if (ProcessCtl(SELF, InfoTimer, (Address) &info) != API::Success)
100  return Timer::IOError;
101 
102  // Frequency must be set
103  if (!info.frequency)
104  return Timer::IOError;
105 
106  // Set time to wait (very rough approximation)
107  u32 msecPerTick = 1000.0 / info.frequency;
108  u32 msecToWait = microseconds / 1000;
109  info.ticks += (msecToWait / msecPerTick) + 1;
110 
111  // Wait until the timer expires
112  if (ProcessCtl(SELF, WaitTimer, (Address) &info) != API::Success)
113  return Timer::IOError;
114  }
115  else
116  {
117  Size usecPerInt = 1000000 / m_frequency;
118  Size usecPerTick = usecPerInt / m_io.read(InitialCount);
119  u32 t1 = m_io.read(CurrentCount), t2;
120  u32 waited = 0;
121 
122  while (waited < microseconds)
123  {
124  t2 = m_io.read(CurrentCount);
125  if (t2 < t1)
126  {
127  waited += (t1 - t2) * usecPerTick;
128  t1 = t2;
129  }
130  t1 = t2;
131  }
132  }
133  return Timer::Success;
134 }

181 IntController::Result IntelAPIC::sendStartupIPI(uint cpuId, Address addr)
182 {
183  ulong cfg;
184 
185  // Write APIC Destination
186  cfg = m_io.read(IntCommand2);
187  cfg &= 0x00ffffff;
188  m_io.write(IntCommand2, cfg | APIC_DEST(cpuId));
189 
190  // Assert INIT
191  cfg = m_io.read(IntCommand1);
192  cfg &= ~0xcdfff;
193  cfg |= (APIC_DEST_FIELD | APIC_DEST_LEVELTRIG |
194  APIC_DEST_ASSERT | APIC_DEST_DM_INIT);
195  m_io.write(IntCommand1, cfg);
196 
197  // Wait 10 miliseconds
198  wait(10000);
199 
200  // Send two SIPI's
201  for (Size i = 0; i < 2; i++)
202  {
203  // Write APIC Destination
204  cfg = m_io.read(IntCommand2);
205  cfg &= 0x00ffffff;
206  m_io.write(IntCommand2, cfg | APIC_DEST(cpuId));
207 
208  // Assert STARTUP
209  cfg = m_io.read(IntCommand1);
210  cfg &= ~0xcdfff;
211  cfg |= (APIC_DEST_FIELD | APIC_DEST_DM_STARTUP |
212  (addr >> 12));
213  m_io.write(IntCommand1, cfg);
214 
215  // Wait 1 milisecond
216  wait(1000);
217  }
218 
219  // Startup interrupt delivered.
220  return IntController::Success;
221 }
