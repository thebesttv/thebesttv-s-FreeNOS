28 Error VGA::initialize()
29 {
30  Memory::Range range;
31 
32  // Request VGA memory
33  range.size = PAGESIZE;
34  range.access = Memory::User |
35  Memory::Readable |
36  Memory::Writable;
37  range.virt = ZERO;
38  range.phys = VGA_PADDR;
39  VMCtl(SELF, Map, &range);
40 
41  // Point to the VGA mapping
42  vga = (u16 *) range.virt;
43 
44  // Clear screen
45  for (uint i = 0; i < width * height; i++)
46  {
47  vga[i] = VGA_CHAR(' ', LIGHTGREY, BLACK);
48  }
49 
50  // Disable hardware cursor
51  WriteByte(VGA_IOADDR, 0x0a);
52  WriteByte(VGA_IODATA, 1 << 5);
53 
54  // Successfull
55  return ESUCCESS;
56 }
