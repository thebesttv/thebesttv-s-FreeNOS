38 int Application::run()
39 {
40  // Set application name
41  if (m_argc < 1)
42  {
43  usage();
44  return ExitFailure;
45  }
46  else
47  m_parser.setName(m_argv[0]);
48 
49  // Parse commandline arguments
50  if (m_parser.parse(m_argc, m_argv, m_arguments) != ArgumentParser::Success)
51  {
52  usage();
53  return ExitFailure;
54  }
55 
56  // If the help argument is given, show the usage
57  if (m_arguments.get("help"))
58  {
59  usage();
60  return ExitFailure;
61  }
62 
63  // If the version argument is given, show version
64  if (m_arguments.get("version"))
65  {
66  output(m_version);
67  output("\n");
68  return ExitSuccess;
69  }
70 
71  // Enable debug logging if specified
72  if (m_arguments.get("debug") && Log::instance)
73  Log::instance->setMinimumLogLevel(Log::Debug);
74 
75  // Initialize the application first
76  Result r = initialize();
77  if (r == ShowUsage)
78  usage();
79  if (r != Success)
80  return ExitFailure;
81 
82  // Run the application
83  if (exec() == Success)
84  return ExitSuccess;
85  else
86  return ExitFailure;
87 }
