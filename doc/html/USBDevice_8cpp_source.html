51 Error USBDevice::initialize()
52 {
53  DEBUG("");
54 
55  String usbTransfer;
56  Address actualId = m_id;
57 
58  usbTransfer << *m_busPath << "/transfer";
59 
60  // Try to open the USB transfer file on the USB host controller
61  if ((m_transferFile = ::open(*usbTransfer, O_RDWR)) < 0)
62  {
63  ERROR("failed to open " << *usbTransfer << ": " << strerror(errno) << "\n");
64  return errno;
65  }
66  DEBUG("opened: " << *usbTransfer);
67 
68  // Every unconfigured USB device starts with address zero
69  m_id = 0;
70 
71  // First retrieve the device descriptor in 8-bytes packet only
72  // This is to figure out the maxPacketSize which is needed for transfers
73  DEBUG("getting max packet size");
74  if (getDeviceDescriptor(m_device, 8) != ESUCCESS)
75  {
76  ERROR("failed to get max packet size");
77  return EIO;
78  }
79 
80  DEBUG("maxPacketSize = " << m_device->maxPacketSize);
81 
82  // Retrieve device descriptor
83  DEBUG("getting device descriptor");
84  if (getDeviceDescriptor(m_device) != ESUCCESS)
85  {
86  ERROR("failed to get device descriptor");
87  return EIO;
88  }
89 
90  DEBUG("got descriptor: vendorId=" << m_device->vendorId <<
91  " productId=" << m_device->productId <<
92  " class=" << m_device->deviceClass <<
93  " subClass=" << m_device->deviceSubClass <<
94  " maxpacketsize=" << m_device->maxPacketSize <<
95  " protocol=" << m_device->deviceProtocol);
96 
97  // Retrieve configuration descriptor
98  if (getConfigDescriptor(m_config) != ESUCCESS)
99  {
100  ERROR("failed to get configuration descriptor");
101  return EIO;
102  }
103  // Check configuration length
104  if (m_config->totalLength < sizeof(USBDescriptor::Configuration))
105  {
106  ERROR("invalid size for configuration descriptor: " << m_config->totalLength);
107  return EIO;
108  }
109 
110  // Set address of the USB device.
111  if (setAddress(actualId) != ESUCCESS)
112  {
113  ERROR("failed to set address to: " << actualId);
114  return EIO;
115  }
116 
117  // Activate the first configuration on the USB device.
118  if (setConfiguration(m_config->configurationValue) != ESUCCESS)
119  {
120  ERROR("failed to activate configurationValue: " << m_config->configurationValue);
121  return EIO;
122  }
123 
124  // Retrieve the full configuration descriptor, which
125  // includes other descriptors (interface and endpoints)
126  // just after the configuration descriptor.
127  Address desc = (Address) new u8[m_config->totalLength];
128  if (!desc)
129  {
130  ERROR("failed to allocate descriptors buffer");
131  return EIO;
132  }
133  if (getConfigDescriptor((USBDescriptor::Configuration *) desc, m_config->totalLength) != ESUCCESS)
134  {
135  ERROR("failed to get full configuration descriptors");
136  return EIO;
137  }
138 
139  // Parse all descriptors received
140  for (Size offset = 0; offset < m_config->totalLength;)
141  {
142  USBDescriptor::Configuration *conf = (USBDescriptor::Configuration *) (desc + offset);
143  switch (conf->descriptorType)
144  {
145  case USBDescriptor::DeviceType:
146  DEBUG("::Device at " << offset);
147  offset += conf->length;
148  break;
149 
150  case USBDescriptor::ConfigurationType:
151  DEBUG("::Configuration at " << offset);
152  offset += conf->length;
153  break;
154 
155  case USBDescriptor::InterfaceType: {
156  DEBUG("::Interface at " << offset);
157  USBDescriptor::Interface *iface = (USBDescriptor::Interface *) (desc + offset);
158  m_interfaces.insert(*iface);
159  offset += iface->length;
160  break;
161  }
162  case USBDescriptor::EndpointType: {
163  USBDescriptor::Endpoint *ep = (USBDescriptor::Endpoint *) (desc + offset);
164  DEBUG("::Endpoint at " << offset << " addr = " << (ep->endpointAddress & 0xf) <<
165  " dir = " << (ep->endpointAddress >> 7) << " attr = " << ((ep->attributes) & 0x3));
166  m_endpoints.insert(*ep);
167  offset += ep->length;
168  break;
169  }
170  case USBDescriptor::StringType: {
171  DEBUG("::String at " << offset);
172  USBDescriptor::String *str = (USBDescriptor::String *) (desc + offset);
173  m_strings.insert(*str);
174  offset += str->length;
175  break;
176  }
177  case USBDescriptor::HubType: {
178  DEBUG("::Hub at " << offset);
179  offset += conf->length;
180  break;
181  }
182  default: {
183  DEBUG("unknown descriptor at " << offset);
184  offset += conf->length;
185  break;
186  }
187  }
188  }
189  return ESUCCESS;
190 }

268 Error USBDevice::controlMessage(u8 request,
269  const USBTransfer::Direction direction,
270  const USBTransfer::RequestType type,
271  const USBTransfer::Recipient recipient,
272  u16 value,
273  u16 index,
274  void *buffer,
275  Size size)
276 {
277  DEBUG("");
278 
279  USBMessage msg;
280  msg.direction = direction;
281  msg.speed = m_speed;
282  msg.type = USBTransfer::Control;
283  msg.state = USBMessage::Setup;
284  msg.maxPacketSize = !m_device->maxPacketSize ? 8 : m_device->maxPacketSize;
285  msg.hubAddress = 1;
286  msg.portAddress = 1;
287 
288  msg.setup.requestType = (direction << 7) | (type << 5) | recipient;
289  msg.setup.request = request;
290  msg.setup.value = value;
291  msg.setup.index = index;
292  msg.setup.length = size;
293 
294  msg.deviceId = m_id;
295  msg.endpointId = 0;
296  msg.size = size;
297 
298  // Use the physical address of the buffer
299  if (buffer)
300  {
301  Memory::Range range;
302  range.virt = (Address) buffer;
303  VMCtl(SELF, LookupVirtual, &range);
304  msg.buffer = range.phys;
305  }
306  else
307  msg.buffer = 0;
308 
309  // Perform control message transfer
310  return submit(msg);
311 }

347 Error USBDevice::beginTransfer(
348  const USBTransfer::Type type,
349  const USBTransfer::Direction direction,
350  Address endpointId,
351  void *buffer,
352  Size size,
353  Size maxPacketSize,
354  CallbackFunction *callback)
355 {
356  DEBUG("");
357 
358  USBMessage *msg = new USBMessage;
359  msg->direction = direction;
360  msg->speed = m_speed;
361  msg->type = type;
362  msg->state = USBMessage::Status;
363 
364  if (maxPacketSize)
365  msg->maxPacketSize = maxPacketSize;
366  else
367  msg->maxPacketSize = !m_device->maxPacketSize ? 8 : m_device->maxPacketSize;
368 
369  msg->deviceId = m_id;
370  msg->endpointId = endpointId & 0xf;
371  msg->size = size;
372  msg->packetId = m_endpointsPacketId.at(msg->endpointId);
373 
374  // Use the physical address of the buffer
375  Memory::Range range;
376  range.virt = (Address) buffer;
377  VMCtl(SELF, LookupVirtual, &range);
378  msg->buffer = range.phys;
379 
380  FileSystemMessage fs;
381  FileDescriptor *files = getFiles();
382 
383  // Write the file
384  if (m_transferFile > 0 && m_transferFile < FILE_DESCRIPTOR_MAX && files[m_transferFile].open)
385  {
386  // Prepare the message. We use the offset field to remember virtual input buffer address
387  FileDescriptor *fd = &files[m_transferFile];
388  fs.type = ChannelMessage::Request;
389  fs.action = WriteFile;
390  fs.path = fd->path;
391  fs.buffer = (char *) msg;
392  fs.size = sizeof(*msg);
393  fs.offset = (Size) buffer;
394  fs.from = SELF;
395  fs.deviceID.minor = fd->identifier;
396  if (ChannelClient::instance->sendRequest(fd->mount, &fs, callback) == ChannelClient::Success)
397  return ESUCCESS;
398  }
399  return EIO;
400 }

416 Error USBDevice::submit(USBMessage & msg)
417 {
418  DEBUG("");
419 
420  // Make an USB transfer by writing the USB transfer file
421  if (::write(m_transferFile, &msg, sizeof(msg)) != sizeof(msg))
422  {
423  ERROR("failed to write USB transfer file: " << strerror(errno));
424  return EIO;
