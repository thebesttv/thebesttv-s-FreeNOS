80 {
81  struct dirent *dent;
82  struct stat st;
83  char tmp[PATH_MAX];
84  String out;
85  DIR *d;
86  Result r = Success;
87 
88  // Retrieve file status
89  if (stat(*path, &st) != 0)
90  {
91  ERROR("failed to stat '" << *path << "': " << strerror(errno));
92  return IOError;
93  }
94 
95  // Is the given file a directory?
96  if (S_ISDIR(st.st_mode))
97  {
98  // Attempt to open the directory
99  if (!(d = opendir(*path)))
100  {
101  ERROR("failed to open '" << *path << "': " << strerror(errno));
102  return IOError;
103  }
104 
105  // Read directory
106  while ((dent = readdir(d)))
107  {
108  // Construct full path
109  snprintf(tmp, sizeof(tmp),
110  "%s/%s", *path, dent->d_name);
111 
112  if ((r = printSingleFile(tmp, out)) != Success)
113  break;
114  }
115  // Close it
116  closedir(d);
117  }
118  // The given file is not a directory
119  else
120  {
121  r = printSingleFile(path, out);
122  }
123 
124  // Final newline
125  if (!arguments().get("long"))
126  out << "\r\n";
127 
128  // Write to standard output
129  write(1, *out, out.length());
130 
131  // Success
132  return r;

169

172  out << BLUE;
173
