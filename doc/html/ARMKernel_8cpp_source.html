27 ARMKernel::ARMKernel(ARMInterrupt *intr,
28  CoreInfo *info)
29  : Kernel(info)
30 #ifdef BCM2836
31  , m_bcm(info->coreId)
32 #endif /* BMC2836 */
33 {
34  ARMControl ctrl;
35 
36  NOTICE("");
37 
38  // Setup interrupt callbacks
39  m_intControl = intr;
40  intr->install(ARMInterrupt::UndefinedInstruction, undefinedInstruction);
41  intr->install(ARMInterrupt::SoftwareInterrupt, trap);
42  intr->install(ARMInterrupt::PrefetchAbort, prefetchAbort);
43  intr->install(ARMInterrupt::DataAbort, dataAbort);
44  intr->install(ARMInterrupt::Reserved, reserved);
45  intr->install(ARMInterrupt::IRQ, interrupt);
46  intr->install(ARMInterrupt::FIQ, interrupt);
47 
48  // Configure clocks and irqs. For BCM2836, only use the generic ARM timer
49  // when running under Qemu. Unfortunately, Qemu dropped support for the
50  // broadcom timer in recent versions. On hardware, use the broadcom timer.
51 #ifdef BCM2836
52  u32 system_frequency = ctrl.read(ARMControl::SystemFrequency);
53  NOTICE("sysfreq = " << system_frequency);
54  if (system_frequency == 62500000)
55  {
56  // Use ARM generic timer
57  m_timer = &m_armTimer;
58  m_timerIrq = GTIMER_PHYS_1_IRQ;
59  m_armTimer.setFrequency(100);
60 
61  // Setup IRQ routing
62  m_bcm.setCoreTimerIrq(Broadcom2836::PhysicalTimer1, true);
63  }
64 #endif /* BCM2836 */
65 
66  /* Default to broadcom timer and interrupt handling */
67  if (m_timer == NULL)
68  {
69  m_timer = &m_bcmTimer;
70  m_timerIrq = BCM_IRQ_SYSTIMERM1;
71  m_bcmTimer.setFrequency( 250 ); /* trigger timer interrupts at 250Hz (clock runs at 1Mhz) */
72  m_intControl->enable(BCM_IRQ_SYSTIMERM1);
73  }
74 
75  // Set ARMCore modes
76  ctrl.set(ARMControl::AlignmentFaults);
77 
78 #ifdef ARMV6
79  ctrl.unset(ARMControl::AlignmentCorrect);
80  ctrl.unset(ARMControl::BigEndian);
81 #endif
82 }

84 void ARMKernel::interrupt(volatile CPUState state)
85 {
86  ARMKernel *kernel = (ARMKernel *) Kernel::instance;
87  ARMInterrupt *intr = (ARMInterrupt *) kernel->m_intControl;
88  ARMProcess *proc = (ARMProcess *) Kernel::instance->getProcessManager()->current(), *next;
89  bool tick;
90 
91  DEBUG("procId = " << proc->getID());
92 
93 #ifdef BCM2836
94  if (kernel->m_timer == &kernel->m_armTimer)
95  {
96  tick = kernel->m_bcm.getCoreTimerIrqStatus(Broadcom2836::PhysicalTimer1);
97  }
98  else
99 #endif /* BCM2836 */
100  {
101  tick = intr->isTriggered(BCM_IRQ_SYSTIMERM1);
102  }
103 
104  if (tick)
105  {
106  kernel->m_timer->tick();
107  next = (ARMProcess *)kernel->getProcessManager()->schedule();
108  if (next)
109  {
110  proc->setCpuState((const CPUState *)&state);
111  MemoryBlock::copy((void *)&state, next->cpuState(), sizeof(state));
112  }
113  }
114 
115  for (uint i = kernel->m_timerIrq + 1; i < 64; i++)
116  {
117  if (intr->isTriggered(i))
118  {
119  kernel->executeIntVector(i, (CPUState *)&state);
120  }
121  }
122 
123 }

158 void ARMKernel::trap(volatile CPUState state)
159 {
160  ProcessManager *mgr = Kernel::instance->getProcessManager();
161  ARMProcess *proc = (ARMProcess *) mgr->current(), *proc2;
162  ProcessID procId = proc->getID();
163 
164  DEBUG("procId = " << procId << " api = " << state.r0);
165 
166  // Execute the kernel call
167  u32 r = Kernel::instance->getAPI()->invoke(
168  (API::Number) state.r0,
169  state.r1,
170  state.r2,
171  state.r3,
172  state.r4,
173  state.r5
174  );
175 
176  // Did we change process?
177  proc2 = (ARMProcess *) mgr->current();
178  DEBUG("result = " << r << " scheduled = " << (bool)(proc != proc2));
179 
180  if (proc != proc2)
181  {
182  // Only if the previous process still exists (not killed in API)
183  if (mgr->get(procId) != NULL)
184  {
185  state.r0 = 0;
186  proc->setCpuState((const CPUState *)&state);
187  }
188  MemoryBlock::copy((void*)&state, proc2->cpuState(), sizeof(state));
189  }
190  else
191  state.r0 = r;
192 }
