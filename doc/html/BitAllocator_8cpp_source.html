60 Allocator::Result BitAllocator::allocate(Size *size,
61  Address *addr,
62  Size align,
63  Address allocStart)
64 {
65  Size num = (*size) / m_chunkSize;
66  BitArray::Result result;
67  Size bit;
68 
69  if ((*size) % m_chunkSize)
70  num++;
71 
72  if (!align)
73  align = 1;
74  else if (align % m_chunkSize)
75  return InvalidAlignment;
76  else
77  align /= m_chunkSize;
78 
79  result = m_array.setNext(&bit, num, allocStart / m_chunkSize, align);
80  if (result != BitArray::Success)
81  return OutOfMemory;
82 
83  *addr = m_base + (bit * m_chunkSize);
84  return Success;
85 }
