133  virtual bool insert(const K & key, const V & value)
134  {
135  assertRead(key);
136  assertRead(value);
137 
138  Size idx = hash(key, m_table.size());
139 
140  // See if the given key exists. Overwrite if so.
141  for (ListIterator<Bucket> i(m_table[idx]); i.hasCurrent(); i++)
142  {
143  if (i.current().key == key)
144  {
145  i.current().value = value;
146  return true;
147  }
148  }
149 
150  // Key does not exist. Append it.
151  m_table[idx].append(Bucket(key, value));
152  m_count++;
153  return true;
