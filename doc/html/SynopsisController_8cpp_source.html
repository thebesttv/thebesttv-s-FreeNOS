32 Error SynopsisController::initialize()
33 {
34  Error r = USBController::initialize();
35 
36  if (r != ESUCCESS)
37  return r;
38 
39  // Map USB host controller registers
40  if (m_io.map(IO_BASE + Base, PAGESIZE*2,
41  Memory::User|Memory::Readable|Memory::Writable|Memory::Device) != IO::Success)
42  {
43  ERROR("failed to map I/O registers");
44  return EIO;
45  }
46 
47  // Check device ID
48  if (m_io.read(VendorId) != DefaultVendorId)
49  {
50  ERROR("incompatible vendorId: " << m_io.read(VendorId) << " != " << DefaultVendorId);
51  return EIO;
52  }
53 
54  DEBUG("UserId: " << m_io.read(UserId) << " VendorId: " << m_io.read(VendorId));
55  NOTICE("Synopsis Design Ware USB-on-the-go Host Controller found");
56 
57  // Initialize power manager
58  if (m_power.initialize() != BroadcomPower::Success)
59  {
60  ERROR("failed to initialize power manager");
61  return EIO;
62  }
63  // Power on the USB subsystem
64  if (m_power.enable(BroadcomPower::USB) != BroadcomPower::Success)
65  {
66  ERROR("failed to power on the USB subsystem");
67  return EIO;
68  }
69  DEBUG("powered on");
70 
71  // Soft-Reset
72  softReset();
73 
74  // Setup DMA
75  m_io.write(RxFIFOSize, 1024);
76  m_io.write(TxFIFOSize, (1024 << 16) | 1024);
77  m_io.write(PeriodTxFIFOSize, (1024 << 16) | 2048);
78  m_io.set(AHBConfig, DMAEnable | AXIWait);
79 
80  // Enable the USB controller interrupt on the ARM's interrupt controller
81  addIRQHandler(InterruptNumber, (IRQHandlerFunction) &SynopsisController::interruptHandler);
82  addIRQHandler(0, (IRQHandlerFunction) &SynopsisController::interruptHandler);
83  ProcessCtl(SELF, WatchIRQ, InterruptNumber);
84  ProcessCtl(SELF, EnableIRQ, InterruptNumber);
85 
86  // Clear all pending core interrupts
87  m_io.write(CoreIntMask, 0);
88  m_io.write(CoreInterrupt, 0xffffffff);
89 
90  // Enable core host channel and port interrupts
91  m_io.write(CoreIntMask, CoreIntChannel | CoreIntPort);
92 
93  // Enable interrupts globally on the USB host controller.
94  m_io.set(AHBConfig, InterruptEnable);
95 
96  // Power-on host port (virtual root hub)
97  u32 val = m_io.read(HostPortControl);
98  val &= ~(HostPortEnable | HostPortEnableChanged |
99  HostPortConnectChanged | HostPortCurrentChanged);
100  val |= HostPortPower;
101  m_io.write(HostPortControl, val);
102 
103  // Begin host port reset (raise the reset signal)
104  val = m_io.read(HostPortControl);
105  val &= ~(HostPortEnable | HostPortEnableChanged |
106  HostPortConnectChanged | HostPortCurrentChanged);
107  val |= HostPortReset;
108  m_io.write(HostPortControl, val);
109 
110  // Wait until the host port becomes available
111  sleep(3);
112 
113  // Finish host port reset (lower the reset signal)
114  val = m_io.read(HostPortControl);
115  val &= ~(HostPortReset | HostPortEnable | HostPortEnableChanged |
116  HostPortConnectChanged | HostPortCurrentChanged);
117  m_io.write(HostPortControl, val);
118 
119  // Initialize channels
120  for (Size i = 0; i < ChannelCount; i++)
121  m_channels[i].initialize();
122 
123  // Done.
124  return ESUCCESS;
125 }

127 void SynopsisController::interruptHandler(Size vector)
128 {
129  DEBUG("coreint =" << m_io.read(CoreInterrupt));
130 
131  u32 coreInt = m_io.read(CoreInterrupt);
132 
133  // Host port interrupt occurred
134  if (coreInt & CoreIntPort)
135  {
136  hostPortChanged();
137  }
138 
139  // Channel interrupt occurred
140  if (coreInt & CoreIntChannel)
141  {
142  u32 channelInt = m_io.read(HostChannelInt);
143 
144  // Post-process channel interrupt for each channel
145  // which has the interrupt flag set high.
146  for (Size i = 0; i < ChannelCount; i++)
147  {
148  if (channelInt & (1 << i))
149  m_channels[i].interrupt();
150  }
151  }
152  // Re-enable IRQ in the kernel
153  ProcessCtl(SELF, EnableIRQ, InterruptNumber);
154 
155  // Post-process in libfs
156  return DeviceServer::interruptHandler(vector);
157 }

159 void SynopsisController::hostPortChanged()
160 {
161  u32 portStatus = m_io.read(HostPortControl);
162 
163  NOTICE("connected = " << (int)(portStatus & HostPortConnect) <<
164  " connectedChanged = " << (int)(portStatus & HostPortConnectChanged) <<
165  " enabled = " << (int)(portStatus & HostPortEnable) <<
166  " enabledChanged = " << (int)(portStatus & HostPortEnableChanged) <<
167  " overcurrent = " << (int)(portStatus & HostPortCurrent) <<
168  " overcurrentChanged = " << (int)(portStatus & HostPortCurrentChanged) <<
169  " reset = " << (int)(portStatus & HostPortReset) <<
170  " power = " << (int)(portStatus & HostPortPower) <<
171  " speed = " << (int)(portStatus & HostPortSpeed));
172 
173  // Clear host port interrupt flags
174  portStatus &= ~(HostPortEnable | HostPortReset);
175  portStatus |= HostPortEnableChanged | HostPortConnectChanged | HostPortCurrentChanged;
176  m_io.write(HostPortControl, portStatus);
177 }

179 void SynopsisController::softReset()
180 {
181  DEBUG("");
182 
183  // Raise reset flag
184  m_io.write(CoreReset, 1);
185 
186  // Wait until cleared
187  while (m_io.read(CoreReset) & 1);
188 
189  // Wait for some time to give the hardware
190  // enough time to fully initialize. Without this wait,
191  // the hardware does not process transfers / interrupts properly.
192  sleep(3);
193 }

195 SynopsisChannel * SynopsisController::getChannel(const FileSystemMessage *msg,
196  USBMessage *usb)
197 {
198  // Check if the message is already in process inside a channel.
199  for (Size i = 0; i < ChannelCount; i++)
200  {
201  SynopsisChannel *ch = &m_channels[i];
202 
203  if (ch->getState() != SynopsisChannel::Idle &&
204  ch->getMessage() == msg)
205  return ch;
206  }
207  // Message is not in process. Try to find unused channel.
208  for (Size i = 0; i < ChannelCount; i++)
209  {
210  SynopsisChannel *ch = &m_channels[i];
211 
212  if (ch->getState() == SynopsisChannel::Idle)
213  return ch;
214  }
215  // All channels busy
216  return ZERO;
217 }

219 Error SynopsisController::transfer(const FileSystemMessage *msg,
220  USBMessage *usb)
221 {
222  DEBUG("");
223 
224  SynopsisChannel *ch = getChannel(msg, usb);
225  if (!ch)
226  return EAGAIN;
227 
228  switch (usb->state)
229  {
230  case USBMessage::Setup:
231  case USBMessage::Data:
232  case USBMessage::Status:
233 
234  switch (ch->transfer(msg, usb))
235  {
236  case SynopsisChannel::TransferStarted: return EAGAIN;
237  case SynopsisChannel::Success: return ESUCCESS;
238  default: return EIO;
239  }
240  break;
241 
242  case USBMessage::Success:
243  return ESUCCESS;
244 
245  case USBMessage::Failure:
246  return EIO;
247 
248  default:
249  ERROR("unhandled USBMessage.state = " << (int) usb->state);
250  break;
251  }
252  return ENOTSUP;
253 }
