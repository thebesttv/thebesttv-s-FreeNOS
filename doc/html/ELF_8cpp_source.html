37 ELF::Result ELF::detect(const u8 *image, Size size, ExecutableFormat **fmt)
38 {
39  ELFHeader *header = (ELFHeader *) image;
40 
41  // Verify ELF magic bytes
42  if (header->ident[ELF_INDEX_MAGIC0] == ELF_MAGIC0 &&
43  header->ident[ELF_INDEX_MAGIC1] == ELF_MAGIC1 &&
44  header->ident[ELF_INDEX_MAGIC2] == ELF_MAGIC2 &&
45  header->ident[ELF_INDEX_MAGIC3] == ELF_MAGIC3)
46  {
47  // Only accept current version, 32-bit ELF executable programs
48  if (header->ident[ELF_INDEX_CLASS] == ELF_CLASS_32 &&
49  header->version == ELF_VERSION_CURRENT &&
50  header->type == ELF_TYPE_EXEC)
51  {
52  (*fmt) = new ELF(image, size);
53  return Success;
54  }
55  }
56  return InvalidFormat;
57 }

59 ELF::Result ELF::regions(ELF::Region *regions, Size *count) const
60 {
61  ELFSegment *segments;
62  ELFHeader *header = (ELFHeader *) m_image;
63  Size max = *count, num = header->programHeaderEntryCount, c = 0;
64 
65  // Must be of the same sizes
66  if (!(header->programHeaderEntrySize == sizeof(ELFSegment) &&
67  header->programHeaderEntryCount < 16))
68  {
69  return InvalidFormat;
70  }
71 
72  // Point to the program segments
73  segments = (ELFSegment *) (m_image + header->programHeaderOffset);
74  (*count) = 0;
75 
76  // Fill in the memory regions
77  for (Size i = 0; i < max && i < num; i++)
78  {
79  // We are only interested in loadable segments
80  if (segments[i].type != ELF_SEGMENT_LOAD)
81  continue;
82 
83  regions[c].virt = segments[i].virtualAddress;
84  regions[c].size = segments[i].memorySize;
85  regions[c].access = Memory::User | Memory::Readable | Memory::Writable;
86  regions[c].data = new u8[segments[i].memorySize];
87 
88  // Read segment contents from file
89  MemoryBlock::copy(regions[c].data, m_image + segments[i].offset,
90  segments[i].fileSize);
91 
92  // Nulify remaining space
93  if (segments[i].memorySize > segments[i].fileSize)
94  {
95  memset(regions[c].data + segments[i].fileSize, 0,
96  segments[i].memorySize - segments[i].fileSize);
97  }
98  c++;
99  }
100 
101  // All done
102  (*count) = c;
103  return Success;
104 }
