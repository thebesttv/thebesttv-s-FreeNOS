69 NetReceive::Result NetReceive::receiveArp()
70 {
71  u8 packet[1500];// sizeof(Ethernet::Header) +
72  // sizeof(ARP::Header) ];
73 
74  while (true)
75  {
76  if (receivePacket(packet, sizeof(packet)) != Success)
77  continue;
78 
79  Ethernet::Header *ether = (Ethernet::Header *) packet;
80  ARP::Header *arp = (ARP::Header *) (ether+1);
81 
82  if (be16_to_cpu(ether->type) == Ethernet::ARP)
83  {
84  printf("ARP: hw=%u proto=%u hwlen=%u protolen=%u op=%u\n",
85  be16_to_cpu(arp->hardwareType),
86  be16_to_cpu(arp->protocolType),
87  arp->hardwareLength,
88  arp->protocolLength,
89  be16_to_cpu(arp->operation));
90  printf("ARP: mac src=%x", arp->etherSender.addr[0]);
91 
92  for (int i = 1; i < 6; i++)
93  printf(":%x", arp->etherSender.addr[i]);
94 
95  printf(" mac dest=%x", arp->etherTarget.addr[0]);
96  for (int i = 1; i < 6; i++)
97  printf(":%x", arp->etherTarget.addr[i]);
98 
99  printf("\n");
100 
101  printf("ARP: ip src=%x ip dst=%x\n", be32_to_cpu(arp->ipSender), be32_to_cpu(arp->ipTarget));
102  return Success;
103  }
104  else
105  printf("skipped (not ARP, type %x)\n", be16_to_cpu(ether->type));
106  }
107  return Success;

110 NetReceive::Result NetReceive::receivePacket(u8 *packet, Size size)
111 {
112 
113  DEBUG("");
114 
115  // Receive on physical device
116  const char *device = *arguments().getPositionals()[0]->getValue();
117  int fd = open(device, O_RDWR);
118  int result;
119 
120  if (fd < 0)
121  {
122  printf("failed to open device '%s': %s\n", device, strerror(errno));
123  exit(EXIT_FAILURE);
124  }
125  printf("receiving on device: %s\n", device);
126 
127  if ((result = read(fd, packet, size)) <= 0)
128  {
129  printf("failed to receive packet on device '%s': %s\n",
130  device, strerror(errno));
131  return IOError;
132  }
133  printf("received %d bytes\n", result);
134 
135  for (int i = 0; i < result; i++)

137  printf("\n");
138 
139  // Done
