23 unsigned int sleep(unsigned int seconds)
24 {

26 
27  // Get current kernel timer ticks
28  if (ProcessCtl(SELF, InfoTimer, (Address) &info) != API::Success)
29  {
30  errno = EAGAIN;

32  }

39  {
