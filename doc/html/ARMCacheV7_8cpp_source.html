52 ARMCacheV7::Result ARMCacheV7::cleanInvalidate(ARMCacheV7::Type type)
53 {
54  switch (type)
55  {
56  case Instruction:
57  //
58  // Invalidate all instruction caches to PoU.
59  // Also flushes branch target cache.
60  //
61  dsb();
62  mcr(p15, 0, 0, c7, c5, 0);
63 
64  // Invalidate entire branch predictor array
65  flushBranchPrediction();
66  isb();

68 
69  case Data:
70  dataFlush(true);
71  dsb();
72  break;
73 
74  case Unified:
75  cleanInvalidate(Data);
76  cleanInvalidate(Instruction);
77  break;
78  }
79  return Success;

143 static inline s32 log_2_n_round_up(u32 n)
144 {
145  s32 log2n = -1;
146  u32 temp = n;
147 
148  while (temp) {
149  log2n++;
150  temp >>= 1;
151  }
152 
153  if (n & (n - 1))
154  return log2n + 1; // not power of 2 - round up
155  else
156  return log2n; // power of 2
157 }

160 ARMCacheV7::Result ARMCacheV7::flushLevel(u32 level, bool clean)
161 {
162  u32 ccsidr = readCacheSize(level, 0);
163  int way, set, setway;
164  u32 log2_line_len = ((ccsidr & CCSIDR_LINE_SIZE_MASK) >>
165  CCSIDR_LINE_SIZE_OFFSET) + 2;
166 
167  // Converting from words to bytes
168  log2_line_len += 2;
169 
170  u32 num_ways = ((ccsidr & CCSIDR_ASSOCIATIVITY_MASK) >>
171  CCSIDR_ASSOCIATIVITY_OFFSET) + 1;
172  u32 num_sets = ((ccsidr & CCSIDR_NUM_SETS_MASK) >>
173  CCSIDR_NUM_SETS_OFFSET) + 1;
174  //
175  // According to ARMv7 ARM number of sets and number of ways need
176  // not be a power of 2
177  //
178  u32 log2_num_ways = ::log_2_n_round_up(num_ways);
179  u32 way_shift = (32 - log2_num_ways);
180 
181  // Invoke the Clean & Invalidate cache line by set/way on the CP15.
182  for (way = num_ways - 1; way >= 0 ; way--)
183  {
184  for (set = num_sets - 1; set >= 0; set--)
185  {
186  setway = (level << 1) | (set << log2_line_len) |
187  (way << way_shift);
188  //
189  // Clean & Invalidate data/unified
190  // cache line by set/way
191  //
192  if (clean)
193  {
194  asm volatile (" mcr p15, 0, %0, c7, c14, 2"
195  : : "r" (setway));
196  }
197  else
198  {
199  asm volatile (" mcr p15, 0, %0, c7, c6, 2"
200  : : "r" (setway));
201  }
202  }
203  }
204  // Data Synchronisation Barrier to ensure operations are complete
205  dsb();
206  return Success;
207 }

209 ARMCacheV7::Result ARMCacheV7::dataFlush(bool clean)
210 {
211  u32 levelId = getCacheLevelId();
212  u32 cacheType, startBit = 0;
213 
214  for (u32 level = 0; level < 7; level++)
215  {
216  cacheType = (levelId >> startBit) & 7;
217  if (cacheType == CacheLevelData ||
218  cacheType == CacheLevelInstruction ||
219  cacheType == CacheLevelUnified)
220  {
221  flushLevel(level, clean);
222  }
223  startBit += 3;
224  }
225  return Success;
226 }
