33 Error LinnFile::read(IOBuffer & buffer, Size size, Size offset)
34 {
35  LinnSuperBlock *sb;
36  Size bytes = 0, blockNr = 0;
37  u64 storageOffset, copyOffset = offset;
38  u8 *block;
39  Size total = 0;
40  Error e;
41 
42  // Initialize variables.
43  sb = fs->getSuperBlock();
44  block = new u8[sb->blockSize];
45 
46  // Skip ahead blocks.
47  while ((sb->blockSize * (blockNr + 1)) <= copyOffset)
48  {
49  blockNr++;
50  }
51  // Adjust the copy offset within this block.
52  copyOffset -= sb->blockSize * blockNr;
53 
54  // Loop all blocks.
55  while (blockNr < LINN_INODE_NUM_BLOCKS(sb, inode) &&
56  total < size && inode->size - (offset + total) > 0)
57  {
58  // Calculate the offset in storage for this block.
59  storageOffset = fs->getOffset(inode, blockNr);
60 
61  // Fetch the next block.
62  if (fs->getStorage()->read(storageOffset, block, sb->blockSize) < 0)
63  {
64  delete block;
65  return EIO;
66  }
67  // Calculate the number of bytes to copy.
68  bytes = sb->blockSize - copyOffset;
69 
70  // Respect the inode size.
71  if (bytes > inode->size - (offset + total))
72  {
73  bytes = inode->size - (offset + total);
74  }
75  // Respect the remote process buffer.
76  if (bytes > size - total)
77  {
78  bytes = size - total;
79  }
80  // Copy into the buffer.
81  if ((e = buffer.write(block + copyOffset, bytes, total)) < 0)
82  {
83  delete block;
84  return e;
85  }
86  // Update state.
87  total += bytes;
88  copyOffset = 0;
89  blockNr++;
90  }
91  // Success.
92  delete block;
93  return (Error) total;
94 }
