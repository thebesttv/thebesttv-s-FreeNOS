150 Process::Result Process::raiseEvent(ProcessEvent *event)
151 {
152  // Write the message. Be sure to flush the caches because
153  // the kernel has mapped the channel pages separately in low memory.
154  m_kernelChannel->write(event);
155  m_kernelChannel->flush();
156 
157  // Wakeup the Process, if needed
158  return wakeup();
159 }

161 Process::Result Process::initialize()
162 {
163  Memory::Range range;
164  Address paddr, vaddr;
165  Arch::Cache cache;
166 
167  // Allocate two pages for the kernel event channel
168  if (Kernel::instance->getAllocator()->allocateLow(PAGESIZE*2, &paddr) != Allocator::Success)
169  return OutOfMemory;
170 
171  // Translate to virtual address in kernel low memory
172  vaddr = (Address) Kernel::instance->getAllocator()->toVirtual(paddr);
173  MemoryBlock::set((void *)vaddr, 0, PAGESIZE*2);
174  cache.cleanData(vaddr);
175  cache.cleanData(vaddr + PAGESIZE);
176 
177  // Map data and feedback pages in userspace
178  range.phys = paddr;
179  range.access = Memory::User | Memory::Readable;
180  range.size = PAGESIZE * 2;
181  m_memoryContext->findFree(range.size, MemoryMap::UserPrivate, &range.virt);
182  m_memoryContext->mapRange(&range);
183 
184  // Remap the feedback page with write permissions
185  m_memoryContext->unmap(range.virt + PAGESIZE);
186  m_memoryContext->map(range.virt + PAGESIZE,
187  range.phys + PAGESIZE, Memory::User | Memory::Readable | Memory::Writable);
188 
189  // Create shares entry
190  m_shares.setMemoryContext(m_memoryContext);
191  m_shares.createShare(KERNEL_PID, Kernel::instance->getCoreInfo()->coreId, 0, range.virt, range.size);
192 
193  // Setup the kernel event channel
194  m_kernelChannel->setMode(Channel::Producer);
195  m_kernelChannel->setMessageSize(sizeof(ProcessEvent));
196  m_kernelChannel->setVirtual(vaddr, vaddr + PAGESIZE);
197 
198  return Success;
199 }

201 Process::Result Process::wakeup()
202 {
203  m_wakeups++;
204 
205  if (m_state != Running)
206  m_state = Ready;
207 
208  MemoryBlock::set(&m_sleepTimer, 0, sizeof(m_sleepTimer));

212 Process::Result Process::sleep(Timer::Info *timer)
213 {
214  if (!m_wakeups)
215  {
216  m_state = Sleeping;
217 
218  if (timer)
219  MemoryBlock::copy(&m_sleepTimer, timer, sizeof(m_sleepTimer));
220 
221  return Success;
222  }
223  m_wakeups = 0;
224  return WakeupPending;
