33 {
34  ProcessManager *procs = Kernel::instance->getProcessManager();
35  List<ProcessID> pids;
36 
37  // Cleanup members
38  delete m_kernelChannel;
39 
40  // Make a list of unique process IDs which
41  // have a share with this Process
42  Size size = m_shares.size();
43  for (Size i = 0; i < size; i++)
44  {
45  MemoryShare *sh = (MemoryShare *) m_shares.get(i);

116 ProcessShares::Result ProcessShares::createShare(ProcessShares & instance,
117  ProcessShares::MemoryShare *share)
118 {
119  MemoryShare *localShare = ZERO;
120  MemoryShare *remoteShare = ZERO;
121  MemoryContext *localMem = m_memory;
122  MemoryContext *remoteMem = instance.getMemoryContext();
123  Address paddr, vaddr;
124  Arch::Cache cache;
125 
126  if (share->range.size == 0)
127  return InvalidArgument;
128 
129  // Check if the share already exists
130  if (readShare(share) == Success)
131  return AlreadyExists;
132 
133  // Allocate local
134  localShare = new MemoryShare;
135  if (!localShare)
136  return OutOfMemory;
137 
138  // Allocate remote
139  remoteShare = new MemoryShare;
140  if (!remoteShare)
141  {
142  delete localShare;
143  return OutOfMemory;
144  }
145  // Allocate actual pages
146  if (Kernel::instance->getAllocator()->allocateLow(share->range.size, &paddr) != Allocator::Success)
147  return OutOfMemory;
148 
149  // Zero out the pages
150  vaddr = (Address) Kernel::instance->getAllocator()->toVirtual(paddr);
151  MemoryBlock::set((void *)vaddr, 0, share->range.size);
152  for (Size i = 0; i < share->range.size; i+=PAGESIZE)
153  cache.cleanData(vaddr + i);
154 
155  // Fill the local share object
156  localShare->pid = instance.getProcessID();
157  localShare->coreId = Kernel::instance->getCoreInfo()->coreId;
158  localShare->tagId = share->tagId;
159  localShare->range.phys = paddr;
160  localShare->range.size = share->range.size;
161  localShare->range.access = Memory::User | share->range.access;
162  localShare->attached = true;
163 
164  // Map in the local process
165  if (localMem->findFree(localShare->range.size, MemoryMap::UserShare, &localShare->range.virt) != MemoryContext::Success ||
166  localMem->mapRange(&localShare->range) != MemoryContext::Success)
167  {
168  delete localShare;
169  delete remoteShare;
170  return OutOfMemory;
171  }
172  // Fill the remote share object
173  remoteShare->pid = m_pid;
174  remoteShare->coreId = localShare->coreId;
175  remoteShare->tagId = localShare->tagId;
176  remoteShare->range.phys = localShare->range.phys;
177  remoteShare->range.size = localShare->range.size;
178  remoteShare->range.access = localShare->range.access;
179  remoteShare->attached = true;
180 
181  // Map in the remote process
182  if (remoteMem->findFree(remoteShare->range.size, MemoryMap::UserShare, &remoteShare->range.virt) != MemoryContext::Success ||
183  remoteMem->mapRange(&remoteShare->range) != MemoryContext::Success)
184  {
185  delete localShare;
186  delete remoteShare;
187  return OutOfMemory;
188  }
189  // insert into shares list
190  m_shares.insert(*localShare);
191  instance.m_shares.insert(*remoteShare);
192 
193  // raise event on the remote process
194  Process *proc = Kernel::instance->getProcessManager()->get(instance.getProcessID());
195  ProcessEvent event;
196  event.type = ShareCreated;
197  event.number = m_pid;
198  MemoryBlock::copy(&event.share, remoteShare, sizeof(*remoteShare));
199  proc->raiseEvent(&event);
200 
201  // Update parameter outputs
202  MemoryBlock::copy(share, localShare, sizeof(*share));
203  return Success;
204 }

224 ProcessShares::Result ProcessShares::releaseShare(MemoryShare *s, Size idx)
225 {
226  // Only release physical memory if both processes have detached.
227  // Note that in case all memory shares for a certain ProcessID have
228  // been detached but not yet released, and due to a very unlikely race
229  // condition (ProcessID reuse) a new memory share was just created (before old ones were released)
230  // the new memory share would also be detached here, resulting in a memory
231  // share with is detached in this process but attached and useless in the
232  // other process.
233  if (s->attached)
234  {
235  Process *proc = Kernel::instance->getProcessManager()->get(s->pid);
236  if (proc)
237  {
238  ProcessShares & shares = proc->getShares();
239  Size size = shares.m_shares.size();
240 
241  // Mark all process shares detached in the other process
242  for (Size i = 0; i < size; i++)
243  {
244  MemoryShare *otherShare = (MemoryShare *) shares.m_shares.get(i);
245  if (otherShare && otherShare->pid == m_pid
246  && otherShare->coreId == s->coreId)
247  {
248  otherShare->attached = false;
249  }
250  }
251  }
252  }
253  else
254  {
255  // Only release physical memory pages if the other
256  // process already detached earlier
257  Allocator *alloc = Kernel::instance->getAllocator();
258 
259  for (Size i = 0; i < s->range.size; i += PAGESIZE)
260  alloc->release(s->range.phys + i);
261  }
262  // Unmap the share
263  m_memory->unmapRange(&s->range);
264 
265  // Remove share from our administration
266  m_shares.remove(idx);
267  delete s;
268  return Success;
269 }

272 {

275 
276  for (Size i = 0; i < size; i++)
277  {

279  {
280  if (s->pid == share->pid &&
281  s->coreId == share->coreId &&
282  s->tagId == share->tagId)
283  {
284  MemoryBlock::copy(share, s, sizeof(MemoryShare));
285  return Success;
286  }
