50 IntelMP::Result IntelMP::discover()
51 {
52  MPConfig *mpc = 0;
53  MPEntry *entry;
54 
55  // Clear previous discoveries
56  m_cores.clear();
57 
58  // Try to find MPTable in the BIOS memory.
59  mpc = scanMemory(m_bios.getBase());
60 
61  // Retry in the last 1MB of physical memory if not found.
62  if (!mpc)
63  {
64  mpc = scanMemory(m_lastMemory.getBase());
65  if (!mpc)
66  {
67  ERROR("MP header not found");
68  return NotFound;
69  }
70  }
71 
72  // Found config
73  DEBUG("MP header found at " << (void *) mpc);
74  DEBUG("Local APIC at " << (void *) mpc->apicAddr);
75  entry = (MPEntry *)(mpc + 1);
76 
77  // Search for multiprocessor entries
78  for (uint i = 0; i < mpc->count; i++)
79  entry = parseEntry(entry);
80 
81  return Success;
82 }

84 IntelMP::Result IntelMP::boot(CoreInfo *info)
85 {
86  DEBUG("booting core" << info->coreId << " at " <<
87  (void *) info->memory.phys << " with kernel: " << info->kernelCommand);
88 
89  // Copy 16-bit realmode startup code
90  VMCopy(SELF, API::Write, (Address) bootEntry16, MPEntryAddr, PAGESIZE);
91 
92  // Copy the CoreInfo structure
93  VMCopy(SELF, API::Write, (Address) info, MPInfoAddr, sizeof(*info));
94 
95  // Send inter-processor-interrupt to wakeup the processor
96  if (m_apic.sendStartupIPI(info->coreId, MPEntryAddr) != IntController::Success)
97  return IOError;
98 
99  // Wait until the core raises the 'booted' flag in CoreInfo
100  while (1)
101  {
102  CoreInfo check;
103 
104  VMCopy(SELF, API::Read, (Address) &check, MPInfoAddr, sizeof(check));
105 
106  if (check.booted)
107  break;
108  }
109  return Success;
110 }
