47 Allocator::Result PageAllocator::allocate(Size *size, Address *addr, Size align)
48 {
49  Memory::Range range;
50 
51  // Set return address
52  *addr = m_base + m_allocated;
53 
54  Size bytes = *size > PAGEALLOC_MINIMUM ?
55  *size : PAGEALLOC_MINIMUM;
56 
57  // Align to pagesize
58  bytes = aligned(bytes, PAGESIZE);
59 
60  // Fill in the message
61  range.size = bytes;
62  range.access = Memory::User | Memory::Readable | Memory::Writable;
63  range.virt = m_base + m_allocated;
64  range.phys = ZERO;
65  VMCtl(SELF, Map, &range);
66 
67  // Clear the pages
68  MemoryBlock::set((void *) range.virt, 0, range.size);
69 
70  // Update count
71  m_allocated += range.size;
72 
73  // Success
74  *size = range.size;
75  return Success;
76 }
