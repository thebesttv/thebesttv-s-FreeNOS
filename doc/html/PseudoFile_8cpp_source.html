75 Error PseudoFile::write(IOBuffer & buffer, Size size, Size offset)
76 {
77  // Check for the buffer size
78  if (!m_buffer || m_size < (size + offset))
79  {
80  // Allocate a new buffer and copy the old data
81  char *new_buffer = new char[size+offset];
82  memset(new_buffer, 0, sizeof(size+offset));
83 
84  // Inherit from the old buffer, if needed
85  if (m_buffer)
86  {
87  memcpy(new_buffer, m_buffer, m_size);
88  delete m_buffer;
89  }
90  // Assign buffer
91  m_buffer = new_buffer;
92  m_size = size+offset;
93  }
94  // Copy the input data in the current buffer
