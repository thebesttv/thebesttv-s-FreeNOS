40 Error ATAController::initialize()
41 {
42  ATADrive *drive;
43 
44  // Detect ATA Controller
45  if (ReadByte(ATA_BASE_CMD0 + ATA_REG_STATUS) == 0xff)
46  {
47  exit(EXIT_FAILURE);
48  }
49  pollReady(true);
50 
51  // Attempt to detect first drive
52  WriteByte(ATA_BASE_CMD0 + ATA_REG_SELECT, ATA_SEL_MASTER);
53  pollReady(true);
54  WriteByte(ATA_BASE_CMD0 + ATA_REG_CMD, ATA_CMD_IDENTIFY);
55 
56  switch (ReadByte(ATA_BASE_CMD0 + ATA_REG_STATUS))
57  {
58  case 0:
59  NOTICE("No ATA drive(s) detected");
60  break;
61 
62  default:
63  // Wait until the command completed
64  pollReady();
65 
66  // Allocate a new drive
67  drive = new ATADrive;
68  drives.append(drive);
69 
70  // Read IDENTIFY data
71  for (int i = 0; i < 256; i++)
72  {
73  ((u16 *) &drive->identity)[i] = ReadWord(ATA_BASE_CMD0 + ATA_REG_DATA);
74  }
75 
76  // Fixup ASCII bytes
77  IDENTIFY_TEXT_SWAP(drive->identity.firmware, 8);
78  IDENTIFY_TEXT_SWAP(drive->identity.serial, 20);
79  IDENTIFY_TEXT_SWAP(drive->identity.model, 40);
80 
81  // Print out information
82  NOTICE("ATA drive detected: SERIAL=" << drive->identity.serial <<
83  " FIRMWARE=" << drive->identity.firmware <<
84  " MODEL=" << drive->identity.model <<
85  " MAJOR=" << drive->identity.majorRevision <<
86  " MINOR=" << drive->identity.minorRevision <<
87  " SECTORS=" << drive->identity.sectors28);
88  break;
89  }
90  return ESUCCESS;
91 }

93 Error ATAController::read(IOBuffer & buffer, Size size, Size offset)
94 {
95  u8 sectors = CEIL(size, 512);
96  u16 block[256];
97  u32 lba = offset / 512;
98  Size result = 0;
99 
100  // Verify LBA
101  if (drives.isEmpty() || drives.first()->identity.sectors28 < lba)
102  {
103  return EIO;
104  }
105 
106  // Perform ATA Read Command
107  WriteByte(ATA_BASE_CMD0 + ATA_REG_SELECT, ATA_SEL_MASTER_28);
108  WriteByte(ATA_BASE_CMD0 + ATA_REG_COUNT, sectors);
109  WriteByte(ATA_BASE_CMD0 + ATA_REG_ADDR0, (lba) & 0xff);
110  WriteByte(ATA_BASE_CMD0 + ATA_REG_ADDR1, (lba >> 8) & 0xff);
111  WriteByte(ATA_BASE_CMD0 + ATA_REG_ADDR2, (lba >> 16) & 0xff);
112  WriteByte(ATA_BASE_CMD0 + ATA_REG_CMD, ATA_CMD_READ);
113 
114  // Read out all requested sectors
115  while(result < size)
116  {
117  // Poll the status register
118  pollReady(true);
119 
120  // Read out bytes
121  for (int i = 0; i < 256; i++)
122  {
123  block[i] = ReadWord(ATA_BASE_CMD0 + ATA_REG_DATA);
124  }
125 
126  // Calculate maximum bytes
127  Size bytes = (size - result) < 512 - (offset % 512) ?
128  (size - result) : 512 - (offset % 512);
129 
130  // Copy to buffer
131  buffer.bufferedWrite(((u8 *)block) + (offset % 512), bytes);
132 
133  // Update state
134  result += bytes;
135  offset += bytes;
136  }
137  return result;
138 }
