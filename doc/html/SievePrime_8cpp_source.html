39 void SievePrime::searchSequential(int n, unsigned *map) const
40 {
41  int i, j;
42 
43  // Sequential algorithm
44  // Next is a prime
45  for (i = 2; i < n; i++)
46  {
47  // Prime number?
48  if (map[i])
49  {
50  // Mask off all multiples
51  for (j = i + 1; j < n; j++)
52  {
53  if (!(j % i))
54  map[j] = 0;
55  }
56  }
57  }

61 {
62  uint *map;
63  String output;
64  int n, k = 2, i, last, sqrt_of_n;
65 
66  // Read max number
67  n = atoi(arguments().get("NUMBER"));
68  sqrt_of_n = sqrt(n);
69 
70  // Try to allocate memory
71  if ((map = (uint *) malloc(n * sizeof(uint))) == NULL)
72  {
73  ERROR("malloc failed: " << strerror(errno));
74  return IOError;
75  }
76 
77  // Set to true
78  for (i = 0; i < n; i++)
79  map[i] = 1;
80 
81  // Find all primes below sqrt(n) sequentially
82  searchSequential(sqrt_of_n, map);
83 
84  // Now continue above sqrt(n) sequentially
85  while (k < sqrt_of_n)
86  {
87  // Prime number?
88  if (!map[k])
89  {
90  k++;
91  continue;
92  }
93 
94  // Mark multiples of k in my range
95  i = sqrt_of_n;
96  last = n;
97 
98  while (i < last)
99  {
100  // Do we need to unmark this number? (no prime)
101  if (!(i % k))
102  map[i] = 0;
103 
104  i++;
105  }
106  // Look for the next prime
107  k++;
108  }
109 
110  // Print the result
111  if (arguments().get("stdout"))
112  {
113  for (i = 2; i < n; i++)
114  {
115  if (map[i] == 1)
116  output << " " << i;
117  }
118  // Done
119  output << "\r\n";
120  write(1, *output, output.length());
121  }
122 
123  // Done
124  return Success;
125 }
