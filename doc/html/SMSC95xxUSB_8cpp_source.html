119 void SMSC95xxUSB::readFinished(FileSystemMessage *message)
120 {
121  DEBUG("identifier = " << message->identifier << " result = " << (int)message->result);
122 
123  if (!m_rxPacket)
124  {
125  ERROR("unexpected readFinish with no receive packet buffer");
126  return;
127  }
128 
129  // Extract packet
130  // Offset field contains virtual address of input data buffer
131  u8 *data = (u8 *) message->offset;
132  u32 receiveCmd = data[0] | data[1] << 8 | data[2] << 16 | data[3] << 24;
133  Size frameLength = (receiveCmd & RxCommandFrameLength) >> 16;
134 
135  if (frameLength == 0 || frameLength > m_packetSize)
136  {
137  ERROR("invalid framelength: " << frameLength);
138  }
139  else
140  {
141  DEBUG("packet is " << frameLength << " bytes long");
142 
143  // Publish the packet to our parent
144  m_rxPacket->size = frameLength;
145  m_smsc->process(m_rxPacket, ReceiveCommandSize);
146 
147  // Release the packet buffer
148  m_smsc->getReceiveQueue()->release(m_rxPacket);
149  m_rxPacket = 0;
150  }
151 
152  // Release USB transfer
153  finishTransfer(message);
154 
155  // Restart read transfer
156  readStart();
157 }

212 Error SMSC95xxUSB::transmit(NetworkQueue::Packet *pkt)
213 {
214  DEBUG("size = " << pkt->size);
215 
216  // Check packet size
217  if (pkt->size > m_packetSize - TransmitCommandSize)
218  {
219  ERROR("packet size too large: " << pkt->size);
220  return ERANGE;
221  }
222  // Prepend the physical header to the packet.
223  // Note that the Ethernet/IP and/or other headers
224  // are already in place inside the packet buffer.
225 
226  // Fill command word A
227  u32 tx_cmd_a = (pkt->size - TransmitCommandSize) | TxCommandAFirstSeg | TxCommandALastSeg;
228  pkt->data[0] = (tx_cmd_a >> 0) & 0xff;
229  pkt->data[1] = (tx_cmd_a >> 8) & 0xff;
230  pkt->data[2] = (tx_cmd_a >> 16) & 0xff;
231  pkt->data[3] = (tx_cmd_a >> 24) & 0xff;
232 
233  // Fill command word B
234  u32 tx_cmd_b = (pkt->size - TransmitCommandSize);
235  pkt->data[4] = (tx_cmd_b >> 0) & 0xff;
236  pkt->data[5] = (tx_cmd_b >> 8) & 0xff;
237  pkt->data[6] = (tx_cmd_b >> 16) & 0xff;
238  pkt->data[7] = (tx_cmd_b >> 24) & 0xff;
239 
240  // Submit packet for transmit
241  m_smsc->getTransmitQueue()->push(pkt);
242 
243  // begin write, if not already doing
244  if (!m_txPacket)
245  writeStart();
246 
247  // Packet in transmission
248  return pkt->size;
249 }
