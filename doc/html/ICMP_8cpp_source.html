50 Error ICMP::process(NetworkQueue::Packet *pkt, Size offset)
51 {
52  IPV4::Header *iphdr = (IPV4::Header *) (pkt->data + offset - sizeof(IPV4::Header));
53  Header *hdr = (Header *) (pkt->data + offset);
54  IPV4::Address source = be32_to_cpu(iphdr->source);
55 
56  DEBUG("source = " << (uint)source << " type = " << hdr->type << " code = " << hdr->code << " id = " << hdr->id);
57 
58  switch (hdr->type)
59  {
60  case EchoRequest: {
61  DEBUG("request");
62  hdr->type = EchoReply;
63  hdr->checksum = 0;
64  hdr->checksum = checksum(hdr);
65  return sendPacket(source, hdr);
66  }
67  case EchoReply: {
68  DEBUG("reply");
69 
70  for (Size i = 0; i < m_sockets.size(); i++)
71  {
72  ICMPSocket *s = (ICMPSocket *) m_sockets.get(i);
73  if (s && s->getAddress() == source)
74  s->setReply(hdr);
75  }
76  break;
77  }
78  }
79  return ESUCCESS;
80 }

104 Error ICMP::sendPacket(IPV4::Address ip, ICMP::Header *header)
105 {
106  DEBUG("");
107 
108  NetworkQueue::Packet *pkt;
109  Error r;
110 
111  // Get a fresh IP packet
112  r = m_ipv4->getTransmitPacket(
113  &pkt, ip, IPV4::ICMP, sizeof(Header)
114  );
115  if (r != ESUCCESS)
116  return r;
117 
118  // Fill payload
119  MemoryBlock::copy(pkt->data + pkt->size, header, sizeof(ICMP::Header));
120  pkt->size += sizeof(ICMP::Header);
121 
122  // Transmit the packet
123  return m_device->transmit(pkt);
