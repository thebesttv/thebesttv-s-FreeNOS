29 {

31  ExecutableFormat::Region regions[16];
32  Memory::Range range;
33  Arch::MemoryMap map;
34  uint count = 0;
35  pid_t pid = 0;
36  Size numRegions = 16;
37  int fd;
38  struct stat st;
39  u8 *image;
40  Address entry;
41 
42  // Find program image
43  if (stat(path, &st) != 0)
44  return -1;
45 
46  // Open program image
47  if ((fd = open(path, O_RDONLY)) < 0)
48  return -1;
49 
50  // Read the program image
51  image = new u8[st.st_size];
52  if (read(fd, image, st.st_size) != st.st_size)
53  {
54  delete image;
55  close(fd);
56  return -1;
57  }
58  close(fd);
59 
60  // Attempt to read executable format
61  if (ExecutableFormat::find(image, st.st_size, &fmt) != ExecutableFormat::Success)
62  {
63  delete image;
64  errno = ENOEXEC;
65  return -1;
66  }
67 
68  // Retrieve entry point
69  if (fmt->entry(&entry) != ExecutableFormat::Success)
70  {
71  delete fmt;
72  delete image;
73  errno = ENOEXEC;
74  return -1;
75  }
76 
77  // Create new process
78  pid = ProcessCtl(ANY, Spawn, entry);
79  if (pid == (pid_t) -1)
80  {
81  delete fmt;
82  delete image;
83  errno = EIO;
84  return -1;
85  }
86 
87  // Retrieve memory regions
88  if (fmt->regions(regions, &numRegions) != ExecutableFormat::Success)
89  {
90  delete fmt;
91  delete image;
92  errno = ENOEXEC;
93  return -1;
94  }
95 
96  // Not needed anymore
97  delete fmt;
98  delete image;
99 
100  // Map program regions into virtual memory of the new process
101  for (Size i = 0; i < numRegions; i++)
102  {
103  // Copy executable memory from this region
104  range.virt = regions[i].virt;
105  range.phys = ZERO;
106  range.size = regions[i].size;
107  range.access = Memory::User |
108  Memory::Readable |
109  Memory::Writable |
110  Memory::Executable;
111 
112  // Create mapping first
113  if (VMCtl(pid, Map, &range) != 0)
114  {
115  errno = EFAULT;
116  return -1;
117  }
118  // Copy bytes
119  VMCopy(pid, API::Write, (Address) regions[i].data,
120  regions[i].virt, regions[i].size);
121 
122  // Release buffer
123  delete regions[i].data;
124  }
125 
126  // Create mapping for command-line arguments
127  range = map.range(MemoryMap::UserArgs);
128  range.phys = ZERO;
129  range.access = Memory::User | Memory::Readable | Memory::Writable;
130  VMCtl(pid, Map, &range);
131 
132  // Allocate arguments and current working directory
133  char *arguments = new char[PAGESIZE*2];
134  memset(arguments, 0, PAGESIZE);
135 
136  // Fill in arguments
137  while (argv[count] && count < PAGESIZE / ARGV_SIZE)
138  {
139  strlcpy(arguments + (ARGV_SIZE * count), argv[count], ARGV_SIZE);
140  count++;
141  }
142 
143  // Fill in the current working directory
144  strlcpy(arguments + PAGESIZE, **(getCurrentDirectory()), PATH_MAX);
145 
146  // Copy argc/argv into the new process
147  if ((VMCopy(pid, API::Write, (Address) arguments, range.virt, PAGESIZE * 2)) < 0)
148  {
149  delete[] arguments;
150  errno = EFAULT;
151  ProcessCtl(pid, KillPID);
152  return -1;
153  }
154 
155  // Copy fds into the new process.
156  if ((VMCopy(pid, API::Write, (Address) getFiles(),
157  range.virt + (PAGESIZE * 2), range.size - (PAGESIZE * 2))) < 0)
158  {
159  delete[] arguments;
160  errno = EFAULT;
161  ProcessCtl(pid, KillPID);
162  return -1;
163  }
164 
165  // Let the Child begin execution
166  ProcessCtl(pid, Resume);
167 
168  // Done. Cleanup.
169  delete[] arguments;
170  return pid;
171 }
