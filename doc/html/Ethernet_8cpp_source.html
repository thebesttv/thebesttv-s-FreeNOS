84 NetworkQueue::Packet * Ethernet::getTransmitPacket(const Ethernet::Address *destination,
85  Ethernet::PayloadType type)
86 {
87  NetworkQueue::Packet *pkt = m_device->getTransmitQueue()->get();
88  if (!pkt)
89  return ZERO;
90 
91  Ethernet::Header *ether = (Ethernet::Header *) (pkt->data + pkt->size);
92  MemoryBlock::copy(&ether->source, &m_address, sizeof(Address));
93  MemoryBlock::copy(&ether->destination, destination, sizeof(Address));
94  ether->type = cpu_to_be16(type);
95 
96  pkt->size += sizeof(Ethernet::Header);

100 Error Ethernet::process(NetworkQueue::Packet *pkt, Size offset)
101 {
102  DEBUG("");
103 
104  const Ethernet::Header *ether = (Ethernet::Header *) (pkt->data + offset);
105  u16 type = be16_to_cpu(ether->type);
106 
107  switch (type)
108  {
109  case Ethernet::ARP:
110  if (m_arp)
111  return m_arp->process(pkt, offset + sizeof(Ethernet::Header));
112  break;
113 
114  case Ethernet::IPV4:
115  if (m_ipv4)
116  return m_ipv4->process(pkt, offset + sizeof(Ethernet::Header));
117  break;
118 
119  default:
120  DEBUG("dropped unknown ethernet type: " << (int) ether->type);
121  break;
122  }
123  return EINVAL;
124 }
