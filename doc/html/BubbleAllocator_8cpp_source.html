39 Allocator::Result BubbleAllocator::allocate(Size *sz, Address *addr, Size align)
40 {
41  Size needed = aligned(*sz, MEMALIGN);
42 
43  // Do we still have enough room?
44  if (m_current + needed < m_start + m_size)
45  {
46  m_current += needed;
47  *addr = (Address) (m_current - needed);
48  return Success;
49  }
50  // No more memory available
51  return OutOfMemory;
52 }
