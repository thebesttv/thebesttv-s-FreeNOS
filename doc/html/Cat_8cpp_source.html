72  DEBUG("file = " << file);

76 
77  // Stat the file
78  if (stat(file, &st) != 0)
79  {
80  printf("%s: failed to stat '%s': %s\r\n",

85  // Must be a regular file or device
86  if (!S_ISREG(st.st_mode) && !S_ISCHR(st.st_mode))
87  {

89  return InvalidArgument;
90  }
91 
92  // Clear buffer
93  memset(buf, 0, sizeof(buf));
94

96  if ((fd = open(file, O_RDONLY)) < 0)

99  name, file, strerror(errno));
100  return IOError;
101  }
102 
103  // Read contents
104  while (1)
105  {
106  e = read(fd, buf, sizeof(buf) - 1);
107  switch (e)
108  {
109  // Error occurred
110  case -1:
111  printf("%s: failed to read '%s': %s\r\n",
112  name, file, strerror(errno));
113  close(fd);
114  return IOError;
115 
116  // End of file

118  close(fd);
119  return Success;
