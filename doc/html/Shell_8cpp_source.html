59 Shell::Result Shell::exec()
60 {
61  const Vector<Argument *> & positionals = arguments().getPositionals();
62  FILE *fp;
63  struct stat st;
64  char *contents;
65 
66  // Refresh mount points
67  refreshMounts(0);
68 
69  // Check if shell script was given as argument
70  if (positionals.count() > 0)
71  {
72  // Execute commands in each file
73  for (Size i = 0; i < positionals.count(); i++)
74  {
75  const char *file = *(positionals[i]->getValue());
76 
77  // Query the file size
78  if (stat(file, &st) != 0)
79  {
80  ERROR("failed to stat() `" << file << "': " << strerror(errno));
81  continue;
82  }
83 
84  // Open file
85  if ((fp = fopen(file, "r")) == NULL)
86  {
87  ERROR("failed to fopen() `" << file << "': " << strerror(errno));
88  continue;
89  }
90 
91  // Allocate buffer storage
92  contents = new char[st.st_size + 1];
93 
94  // Read the entire file into memory
95  if (fread(contents, st.st_size, 1, fp) != (size_t) st.st_size)
96  {
97  ERROR("failed to fread() `" << file << "': " << strerror(errno));
98  fclose(fp);
99  continue;
100  }
101  // Null terminate
102  contents[st.st_size] = 0;
103 
104  // Parse it into lines
105  String contentString(contents);
106  List<String> lines = contentString.split('\n');
107 
108  // Execute each command
109  for (ListIterator<String> i(lines); i.hasCurrent(); i++)
110  {
111  executeInput((char *) *i.current());
112  }
113 
114  // Cleanup
115  delete contents;
116  fclose(fp);
117  }
118  }
119  // Run an interactive Shell
120  else
121  {
122  // Show the user where to get help
123  printf( "\r\n"
124  "Entering interactive Shell. Type 'help' for the command list.\r\n"
125  "\r\n");
126 
127  // Begin loop
128  return runInteractive();
129  }
130 
131  // Done
132  return Success;
133 }

161 int Shell::executeInput(char *command)
162 {
163  char *argv[MAX_ARGV];
164  char tmp[128];
165  ShellCommand *cmd;
166  Size argc;
167  int pid, status;
168  bool background;
169 
170  // Valid argument?
171  if (!strlen(command))
172  {
173  return EXIT_SUCCESS;
174  }
175 
176  DEBUG("command = '" << command << "'");
177 
178  // Attempt to extract arguments
179  argc = parse(command, argv, MAX_ARGV, &background);
180 
181  // Ignore comments
182  if (argv[0][0] == '#')
183  return EXIT_SUCCESS;
184 
185  // Do we have a matching ShellCommand?
186  if (!(cmd = getCommand(argv[0])))
187  {
188  // If not, try to execute it as a file directly
189  if ((pid = forkexec(argv[0], (const char **) argv)) != -1)
190  {
191  if (!background)
192  {
193  waitpid(pid, &status, 0);
194  return status;
195  } else
196  return EXIT_SUCCESS;
197  }
198 
199  // Try to find it on the filesystem. (temporary hardcoded PATH)
200  else if (snprintf(tmp, sizeof(tmp), "/bin/%s", argv[0]) &&
201  (pid = forkexec(tmp, (const char **) argv)) != -1)
202  {
203  if (!background)
204  {
205  waitpid(pid, &status, 0);
206  return status;
207  } else
208  return EXIT_SUCCESS;
209  }
210  else
211  {
212  ERROR("forkexec `" << argv[0] << "' failed: " << strerror(errno));
213  }
214  }
215  // Enough arguments given?
216  else if (argc - 1 < cmd->getMinimumParams())
217  {
218  ERROR(cmd->getName() << ": not enough arguments (" << cmd->getMinimumParams() << " required)");
219  }
220  // Execute it
221  else
222  {
223  return cmd->execute(argc - 1, argv + 1);
224  }
225  // Not successful
226  return EXIT_FAILURE;
227 }

229 char * Shell::getInput()
230 {
231  static char line[1024];
232  Size total = 0;
233 
234  // Read a line
235  while (total < sizeof(line) - 1)
236  {
237  // Read a character
238  read(0, line + total, 1);
239 
240  // Process character
241  switch (line[total])
242  {
243  // Enter
244  case '\r':
245  case '\n':
246  printf("\r\n");
247  line[total] = ZERO;
248  return line;
249 
250  // Backspace
251  case '\b':
252  case 0x7F:
253  if (total > 0)
254  {
255  total--;
256  printf("\b \b");
257  }
258  break;
259 
260  default:
261  printf("%c", line[total]);
262  total++;
263  break;
264  }
265  }
266  line[total] = ZERO;
267  return line;
268 }

300 Size Shell::parse(char *cmdline, char **argv, Size maxArgv, bool *background)
301 {
302  Size argc;
303 
304  *background = false;
305 
306  for (argc = 0; argc < maxArgv && *cmdline; argc++)
307  {
308  while (*cmdline && *cmdline == ' ')
309  cmdline++;
310 
311  if (*cmdline == '&')
312  {
313  *background = true;
314  break;
315  }
316 
317  argv[argc] = cmdline;
318 
319  while (*cmdline && *cmdline != ' ')
320  cmdline++;
321 
322  if (*cmdline) *cmdline++ = ZERO;
323  }
324  argv[argc] = ZERO;
