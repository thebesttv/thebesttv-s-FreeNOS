31 Error i8250::initialize()
32 {
33  // 8bit Words, no parity
34  WriteByte(base + LINECONTROL, 3);
35 
36  // Enable interrupts
37  WriteByte(base + IRQCONTROL, 1);
38 
39  // No FIFO
40  WriteByte(base + FIFOCONTROL, 0);
41 
42  // Data Ready, Request to Send
43  WriteByte(base + MODEMCONTROL, 3);
44 
45  // Set baudrate
46  WriteByte(base + LINECONTROL, ReadByte(base + LINECONTROL) | DLAB);
47  WriteByte(base + DIVISORLOW, (11500 / BAUDRATE) & 0xff);
48  WriteByte(base + DIVISORHIGH, (11500 / BAUDRATE) >> 8);
49  WriteByte(base + LINECONTROL, ReadByte(base + LINECONTROL) & ~(DLAB));
50 
51  INFO("i8250 initialized");
52 
53  // Done
54  return ESUCCESS;
55 }

63 Error i8250::read(IOBuffer & buffer, Size size, Size offset)
64 {
65  Size bytes = 0;
66  u8 byte;
67 
68  // Read as much bytes as possible
69  while (ReadByte(base + LINESTATUS) & RXREADY && bytes < size)
70  {
71  byte = ReadByte(base);
