61 int TestRunner::run(void)
62 {
63  // Prepare for testing.
64  List<TestInstance *> *tests = TestSuite::instance->getTests();
65  m_reporter->begin(*tests);
66 
67  // Execute tests. Report per-test stats.
68  for (ListIterator<TestInstance *> i(tests); i.hasCurrent(); i++)
69  {
70  TestInstance *test = i.current();
71  if (!test)
72  break;
73 
74  m_reporter->prepare(*test);
75  TestResult result = test->run();
76  m_reporter->collect(*test, result);
77  }
78  // Finish testing. Report final stats.
