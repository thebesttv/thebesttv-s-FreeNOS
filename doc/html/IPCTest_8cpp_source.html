41 IPCTest::Result IPCTest::exec()
42 {
43  ProcessID pid = 0;
44  ProcessInfo info;
45  Arch::MemoryMap map;
46  IPCTestMessage msg;
47  ChannelClient::Result r;
48  Memory::Range range = map.range(MemoryMap::UserArgs);
49  char cmd[PAGESIZE];
50 
51  // Find the PID of ipctest server. Loop processes
52  for (uint i = 0; i < MAX_PROCS; i++)
53  {
54  // Request kernel's process information
55  if (ProcessCtl(i, InfoPID, (Address) &info) != API::NotFound)
56  {
57  // Get the command
58  VMCopy(i, API::Read, (Address) cmd, range.virt, PAGESIZE);
59 
60  if (strncmp(cmd, IPCTESTSERVER_CMD, strlen(IPCTESTSERVER_CMD)) == 0)
61  {
62  pid = i;
63  break;
64  }
65  }
66  }
67 
68  // Did we find the IPCTestServer process?
69  if (!pid)
70  {
71  ERROR("failed to find PID for ipctest server");
72  return InvalidArgument;
73  }
74  NOTICE("found ipctest server at PID " << pid);
75 
76  // Connect first, to ensure message size matches
77  if ((r = ChannelClient::instance->connect(pid, sizeof(msg))) != ChannelClient::Success)
78  {
79  ERROR("failed to connect to ipctest server: error " << (int)r);
80  return IOError;
81  }
82 
83  // Prepare message
84  msg.action = TestActionB;
85  msg.data = 0x12345678;
86  NOTICE("sending message with data = " << (void *) msg.data);
87 
88  // Send message
89  if ((r = ChannelClient::instance->syncSendTo(&msg, pid)) != ChannelClient::Success)
90  {
91  ERROR("failed to send message to PID " << pid << ": error " << (int) r);
92  return IOError;
93  }
94  NOTICE("send finished. receiving reply");
95 
96  // Receive reply from server
97  if ((r = ChannelClient::instance->syncReceiveFrom(&msg, pid)) != ChannelClient::Success)
98  {
99  ERROR("failed to receive message from PID " << pid << ": error " << (int) r);
100  return IOError;
101  }
102 
103  // Done
104  NOTICE("reply received with data = " << (void *) msg.data);
105  return Success;
106 }
