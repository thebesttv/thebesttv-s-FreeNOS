183 static void
184 teken_input_char(teken_t *t, teken_char_t c)
185 {
186 
187  switch (c) {
188  case '\0':
189  break;
190  case '\a':
191  teken_subr_bell(t);
192  break;
193  case '\b':
194  teken_subr_backspace(t);
195  break;
196  case '\n':
197  case '\x0B':
198  teken_subr_newline(t);
199  break;
200  case '\x0C':
201  teken_subr_newpage(t);
202  break;
203 #if defined(TEKEN_XTERM) && defined(TEKEN_UTF8)
204  case '\x0E':
205  teken_scs_switch(t, 1);
206  break;
207  case '\x0F':
208  teken_scs_switch(t, 0);
209  break;
210 #endif /* TEKEN_XTERM && TEKEN_UTF8 */
211  case '\r':
212  teken_subr_carriage_return(t);
213  break;
214  case '\t':
215  teken_subr_horizontal_tab(t);
216  break;
217  default:
218  t->t_nextstate(t, c);
219  break;
220  }
221 
222  /* Post-processing assertions. */
223  teken_assert(t->t_cursor.tp_row >= t->t_originreg.ts_begin);
224  teken_assert(t->t_cursor.tp_row < t->t_originreg.ts_end);
225  teken_assert(t->t_cursor.tp_row < t->t_winsize.tp_row);
226  teken_assert(t->t_cursor.tp_col < t->t_winsize.tp_col);
227  teken_assert(t->t_saved_cursor.tp_row < t->t_winsize.tp_row);
228  teken_assert(t->t_saved_cursor.tp_col < t->t_winsize.tp_col);
229  teken_assert(t->t_scrollreg.ts_end <= t->t_winsize.tp_row);
230  teken_assert(t->t_scrollreg.ts_begin < t->t_scrollreg.ts_end);
231  /* Origin region has to be window size or the same as scrollreg. */
232  teken_assert((t->t_originreg.ts_begin == t->t_scrollreg.ts_begin &&
233  t->t_originreg.ts_end == t->t_scrollreg.ts_end) ||
234  (t->t_originreg.ts_begin == 0 &&
235  t->t_originreg.ts_end == t->t_winsize.tp_row));
236 }

238 static void
239 teken_input_byte(teken_t *t, unsigned char c)
240 {
241 
242 #ifdef TEKEN_UTF8
243  /*
244  * UTF-8 handling.
245  */
246  if ((c & 0x80) == 0x00) {
247  /* One-byte sequence. */
248  t->t_utf8_left = 0;
249  teken_input_char(t, c);
250  } else if ((c & 0xe0) == 0xc0) {
251  /* Two-byte sequence. */
252  t->t_utf8_left = 1;
253  t->t_utf8_partial = c & 0x1f;
254  } else if ((c & 0xf0) == 0xe0) {
255  /* Three-byte sequence. */
256  t->t_utf8_left = 2;
257  t->t_utf8_partial = c & 0x0f;
258  } else if ((c & 0xf8) == 0xf0) {
259  /* Four-byte sequence. */
260  t->t_utf8_left = 3;
261  t->t_utf8_partial = c & 0x07;
262  } else if ((c & 0xc0) == 0x80) {
263  if (t->t_utf8_left == 0)
264  return;
265  t->t_utf8_left--;
266  t->t_utf8_partial = (t->t_utf8_partial << 6) | (c & 0x3f);
267  if (t->t_utf8_left == 0) {
268  teken_printf("Got UTF-8 char %x\n", t->t_utf8_partial);
269  teken_input_char(t, t->t_utf8_partial);
270  }
271  }
272 #else /* !TEKEN_UTF8 */
273  teken_input_char(t, c);
274 #endif /* TEKEN_UTF8 */
275 }

349 static int
350 teken_state_numbers(teken_t *t, teken_char_t c)
351 {
352 
353  teken_assert(t->t_curnum < T_NUMSIZE);
354 
355  if (c >= '0' && c <= '9') {
356  /*
357  * Don't do math with the default value of 1 when a
358  * custom number is inserted.
359  */
360  if (t->t_stateflags & TS_FIRSTDIGIT) {
361  t->t_stateflags &= ~TS_FIRSTDIGIT;
362  t->t_nums[t->t_curnum] = 0;
363  } else {
364  t->t_nums[t->t_curnum] *= 10;
365  }
366 
367  t->t_nums[t->t_curnum] += c - '0';
368  return (1);
369  } else if (c == ';') {
370  if (t->t_stateflags & TS_FIRSTDIGIT)
371  t->t_nums[t->t_curnum] = 0;
372 
373  /* Only allow a limited set of arguments. */
374  if (++t->t_curnum == T_NUMSIZE) {
375  teken_state_switch(t, teken_state_init);
376  return (1);
377  }
378 
379  t->t_stateflags |= TS_FIRSTDIGIT;
380  return (1);
381  } else {
382  if (t->t_stateflags & TS_FIRSTDIGIT && t->t_curnum > 0) {
383  /* Finish off the last empty argument. */
384  t->t_nums[t->t_curnum] = 0;
385  t->t_curnum++;
386  } else if ((t->t_stateflags & TS_FIRSTDIGIT) == 0) {
387  /* Also count the last argument. */
388  t->t_curnum++;
389  }
390  }
391 
392  return (0);
393 }
