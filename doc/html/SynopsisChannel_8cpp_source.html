56 SynopsisChannel::Result SynopsisChannel::interrupt()
57 {
58  u32 val = m_io->read(m_base + ChannelInterrupt);
59 
60  DEBUG("chanid =" << m_id << " intr = " << val);
61 
62  // Print interrupt flags
63  if (val & StallResponse) { ERROR("stalled"); m_usb->state = USBMessage::Failure; }
64  if (val & NAKResponse) { ERROR("no acknowledge"); m_usb->state = USBMessage::Failure; }
65  if (val & NYETResponse) { ERROR("NYET"); m_usb->state = USBMessage::Failure; }
66  if (val & TransferError) { ERROR("txerror"); m_usb->state = USBMessage::Failure; }
67  if (val & BusError) { ERROR("bus error"); m_usb->state = USBMessage::Failure; }
68  if (val & FrameOverrun) { ERROR("frameoverrun"); m_usb->state = USBMessage::Failure; }
69  if (val & DataToggleError) { ERROR("datatoggle"); m_usb->state = USBMessage::Failure; }
70  if (val & BufferNotReady) { ERROR("bufnotready"); m_usb->state = USBMessage::Failure; }
71  if (val & TransferExcess) { ERROR("txexcess"); m_usb->state = USBMessage::Failure; }
72  if (val & FrameListRolover) { ERROR("framelist"); m_usb->state = USBMessage::Failure; }
73 
74  // Clear and mask all interrupts
75  m_io->write(m_base + ChannelMask, 0);
76  m_io->write(m_base + ChannelInterrupt, 0);
77 
78  // Post-process interrupt
79  switch (m_usb->state)
80  {
81  case USBMessage::Setup:
82  m_state = Active;
83  if (m_usb->size)
84  m_usb->state = USBMessage::Data;
85  else
86  m_usb->state = USBMessage::Status;
87  break;
88 
89  case USBMessage::Data:
90  m_state = Active;
91  m_usb->state = USBMessage::Status;
92  break;
93 
94  case USBMessage::Status:
95  m_usb->state = USBMessage::Success;
96 
97  // Only for IN transfers update the size field
98  // which represents the actual number of bytes transferred.
99  if (m_usb->direction == USBTransfer::In)
100  {
101  m_usb->size -= (m_io->read(m_base + ChannelTransfer) & 0x7ffff);
102  DEBUG("got " << m_usb->size << " bytes from IN transfer");
103  }
104  // Save the next packet id;
105  m_usb->packetId = (u8) ((m_io->read(m_base + ChannelTransfer) >> 29) & 0x3);
106 
107  // Reset state
108  m_state = Idle;
109  m_usb = ZERO;
110  m_msg = ZERO;
111  break;
112 
113  case USBMessage::Failure:
114  m_state = Idle;
115  break;
116 
117  default:
118  ERROR("unexpected interrupt");
119  break;
120  }
121 
122  // Done
123  return Success;
124 }

126 SynopsisChannel::Result SynopsisChannel::transfer(const FileSystemMessage *msg,
127  USBMessage *usb)
128 {
129  Size size, packetCount, frameNumber;
130  PacketId packetId;
131  USBTransfer::Direction dir;
132  Address data;
133  Memory::Range range;
134 
135  // Dont let channel take a different message if its claimed already
136  if (m_state != Idle && msg != m_msg)
137  {
138  ERROR("refusing transfer: channel is already processing");
139  return Busy;
140  }
141  // Only submit if we are not already transfering
142  if (m_state == Transfer)
143  return TransferStarted;
144 
145  // Check for buffer alignment
146  if (usb->buffer % 4)
147  {
148  ERROR("data address unaligned: " << (void *)usb->buffer);
149  return IOError;
150  }
151 
152  // Update members
153  m_state = Transfer;
154  m_usb = usb;
155  m_msg = msg;
156 
157  if (usb->type == USBTransfer::Control)
158  {
159  switch (usb->state)
160  {
161  case USBMessage::Setup:
162  //DEBUG("id = " << m_id << " setup request = " << (int) m_usb->setup.request);
163  size = sizeof(m_usb->setup);
164  packetId = SetupId;
165  dir = USBTransfer::Out;
166 
167  // Flush the entire L1 cache to ensure setup buffer
168  // is flushed to system memory.
169  VMCtl(SELF, CacheClean, 0);
170 
171  // Lookup physical address of setup buffer
172  range.virt = (Address) &m_usb->setup;
173  VMCtl(SELF, LookupVirtual, &range);
174  data = range.phys;
175  break;
176 
177  case USBMessage::Data:
178  //DEBUG("id = " << m_id << " data buffer =" << (void *) m_usb->buffer << " size = " << usb->size);
179  size = usb->size;
180  packetId = Data1;
181  dir = m_usb->direction;
182  data = m_usb->buffer;
183  break;
184 
185  case USBMessage::Status:
186  //DEBUG("id = " << m_id << " status");
187  size = 0;
188  packetId = Data1;
189  data = ZERO;
190  dir = m_usb->direction == USBTransfer::In ? USBTransfer::Out : USBTransfer::In;
191  break;
192 
193  default:
194  ERROR("unexpected state =" << (int) m_usb->state);
195  return IOError;
196  }
197  }
198  // Non-Control transfer
199  else
200  {
201  size = m_usb->size;
202  data = m_usb->buffer;
203  dir = m_usb->direction;
204 
205  // The packet id is is associated with the endpoint. For non-control
206  // transfers the next packet id is saved from the previous transfer.
207  packetId = (PacketId) m_usb->packetId;
208  }
209 
210  // Calculate packet count
211  packetCount = CEIL(size, m_usb->maxPacketSize);
212  if (!packetCount)
213  packetCount = 1;
214 
215  DEBUG("chanid = " << m_id << " packets = " << packetCount << " maxpacketsize = "
216  << usb->maxPacketSize << " dir = " << (int)dir << " pid = " << (int)packetId);
217 
218  // Program channel registers.
219  m_io->write(m_base + Characteristics, (usb->endpointId << 11) |
220  (dir << 15) |
221  (usb->type << 18) |
222  (usb->maxPacketSize) |
223  (1 << 20) /* packetsPerFrame */ |
224  (usb->deviceId << 22) );
225 
226  // SplitControl is needed for Full & Low speed devices only
227  m_io->write(m_base + SplitControl, 0);
228 
229  // Setup DMA
230  m_io->write(m_base + ChannelDMA, data);
231  m_io->write(m_base + ChannelTransfer,
232  size | (packetId << 29) | (packetCount << 19));
233 
234  // Clear and program interrupts
235  m_io->write(m_base + ChannelMask, 0);
236  m_io->write(m_base + ChannelInterrupt, 0xffffffff);
237  m_io->write(m_base + ChannelMask, ChannelHalted);
238  m_io->set(HostChannelMask, 1 << m_id);
239 
240  // Increase frame number. Set odd flag. Begin transfer.
241  frameNumber = (m_io->read(HostFrameNumber) & 0xffff) + 1;
242  m_io->set(m_base + Characteristics, ((frameNumber & 1) << 29) | ChannelEnable);
243  return TransferStarted;
244 }
