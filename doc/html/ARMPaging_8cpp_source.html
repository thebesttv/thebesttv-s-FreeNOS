27 ARMPaging::ARMPaging(MemoryMap *map, SplitAllocator *alloc)
28  : MemoryContext(map, alloc)
29 {
30  // Allocate page directory from low physical memory.
31  if (alloc->allocateLow(sizeof(ARMFirstTable),
32  &m_firstTableAddr,
33  sizeof(ARMFirstTable)) == Allocator::Success)
34  {
35  m_firstTable = (ARMFirstTable *) alloc->toVirtual(m_firstTableAddr);
36 
37  // Initialize the page directory
38  MemoryBlock::set(m_firstTable, 0, sizeof(ARMFirstTable));
39 
40  // Map the kernel. The kernel has permanently mapped 1GB of
41  // physical memory (i.e. the "low memory" in SplitAllocator). The low
42  // memory starts at its physical base address offset (varies per core).
43  m_firstTable->mapLarge( m_map->range(MemoryMap::KernelData), m_alloc );
44 
45  // Unmap I/O zone
46  for (Size i = 0; i < IO_SIZE; i += MegaByte(1))
47  m_firstTable->unmap(IO_BASE + i, m_alloc);
48 
49  // Map the I/O zone as Device / Uncached memory.
50  Memory::Range io;
51  io.phys = IO_BASE;
52  io.virt = IO_BASE;
53  io.size = IO_SIZE;
54  io.access = Memory::Readable | Memory::Writable | Memory::Device;
55  m_firstTable->mapLarge(io, m_alloc);
56  }
57 }

65 #ifdef ARMV6
66 MemoryContext::Result ARMPaging::enableMMU()
67 {
68  ARMControl ctrl;
69 
70  // Program first level table. Enable L2 cache for page walking.
71  ctrl.write(ARMControl::TranslationTable0, ((u32) m_firstTableAddr | 1));
72  ctrl.write(ARMControl::TranslationTable1, 0);
73  ctrl.write(ARMControl::TranslationTableCtrl, 0);
74 
75  // Set Control flags
76  ctrl.set(ARMControl::DomainClient);
77  ctrl.set(ARMControl::DisablePageColoring);
78  ctrl.set(ARMControl::AccessPermissions);
79  ctrl.set(ARMControl::ExtendedPaging);
80  ctrl.unset(ARMControl::BranchPrediction);
81 
82  // Flush TLB's and caches
83  tlb_flush_all();
84  m_cache.cleanInvalidate(Cache::Unified);
85 
86  // Disable caches.
87  ctrl.unset(ARMControl::InstructionCache);
88  ctrl.unset(ARMControl::DataCache);
89 
90  // Enable the MMU. This re-enables instruction and data cache too.
91  ctrl.set(ARMControl::MMUEnabled);
92  NOTICE("MMUEnabled = " << (ctrl.read(ARMControl::SystemControl) & ARMControl::MMUEnabled));
93  tlb_flush_all();
94 
95  // Reactivate both caches and branch prediction
96  ctrl.set(ARMControl::InstructionCache);
97  ctrl.set(ARMControl::DataCache);
98  ctrl.set(ARMControl::BranchPrediction);
99 
100  return Success;
101 }
102 #elif defined(ARMV7)

104 MemoryContext::Result ARMPaging::enableMMU()
105 {
106  ARMControl ctrl;
107 
108  // Flush TLB's
109  tlb_flush_all();
110  dsb();
111  isb();
112 
113  // Enable branch prediction
114  ctrl.set(ARMControl::BranchPrediction);
115 
116  // Program first level table
117  ctrl.write(ARMControl::TranslationTable0, (((u32) m_firstTableAddr) |
118  (1 << 3) | // outer write-back, write-allocate
119  (1 << 6) // inner write-back, write-allocate
120  ));
121  ctrl.write(ARMControl::TranslationTable1, 0);
122  ctrl.write(ARMControl::TranslationTableCtrl, 0);
123  dsb();
124  isb();
125 
126  // Set as client for all domains
127  ctrl.write(ARMControl::DomainControl, 0x55555555);
128 
129  // Enable the MMU.
130  u32 nControl = ctrl.read(ARMControl::SystemControl);
131 
132  // Raise all caching, MMU and branch prediction flags.
133  nControl |= (1 << 11) | (1 << 2) | (1 << 12) | (1 << 0) | (1 << 5);
134 
135  // Write back to set.
136  ctrl.write(ARMControl::SystemControl, nControl);
137  isb();
138 
139  NOTICE("MMUEnabled = " << (ctrl.read(ARMControl::SystemControl) & ARMControl::MMUEnabled));
140 
141  // Need to enable alignment faults separately of the MMU,
142  // otherwise QEMU will hard reset the CPU
143  ctrl.set(ARMControl::AlignmentFaults);
144 
145  // Flush all
146  tlb_flush_all();
147  dsb();
148  isb();
149  return Success;
150 }
151 #endif /* ARMV7 */

153 MemoryContext::Result ARMPaging::activate()
154 {
155  ARMControl ctrl;
156 
157  // Do we need to (re)enable the MMU?
158  if (!(ctrl.read(ARMControl::SystemControl) & ARMControl::MMUEnabled))
159  {
160  enableMMU();
161  }
162  // MMU already enabled, we only need to change first level table and flush caches.
163  else
164  {
165 #ifdef ARMV6
166  mcr(p15, 0, 0, c7, c5, 0); // flush entire instruction cache
167  mcr(p15, 0, 0, c7, c10, 0); // flush entire data cache
168  mcr(p15, 0, 0, c7, c7, 0); // flush entire cache
169  mcr(p15, 0, 5, c7, c10, 0); // data memory barrier
170  mcr(p15, 0, 4, c7, c10, 0); // memory sync barrier
171 #else
172  m_cache.cleanInvalidate(Cache::Unified);
173 #endif /* ARMV6 */
174 
175  // Switch first page table and re-enable L1 caching
176  ctrl.write(ARMControl::TranslationTable0, (((u32) m_firstTableAddr) |
177  (1 << 3) | /* outer write-back, write-allocate */
178  (1 << 6) /* inner write-back, write-allocate */
179  ));
180 
181  // Flush TLB caches
182  tlb_flush_all();
183 
184  // Synchronize execution stream
185  isb();
186  }
187  // Done. Update currently active context pointer
188  m_current = this;
189  return Success;
190 }

193 {
194  // Modify page tables
195  Result r = m_firstTable->map(virt, phys, acc, m_alloc);
196

206 MemoryContext::Result ARMPaging::unmap(Address virt)
207 {
208  // Clean the given data page in cache
209  if (m_current == this)
210  m_cache.cleanInvalidateAddress(Cache::Data, virt);
211 
212  // Modify page tables
213  Result r = m_firstTable->unmap(virt, m_alloc);
214 
215  // Flush TLB to refresh the mapping
216  if (m_current == this)
217  tlb_invalidate(virt);
218 
219  // Synchronize execution stream
220  isb();
221  return r;
