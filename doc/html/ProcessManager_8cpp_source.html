60 void ProcessManager::remove(Process *proc, uint exitStatus)
61 {
62  if (proc == m_previous)
63  m_previous = ZERO;
64 
65  if (proc == m_idle)
66  m_idle = ZERO;
67 
68  if (proc == m_current)
69  m_current = ZERO;
70 
71  // Wakeup any Processes which are waiting for this Process
72  Size size = m_procs.size();
73 
74  for (Size i = 0; i < size; i++)
75  {
76  if (m_procs[i] != ZERO &&
77  m_procs[i]->getState() == Process::Waiting &&
78  m_procs[i]->getWait() == proc->getID())
79  {
80  m_procs[i]->setState(Process::Ready);
81  m_procs[i]->setWait(exitStatus);
82  }
83  }
84 
85  // Remove process from administration
86  m_procs[proc->getID()] = ZERO;
87 
88  // Free the process memory
89  delete proc;
90 }

92 Process * ProcessManager::schedule(Process *proc)
93 {
94  // If needed, let the scheduler select a new process
95  if (!proc)
96  {
97  proc = m_scheduler->select(&m_procs, m_idle);
98 
99  // If no process ready, let us idle
100  if (!proc)
101  proc = m_idle;
102  }
103 
104  if (!proc)
105  {
106  FATAL("no process found to run!"); for(;;);
107  }
108 
109  // Only execute if its a different process
110  if (proc != m_current)
111  {
112  m_previous = m_current;
113  m_current = proc;
114 
115  if (m_previous && m_previous->getState() == Process::Running)
116  m_previous->setState(Process::Ready);
117 
118  proc->setState(Process::Running);
119  proc->execute(m_previous);
120  return m_current;
121  }
122  return (Process *) NULL;
123 }
