104  Address phys,
105  Memory::Access access)

109  // Check if the address is already mapped

137 MemoryContext::Result ARMSecondTable::access(Address virt, Memory::Access *access) const
138 {
139  u32 entry = m_pages[ TABENTRY(virt) ];
140 
141  if (!(entry & PAGE2_PRESENT))
142  return MemoryContext::InvalidAddress;
143 
144  // Permissions
145  *access = Memory::Readable;
146 
147  if (entry & PAGE2_AP_USER)
148  *access |= Memory::User;
149 
150  if (!(entry & PAGE2_APX))
151  *access |= Memory::Writable;
152 
153  // Caching
154  if (entry & PAGE2_DEVICE_SHARED)
155  *access |= Memory::Device;
156  else if (entry & PAGE2_UNCACHED)
157  *access |= Memory::Uncached;
158  else
159  *access |= Memory::InnerCached | Memory::OuterCached;
160 
161  return MemoryContext::Success;
162 }
