53  FileModes mode, UserID uid, GroupID gid)
54 {
55  LinnGroup *group;
56  LinnInode *inode;
57  BitArray inodeMap(super->inodesPerGroup);
58 
59  // Point to the correct group
60  group = BLOCKPTR(LinnGroup, super->groupsTable);
61  group += inodeNum / super->inodesPerGroup;
62 
63  // Use it to find the inode
64  inode = BLOCKPTR(LinnInode, group->inodeTable);
65  inode += inodeNum % super->inodesPerGroup;
66 
67  // Initialize the inode
68  inode->type = type;
69  inode->mode = mode;
70  inode->uid = uid;
71  inode->gid = gid;
72  inode->size = ZERO;
73  inode->accessTime = ZERO;
74  inode->createTime = time(ZERO);
75  inode->modifyTime = inode->createTime;
76  inode->changeTime = inode->createTime;
77  inode->links = 1;
78 
79  // Update inode BitArray, if needed
80  inodeMap.setArray(BLOCKPTR(u8, group->inodeMap),
81  super->inodesPerGroup);
82  inodeMap.set(inodeNum % super->inodesPerGroup);
83 
84  // Update superblock
85  super->freeInodesCount--;
86  group->freeInodesCount--;
87 
88  // Done

92 le32 LinnCreate::createInode(char *inputFile, struct stat *st)
93 {
94  LinnGroup *group;
95  LinnInode *inode;
96  BitArray inodeMap(super->inodesPerGroup);
97  u32 gn, in;
98 
99  // Loop all available LinnGroups
100  for (gn = 0; gn < LINN_GROUP_COUNT(super); gn++)
101  {
102  // Point to the correct LinnGroup
103  group = BLOCKPTR(LinnGroup, super->groupsTable) + gn;
104 
105  // Does it have any free inodes?
106  if (!group->freeInodesCount)
107  {
108  group = ZERO;
109  continue;
110  }
111  else
112  break;
113  }
114 
115  // Did we find an appropriate group?
116  if (!group)
117  {
118  printf("%s: no free inode available for `%s'\n",
119  prog, inputFile);
120  exit(EXIT_FAILURE);
121  }
122 
123  // Find an empty inode number
124  inodeMap.setArray(BLOCKPTR(u8, group->inodeMap),
125  super->inodesPerGroup);
126  inodeMap.setNext(&in);
127 
128  // Instantiate the inode
129  inode = createInode(in, FILETYPE_FROM_ST(st),
130  FILEMODE_FROM_ST(st), st->st_uid, st->st_gid);
131 
132  // Insert file contents
133  if (S_ISREG(st->st_mode))
134  {
135  insertFile(inputFile, inode, st);
136  }
137 
138  // Debug out
139  if (verbose)
140  {
141  printf("%s inode=%u size=%u\n", inputFile, in, inode->size);
142  }
143  return in;
144 }

146 void LinnCreate::insertIndirect(le32 *ptr, le32 blockNumber,
147  le32 blockValue, Size depth)
148 {
149  Size remain = 1;
150 
151  // Does the block map itself have a block?
152  if (!*ptr)
153  {
154  *ptr = BLOCK(super);
155  }
156 
157  // Point to the block map
158  ptr = BLOCKPTR(u32, *ptr);
159 
160  // Calculate the number of blocks remaining per entry
161  for (Size i = 0; i < depth - 1; i++)
162  {
163  remain *= LINN_SUPER_NUM_PTRS(super);
164  }
165 
166  // More indirection?
167  if (remain == 1)
168  {
169  ptr[blockNumber % LINN_SUPER_NUM_PTRS(super)] = blockValue;
170  }
171  // Traverse indirection
172  else
173  {
174  insertIndirect(&ptr[blockNumber / remain],
175  blockNumber, blockValue, depth - 1);
176  }
177 }

179 void LinnCreate::insertFile(char *inputFile, LinnInode *inode,
180  struct stat *st)
181 {
182  int fd, bytes;
183  le32 blockNr;
184 
185  // Open the local file
186  if ((fd = open(inputFile, O_RDONLY)) < 0)
187  {
188  printf("%s: failed to fopen() `%s': %s\n",
189  prog, inputFile, strerror(errno));
190  exit(EXIT_FAILURE);
191  }
192 
193  // Read blocks from the file
194  while (true)
195  {
196  // Grab a block
197  blockNr = BLOCK(super);
198 
199  // Read a block
200  if ((bytes = read(fd, BLOCKPTR(u8, blockNr), super->blockSize)) < 0)
201  {
202  printf("%s: failed to read() `%s': %s\n",
203  prog, inputFile, strerror(errno));
204  exit(EXIT_FAILURE);
205  }
206 
207  // End of file?
208  if (!bytes) break;
209 
210  // Insert the block (direct)
211  if (LINN_INODE_NUM_BLOCKS(super, inode) <
212  LINN_INODE_DIR_BLOCKS)
213  {
214  inode->block[LINN_INODE_NUM_BLOCKS(super,inode)] = blockNr;
215  }
216  // Insert the block (indirect)
217  else if (LINN_INODE_NUM_BLOCKS(super, inode) <
218  LINN_INODE_DIR_BLOCKS + LINN_SUPER_NUM_PTRS(super))
219  {
220  insertIndirect(&inode->block[LINN_INODE_IND_BLOCKS-1],
221  LINN_INODE_NUM_BLOCKS(super, inode) -
222  LINN_INODE_DIR_BLOCKS, blockNr, 1);
223  }
224  // Insert the block (double indirect)
225  else if (LINN_INODE_NUM_BLOCKS(super, inode) <
226  LINN_INODE_DIR_BLOCKS + (LINN_SUPER_NUM_PTRS(super) *
227  LINN_SUPER_NUM_PTRS(super)))
228  {
229  insertIndirect(&inode->block[LINN_INODE_DIND_BLOCKS-1],
230  LINN_INODE_NUM_BLOCKS(super, inode) -
231  LINN_INODE_DIR_BLOCKS, blockNr, 2);
232  }
233  // Insert the blck (triple indirect)
234  else if (LINN_INODE_NUM_BLOCKS(super, inode) <
235  LINN_INODE_DIR_BLOCKS + (LINN_SUPER_NUM_PTRS(super) *
236  LINN_SUPER_NUM_PTRS(super) *
237  LINN_SUPER_NUM_PTRS(super)))
238  {
239  insertIndirect(&inode->block[LINN_INODE_TIND_BLOCKS-1],
240  LINN_INODE_NUM_BLOCKS(super, inode) -
241  LINN_INODE_DIR_BLOCKS, blockNr, 3);
242  }
243  // Maximum file capacity reached
244  else
245  {
246  printf("%s: maximum file size reached for `%s'\n",
247  prog, inputFile);
248  break;
249  }
250  // Increment size appropriately
251  inode->size += bytes;
252  }
253  // Cleanup
254  close(fd);
255 }

257 void LinnCreate::insertEntry(le32 dirInode, le32 entryInode,
258  const char *name, FileType type)
259 {
260  LinnGroup *group;
261  LinnInode *inode;
262  LinnDirectoryEntry *entry;
263  le32 entryNum, blockNum;
264 
265  // Point to the correct group
266  group = BLOCKPTR(LinnGroup, super->groupsTable);
267  if (dirInode != ZERO)
268  {
269  group += (dirInode / super->inodesPerGroup);
270  }
271  // Fetch inode
272  inode = BLOCKPTR(LinnInode, group->inodeTable) +
273  (dirInode % super->inodesPerGroup);
274 
275  // Calculate entry and block number
276  entryNum = inode->size / sizeof(LinnDirectoryEntry);
277  blockNum = (entryNum * sizeof(LinnDirectoryEntry)) /
278  super->blockSize;
279 
280  // Direct block
281  if (blockNum < LINN_INODE_DIR_BLOCKS)
282  {
283  // Allocate a new block, if needed
284  if (!inode->block[blockNum])
285  {
286  inode->block[blockNum] = BLOCK(super);
287  }
288  // Point to the fresh entry
289  entry = BLOCKPTR(LinnDirectoryEntry, inode->block[blockNum]) +
290  (entryNum % super->blockSize);
291  // Fill it
292  entry->inode = entryInode;
293  entry->type = type;
294  strncpy(entry->name, name, LINN_DIRENT_NAME_LEN);
295  entry->name[LINN_DIRENT_NAME_LEN - 1] = ZERO;
296  }
297  // Indirect block
298  else
299  {
300  printf("%s: indirect blocks not (yet) supported for directories\n",
301  prog);
302  exit(EXIT_FAILURE);
303  }
304  // Increment directory size
305  inode->size += sizeof(LinnDirectoryEntry);
306 }

308 void LinnCreate::insertDirectory(char *inputDir, le32 inodeNum, le32 parentNum)
309 {
310  struct dirent *ent;
311  struct stat st;
312  DIR *dir;
313  char path[255];
314  le32 child;
315  bool skip = false;
316 
317  // Create '.' and '..'
318  insertEntry(inodeNum, inodeNum, ".", DirectoryFile);
319  insertEntry(inodeNum, parentNum, "..", DirectoryFile);
320 
321  // Open the input directory
322  if ((dir = opendir(inputDir)) == NULL)
323  {
324  printf("%s: failed to opendir() `%s': %s\n",
325  prog, inputDir, strerror(errno));
326  exit(EXIT_FAILURE);
327  }
328  // Read all it's entries
329  while ((ent = readdir(dir)))
330  {
331  // Hidden files
332  skip = ent->d_name[0] == '.';
333 
334  // Excluded files
335  for (ListIterator<String *> e(&excludes); e.hasCurrent(); e++)
336  {
337  String dent = (const char *) ent->d_name;
338 
339  if (dent.match(**e.current()))
340  {
341  skip = true;
342  break;
343  }
344  }
345  // Skip file?
346  if (skip) continue;
347 
348  // Construct local path
349  snprintf(path, sizeof(path), "%s/%s",
350  inputDir, ent->d_name);
351 
352  // Stat the file
353  if (stat(path, &st) != 0)
354  {
355  printf("%s: failed to stat() `%s': %s\n",
356  prog, path, strerror(errno));
357  exit(EXIT_FAILURE);
358  }
359  // Create an inode for the child
360  child = createInode(path, &st);
361 
362  // Insert directory entry
363  insertEntry(inodeNum, child, ent->d_name,
364  FILETYPE_FROM_ST(&st));
365 
366  // Traverse down
367  if (S_ISDIR(st.st_mode))
368  {
369  insertDirectory(path, child, inodeNum);
370  }
371  }
372  // All done
373  closedir(dir);
374 }

376 int LinnCreate::create(Size blockSize, Size blockNum, Size inodeNum)
377 {
378  LinnGroup *group;
379  BitArray map(128);
380 
381  assert(image != ZERO);
382  assert(prog != ZERO);
383  assert(blockNum >= 2);
384  assert(inodeNum > 0);
385 
386  // Allocate blocks
387  blocks = new u8[blockSize * blockNum];
388  memset(blocks, 0, blockSize * blockNum);
389 
390  // Create a superblock
391  super = (LinnSuperBlock *) (blocks + LINN_SUPER_OFFSET);
392  super->magic0 = LINN_SUPER_MAGIC0;
393  super->magic1 = LINN_SUPER_MAGIC1;
394  super->majorRevision = LINN_SUPER_MAJOR;
395  super->minorRevision = LINN_SUPER_MINOR;
396  super->state = LINN_SUPER_VALID;
397  super->blockSize = blockSize;
398  super->blocksPerGroup = LINN_CREATE_BLOCKS_PER_GROUP;
399  super->inodesCount = inodeNum;
400  super->blocksCount = blockNum;
401  super->inodesPerGroup = super->inodesCount / LINN_GROUP_COUNT(super);
402  super->freeInodesCount = super->inodesCount;
403  super->freeBlocksCount = blockNum - 3;
404  super->creationTime = time(ZERO);
405  super->mountTime = ZERO;
406  super->mountCount = ZERO;
407  super->lastCheck = ZERO;
408  super->groupsTable = 2;
409 
410  // Allocate LinnGroups
411  for (Size i = 0; i < LINN_GROUP_COUNT(super); i++)
412  {
413  // Point to the correct LinnGroup
414  group = BLOCKPTR(LinnGroup, 2) + i;
415 
416  // Fill the group
417  group->freeBlocksCount = super->blocksPerGroup;
418  group->freeInodesCount = super->inodesPerGroup;
419  group->blockMap = BLOCKS(super, LINN_GROUP_NUM_BLOCKMAP(super));
420  group->inodeMap = BLOCKS(super, LINN_GROUP_NUM_INODEMAP(super));
421  group->inodeTable = BLOCKS(super, LINN_GROUP_NUM_INODETAB(super));
422  }
423 
424  // Create special inodes
425  createInode(LINN_INODE_ROOT, DirectoryFile,
426  OwnerRWX | GroupRX | OtherRX);
427  createInode(LINN_INODE_LOADER, RegularFile,
428  OwnerRWX | GroupRX | OtherRX);
429  createInode(LINN_INODE_BAD, RegularFile,
430  OwnerRW | GroupR | OtherR);
431  createInode(LINN_INODE_JOURNAL, RegularFile,
432  OwnerRW | GroupR | OtherR);
433 
434  // Insert into directory contents, if set
435  if (input)
436  {
437  insertDirectory(input, LINN_INODE_ROOT,
438  LINN_INODE_ROOT);
439  }
440  // Mark blocks used
441  for (le32 block = 0; block < super->freeBlocksCount; block++)
442  {
443  // Point to group
444  group = BLOCKPTR(LinnGroup, super->groupsTable) +
445  (block / super->blocksPerGroup);
446  group->freeBlocksCount--;
447 
448  // Mark the block used
449  map.setArray(BLOCKPTR(u8, group->blockMap),
450  super->blocksPerGroup);
451  map.set(block % super->blocksPerGroup);
452  }
453  // Write the final image
454  return writeImage();
455 }

476  return EXIT_FAILURE;
477  }
478  // All done
479  fclose(fp);
480  return EXIT_SUCCESS;
