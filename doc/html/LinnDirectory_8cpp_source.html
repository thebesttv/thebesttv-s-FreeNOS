31 Error LinnDirectory::read(IOBuffer & buffer, Size size, Size offset)
32 {
33  LinnSuperBlock *sb = fs->getSuperBlock();
34  LinnDirectoryEntry dent;
35  LinnInode *dInode;
36  Size bytes = ZERO, blk;
37  Error e;
38  Dirent tmp;
39 
40  // Read directory entries
41  for (u32 ent = 0; ent < inode->size / sizeof(LinnDirectoryEntry); ent++)
42  {
43  // Point to correct (direct) block
44  if ((blk = (ent * sizeof(LinnDirectoryEntry)) / sb->blockSize)
45  >= LINN_INODE_DIR_BLOCKS)
46  {
47  break;
48  }
49  // Calculate offset to read.
50  u64 off = (inode->block[blk] * sb->blockSize) +
51  (ent * sizeof(LinnDirectoryEntry));
52 
53  // Get the next entry.
54  if (fs->getStorage()->read(off, &dent,
55  sizeof(LinnDirectoryEntry)) < 0)
56  {
57  return EACCES;
58  }
59  // Can we read another entry?
60  if (bytes + sizeof(Dirent) > size)
61  {
62  return EFAULT;
63  }
64  // Fill in the Dirent.
65  if (!(dInode = fs->getInode(dent.inode)))
66  {
67  return EINVAL;
68  }
69  strlcpy(tmp.name, dent.name, LINN_DIRENT_NAME_LEN);
70  tmp.type = (FileType) dInode->type;
71 
72  // Copy to the buffer.
73  if (( e = buffer.write(&tmp, sizeof(Dirent), bytes)) < 0)
74  {
75  return e;
76  }
77  bytes += sizeof(Dirent);
78  }
79  // All done.
80  return bytes;
81 }

110 bool LinnDirectory::getLinnDirectoryEntry(LinnDirectoryEntry *dent,
111  const char *name)
112 {
113  LinnSuperBlock *sb = fs->getSuperBlock();
114  u64 offset;
115 
116  // Loop all blocks.
117  for (u32 blk = 0; blk < LINN_INODE_NUM_BLOCKS(sb, inode); blk++)
118  {
119  // Read directory entries.
120  for (u32 ent = 0; ent < LINN_DIRENT_PER_BLOCK(sb); ent++)
121  {
122  // Calculate offset to read.
123  offset = (inode->block[blk] * sb->blockSize) +
124  (sizeof(LinnDirectoryEntry) * ent);
125 
126  // Get the next entry.
127  if (fs->getStorage()->read(offset, dent,
128  sizeof(LinnDirectoryEntry)) < 0)
129  {
130  return false;
131  }
132  // Is it the entry we are looking for?
133  if (strcmp(name, dent->name) == 0)
134  {
135  return true;
136  }
137  }
138  }
139  // Not found.
140  return false;
