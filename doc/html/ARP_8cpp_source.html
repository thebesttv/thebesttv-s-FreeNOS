85  Ethernet::Address *ethAddr)

106 Error ARP::sendRequest(IPV4::Address address)
107 {
108  DEBUG("");
109 
110  // Get cache entry
111  ARPCache *entry = getCacheEntry(address);
112  if (!entry)
113  return EHOSTUNREACH;
114 
115  // Update the cache entry administration
116  entry->valid = false;
117  entry->retryCount++;
118  if (entry->retryCount > MaxRetries)
119  {
120  entry->retryCount = 0;
121  return EHOSTUNREACH;
122  }
123 
124  // Destination is broadcast ethernet address
125  Ethernet::Address destAddr;
126  MemoryBlock::set(&destAddr, 0xff, sizeof(destAddr));
127 
128  // Get a fresh ethernet packet
129  NetworkQueue::Packet *pkt = m_ether->getTransmitPacket(
130  &destAddr,
131  Ethernet::ARP
132  );
133  if (!pkt)
134  return EAGAIN;
135 
136  Ethernet::Header *ether = (Ethernet::Header *) (pkt->data + pkt->size - sizeof(Ethernet::Header));
137  ARP::Header *arp = (ARP::Header *) (pkt->data + pkt->size);
138  pkt->size += sizeof(ARP::Header);
139 
140  // Fill the ARP packet
141  arp->hardwareType = cpu_to_be16(ARP::Ethernet);
142  arp->protocolType = cpu_to_be16(ARP::IPV4);
143  arp->hardwareLength = sizeof(Ethernet::Address);
144  arp->protocolLength = sizeof(IPV4::Address);
145  arp->operation = cpu_to_be16(ARP::Request);
146 
147  // Get our current IP
148  IPV4::Address ipaddr;
149  m_ip->getAddress(&ipaddr);
150 
151  // Fill source and destinations
152  MemoryBlock::copy(&arp->etherSender, &ether->source, sizeof(Ethernet::Address));
153  arp->ipSender = cpu_to_be32(ipaddr);
154  MemoryBlock::copy(&arp->etherTarget, &ether->destination, sizeof(Ethernet::Address));
155  arp->ipTarget = cpu_to_be32(address);
156 
157  // Send the packet using the network device
158  return m_device->transmit(pkt);
159 }

203 Error ARP::process(NetworkQueue::Packet *pkt, Size offset)
204 {
205  DEBUG("");
206 
207  const Ethernet::Header *ether = (Ethernet::Header *) (pkt->data + offset - sizeof(Ethernet::Header));
208  const Header *arp = (Header *) (pkt->data + offset);
209  IPV4::Address ipAddr;
210 
211  if (!m_ip)
212  return EINVAL;
213 
214  m_ip->getAddress(&ipAddr);
215 
216  switch (be16_to_cpu(arp->operation))
217  {
218  case Request:
219  // Only send reply if the request is for our IP
220  if (be32_to_cpu(arp->ipTarget) == ipAddr)
221  {
222  updateCacheEntry(be32_to_cpu(arp->ipSender),
223  arp->etherSender);
224  return sendReply(&ether->source, arp->ipSender);
225  }
226  break;
227 
228  case Reply: {
229  updateCacheEntry(be32_to_cpu(arp->ipSender),
230  arp->etherSender);
231  return m_sock->process(pkt);
232  }
233  }
234  // Unknown ARP operation
235  return ENOTSUP;
236 }
