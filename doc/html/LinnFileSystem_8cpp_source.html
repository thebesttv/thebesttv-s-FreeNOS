28 int main(int argc, char **argv)
29 {
30  KernelLog log;
31  Storage *storage = ZERO;
32  const char *path = "/";
33  SystemInformation info;
34 
35  // Only run on core0
36  if (info.coreId != 0)
37  return EXIT_SUCCESS;
38 
39  log.setMinimumLogLevel(Log::Notice);
40 
41  // Mount the given file, or try to use the BootImage embedded rootfs
42  if (argc > 3)
43  {
44  NOTICE("file storage: " << argv[1] << " at offset " << atoi(argv[2]));
45  storage = new FileStorage(argv[1], atoi(argv[2]));
46  path = argv[3];
47  }
48  else
49  {
50  BootImageStorage *bm = new BootImageStorage(LINNFS_ROOTFS_FILE);
51  if (bm->load())
52  {
53  NOTICE("boot image: " << LINNFS_ROOTFS_FILE);
54  storage = bm;
55  } else
56  FATAL("unable to load: " << LINNFS_ROOTFS_FILE);
57  }
58 
59  // Mount, then start serving requests.
60  if (storage)
61  {
62  LinnFileSystem server(path, storage);
63  server.mount();
64  return server.run();
65  }
66  ERROR("no usable storage found");
67  return EXIT_FAILURE;
68 }

70 LinnFileSystem::LinnFileSystem(const char *p, Storage *s)
71  : FileSystem(p), storage(s), groups(ZERO)
72 {
73  LinnInode *rootInode;
74  LinnGroup *group;
75  Size offset;
76  Error e;
77 
78  // Read out the superblock.
79  if ((e = s->read(LINN_SUPER_OFFSET, &super,
80  sizeof(super))) <= 0)
81  {
82  FATAL("reading superblock failed: " <<
83  strerror(e));
84  }
85  // Verify magic.
86  if (super.magic0 != LINN_SUPER_MAGIC0 ||
87  super.magic1 != LINN_SUPER_MAGIC1)
88  {
89  FATAL("magic mismatch");
90  }
91  // Create groups vector.
92  groups = new Vector<LinnGroup *>(LINN_GROUP_COUNT(&super));
93  groups->fill(ZERO);
94 
95  // Read out group descriptors.
96  for (Size i = 0; i < LINN_GROUP_COUNT(&super); i++)
97  {
98  // Allocate buffer.
99  group = new LinnGroup;
100  offset = (super.groupsTable * super.blockSize) +
101  (sizeof(LinnGroup) * i);
102 
103  // Read from storage.
104  if ((e = s->read(offset, group, sizeof(LinnGroup))) <= 0)
105  {
106  FATAL("reading group descriptor failed: " <<
107  strerror(e));
108  }
109  // Insert in the groups vector.
110  groups->insert(i, group);
111  }
112  // Print out superblock information.
113 
114  INFO(LINN_GROUP_COUNT(&super) << " group descriptors");
115  INFO(super.inodesCount - super.freeInodesCount << " inodes, " <<
116  super.blocksCount - super.freeBlocksCount << " blocks");
117 
118  // Read out the root directory.
119  rootInode = getInode(LINN_INODE_ROOT);
120  setRoot(new LinnDirectory(this, rootInode));
121 
122  // Filesystem writes are not supported
123  addIPCHandler(CreateFile, (IPCHandlerFunction) &LinnFileSystem::notSupportedHandler, false);
124  addIPCHandler(DeleteFile, (IPCHandlerFunction) &LinnFileSystem::notSupportedHandler, false);
125  addIPCHandler(WriteFile, (IPCHandlerFunction) &LinnFileSystem::notSupportedHandler, false);
126 
127  // Done.
128  NOTICE("mounted at " << p);
129 }

131 LinnInode * LinnFileSystem::getInode(u32 inodeNum)
132 {
133  LinnGroup *group;
134  LinnInode *inode;
135  Size offset;
136  Error e;
137 
138  // Validate the inode number.
139  if (inodeNum >= super.inodesCount)
140  {
141  return ZERO;
142  }
143  // Do we have this Inode cached already?
144  if (inodes.contains(inodeNum))
145  {
146  return inodes.value(inodeNum);
147  }
148  // Get the group descriptor.
149  if (!(group = getGroupByInode(inodeNum)))
150  {
151  return ZERO;
152  }
153  // Allocate inode buffer.
154  inode = new LinnInode;
155  offset = (group->inodeTable * super.blockSize) +
156  ((inodeNum % super.inodesPerGroup) * sizeof(LinnInode));
157 
158  // Read inode from storage.
159  if ((e = storage->read(offset, inode, sizeof(LinnInode))) <= 0)
160  {
161  ERROR("reading inode failed: " <<
162  strerror(e));
163  return ZERO;
164  }
165  // Insert into the cache.
166  inodes.insert(inodeNum, inode);
167  return inode;
168 }

180 u64 LinnFileSystem::getOffset(LinnInode *inode, u32 blk)
181 {
182  u64 numPerBlock = LINN_SUPER_NUM_PTRS(&super), offset;
183  u32 *block = ZERO;
184  Size depth = ZERO, remain = 1;
185 
186  // Direct blocks.
187  if (blk < LINN_INODE_DIR_BLOCKS)
188  {
189  return inode->block[blk] * super.blockSize;
190  }
191  // Indirect blocks.
192  if (blk - LINN_INODE_DIR_BLOCKS < numPerBlock)
193  {
194  depth = 1;
195  }
196  // Double indirect blocks.
197  else if (blk - LINN_INODE_DIR_BLOCKS < numPerBlock * numPerBlock)
198  {
199  depth = 2;
200  }
201  // Triple indirect blocks.
202  else
203  depth = 3;
204 
205  // Allocate temporary block.
206  block = new u32[LINN_SUPER_NUM_PTRS(&super)];
207  offset = inode->block[(LINN_INODE_DIR_BLOCKS + depth - 1)];
208  offset *= super.blockSize;
209 
210  // Lookup the block number.
211  while (true)
212  {
213  // Fetch block.
214  if (storage->read(offset, block, super.blockSize) < 0)
215  {
216  delete block;
217  return 0;
218  }
219  // Calculate the number of blocks remaining per entry.
220  for (Size i = 0; i < depth - 1; i++)
221  {
222  remain *= LINN_SUPER_NUM_PTRS(&super);
223  }
224  // More indirection?
225  if (remain == 1)
226  {
227  break;
228  }
229  // Calculate the next offset.
230  offset = block[ (blk - LINN_INODE_DIR_BLOCKS) / remain ];
231  offset *= super.blockSize;
232  remain = 1;
233  depth--;
234  }
235  // Calculate the final offset.
236  offset = block[ (blk - LINN_INODE_DIR_BLOCKS) %
237  LINN_SUPER_NUM_PTRS(&super) ];
238  offset *= super.blockSize;
239 
240  // All done.
241  delete block;
242  return offset;
243 }
