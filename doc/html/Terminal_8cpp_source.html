156 Error Terminal::write(IOBuffer & buffer, Size size, Size offset)
157 {
158  char cr = '\r', ch;
159 
160  // Initialize buffer with the current screen first
161  ::lseek(output, 0, SEEK_SET);
162  ::read(output, this->buffer, width * height * 2);
163 
164  // Loop all input characters. Add an additional carriage return
165  // whenever a linefeed is detected.
166  for (Size i = 0; i < size; i++)
167  {
168  if (buffer[i] == '\n')
169  {
170  teken_input(&state, &cr, 1);
171  }
172  ch = buffer[i];
173  teken_input(&state, &ch, 1);
174  }
175 
176  // Flush changes back to our output device
177  ::lseek(output, 0, SEEK_SET);
178  ::write(output, this->buffer, width * height * 2);
179 
180  // Done
181  return size;
182 }

219 {

242 void fill(Terminal *term, const teken_rect_t *rect,
243  teken_char_t ch, const teken_attr_t *attr)
244 {
245  // Make sure we don't overwrite the cursor
246  term->hideCursor();
247 
248  // Fill video memory; loop rows
249  for (Size row = rect->tr_begin.tp_row;
250  row < rect->tr_end.tp_row; row++)
251  {
252  // Loop columns
253  for (Size col = rect->tr_begin.tp_col;
254  col < rect->tr_end.tp_col; col++)
255  {
256  term->getBuffer()[col + (row * term->getWidth())] =
257  VGA_CHAR(ch, tekenToVGA[attr->ta_fgcolor], BLACK);
258  }
259  }
260  // Show cursor again
261  term->showCursor();
262 }

264 void copy(Terminal *term, const teken_rect_t *rect,
265  const teken_pos_t *pos)
266 {
267  // Retrieve variables
268  u16 *buffer = term->getBuffer();
269  Size width = term->getWidth();
270 
271  // Calculate sizes
272  Size numCols = rect->tr_end.tp_col - rect->tr_begin.tp_col;
273  Size numRows = rect->tr_end.tp_row - rect->tr_begin.tp_row;
274 
275  // Hide cursor first
276  term->hideCursor();
277 
278  // Copy video memory
279  memcpy(buffer + pos->tp_col + (pos->tp_row * width),
280  buffer + rect->tr_begin.tp_col + (rect->tr_begin.tp_row * width),
281  numCols + (numRows * width) * sizeof(u16));
282 
283  // Show cursor again
284  term->showCursor();
285 }
