38 Allocator::Result PoolAllocator::allocate(Size *size, Address *addr, Size align)
39 {
40  Size index, nPools = 1;
41  MemoryPool *pool = ZERO;
42 
43  // Find the correct pool size
44  for (index = POOL_MIN_POWER; index < POOL_MAX_POWER; index++)
45  {
46  if (*size <= (Size) 1 << (index + 1)) break;
47  }
48 
49  // Do we need to allocate an initial pool?
50  if (!m_pools[index] && m_parent)
51  {
52  pool = m_pools[index] = newPool(index, POOL_MIN_COUNT(*size));
53  }
54  // Search for pool with enough memory
55  else
56  {
57  // Loop current pools
58  for (pool = m_pools[index]; pool; pool = pool->next, nPools++)
59  {
60  // At least one block still free?
61  if (pool->free)
62  break;
63 
64  // If no pool has free space anymore, allocate another
65  if (!pool->next)
66  {
67  pool = newPool(index, POOL_MIN_COUNT(*size) * nPools);
68  break;
69  }
70  }
71  }
72  // Attempt to allocate
73  if (pool)
74  {
75  *addr = pool->allocate();
76  return Success;
77  }
78  *addr = ZERO;
79  return OutOfMemory;
80 }

82 MemoryPool * PoolAllocator::newPool(Size index, Size cnt)
83 {
84  MemoryPool *pool = 0;
85  Size sz;
86 
87  // Prepare amount to allocate from m_parent
88  sz = cnt * (1 << (index + 1));
89  sz += sizeof(MemoryPool);
90  sz += BITMAP_NUM_BYTES(cnt);
91  sz += MEMALIGN;
92 
93  // Ask m_parent for memory, then fill in the pool
94  if (m_parent->allocate(&sz, (Address *)&pool) == Success)
95  {
96  pool->count = cnt;
97  pool->addr = aligned( ((Address) (pool + 1)) + BITMAP_NUM_BYTES(pool->count), MEMALIGN );
98  pool->next = m_pools[index];
99  pool->free = pool->count;
100  pool->size = (1 << (index + 1));
101  m_pools[index] = pool;
102  MemoryBlock::set(pool->blocks, 0, BITMAP_NUM_BYTES(pool->count));
103  }
104  return pool;
105 }

107 Allocator::Result PoolAllocator::release(Address addr)
108 {
109  for (Size i = POOL_MIN_POWER - 1; i < POOL_MAX_POWER; i++)
110  {
111  for (MemoryPool *p = m_pools[i]; p; p = p->next)
112  {
113  if (addr < p->addr + (p->count * p->size) &&
114  addr >= p->addr)
115  {
116  p->release(addr);
117  return Success;
118  }
119  }
120  }
121  return InvalidAddress;
