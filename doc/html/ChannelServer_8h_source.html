142  int run()
143  {
144  // Enter loop
145  while (true)
146  {
147  // Reset
148  m_sendReply = true;
149 
150  // Process kernel events
151  readKernelEvents();
152 
153  // Process user messages
154  readChannels();
155 
156  // Retry requests until all served (EAGAIN or return value)
157  retryAllRequests();
158 
159  // Sleep with timeout or return in case the process is
160  // woken up by an external (wakeup) interrupt.
161  DEBUG("EnterSleep");
162  Address expiry = 0;
163 
164  if (m_expiry.frequency)
165  expiry = (Address) &m_expiry;
166 
167  Error r = ProcessCtl(SELF, EnterSleep, expiry, (Address) &m_time);
168  DEBUG("EnterSleep returned: " << (int)r);
169 
170  // Check for sleep timeout
171  if (m_expiry.frequency)
172  {
173  if (ProcessCtl(SELF, InfoTimer, (Address) &m_time) != API::Success)
174  {
175  ERROR("failed to retrieve system timer");
176  }
177  else if (m_expiry.ticks < m_time.ticks)
178  {
179  m_expiry.frequency = 0;
180  timeout();
181  }
182  }
183  }
184 
185  // Satify compiler
186  return 0;
187  }

260  Result accept(ProcessID pid, Memory::Range range)
261  {
262  // Create consumer
263  if (!m_registry->getConsumer(pid))
264  {
265  MemoryChannel *consumer = new MemoryChannel;
266  consumer->setMode(Channel::Consumer);
267  consumer->setMessageSize(sizeof(MsgType));
268  consumer->setVirtual(range.virt, range.virt + PAGESIZE);
269  m_registry->registerConsumer(pid, consumer);
270  }
271  // Create producer

273  {

275  producer->setMode(Channel::Producer);
276  producer->setMessageSize(sizeof(MsgType));
277  producer->setVirtual(range.virt + (PAGESIZE*2),
278  range.virt + (PAGESIZE*3));
279  m_registry->registerProducer(pid, producer);
280  }
281  // Done
282  return Success;

290  Result readKernelEvents()
291  {
292  ProcessEvent event;
293 
294  // Try to read a message on the kernel event channel
295  while (m_kernelEvent.read(&event) == Channel::Success)
296  {
297  DEBUG(m_self << ": got kernel event: " << (int) event.type);
298 
299  switch (event.type)
300  {
301  case ShareCreated:
302  {
303  DEBUG(m_self << ": share created for PID: " << event.share.pid);
304  accept(event.share.pid, event.share.range);
305  break;
306  }
307  case InterruptEvent:
308  {
309  DEBUG(m_self << ": interrupt: " << event.number);
310 
311  if (m_irqHandlers->at(event.number))
312  {
313  (m_instance->*(m_irqHandlers->at(event.number))->exec) (event.number);
314  }
315  break;
316  }
317  case ProcessTerminated:
318  {
319  DEBUG(m_self << ": process terminated: PID " << event.number);
320  m_registry->unregisterConsumer(event.number);
321  m_registry->unregisterProducer(event.number);
322 
323  // cleanup the VMShare area now for that process
324  VMShare(event.number, API::Delete, ZERO);
325  break;
326  }
327  default:
328  DEBUG(m_self << ": ???\n");
329  break;
330  }
331  }
332  return Success;
333  }

340  Result readChannels()
341  {
342  MsgType msg;
343 
344  // Try to receive message on each consumer channel
345  for (HashIterator<ProcessID, Channel *> i(m_registry->getConsumers()); i.hasCurrent(); i++)
346  {
347  Channel *ch = i.current();
348  DEBUG(m_self << ": trying to receive from PID " << i.key());
349 
350  // Read all messages in the consumer channel
351  while (ch->read(&msg) == Channel::Success)
352  {
353  DEBUG(m_self << ": received message");
354  msg.from = i.key();
355 
356  // Is the message a response from earlier client request?
357  if (msg.type == ChannelMessage::Response)
358  {
359  if (m_client->processResponse(msg.from, &msg) != ChannelClient::Success)
360  {
361  ERROR(m_self << ": failed to process client response from PID " <<
362  msg.from << " with identifier " << msg.identifier);
363  }
364  }
365  // Message is a request to us
366  else if (m_ipcHandlers->at(msg.action))
367  {
368  m_sendReply = m_ipcHandlers->at(msg.action)->sendReply;
369  (m_instance->*(m_ipcHandlers->at(msg.action))->exec) (&msg);
370 
371  // Send reply
372  if (m_sendReply)
373  {
374  Channel *ch = m_registry->getProducer(i.key());
375  if (!ch)
376  {
377  ERROR(m_self << ": no producer channel found for PID: " << i.key());
378  }
379  else if (ch->write(&msg) != Channel::Success)
380  {
381  ERROR(m_self << ": failed to send reply message to PID: " << i.key());
382  }
383  else
384  ProcessCtl(i.key(), Resume, 0);
385  }
386  }
387  }
388  }
389  return Success;
390  }
