38 Error Time::read(IOBuffer & buffer, Size size, Size offset)
39 {
40  unsigned int year, month, day, hour, min, sec = 0, time;
41  char tmp[16];
42  int n;
43 
44  // PHONY read
45  if(offset >= 10)
46  {
47  return 0;
48  }
49 
50  // If UIP is clear, then we have >= 244 microseconds before
51  // RTC registers will be updated. Spec sheet says that this
52  // is the reliable way to read RTC - registers. If UIP is set
53  // then the register access might be invalid
54  while((readCMOS(RTC_STATUS_A) & RTC_UIP));
55 
56  // Read the date/time values from the CMOS
57  sec = readCMOS(RTC_SECONDS);
58  min = readCMOS(RTC_MINUTES);
59  hour = readCMOS(RTC_HOURS);
60  day = readCMOS(RTC_DAY_OF_MONTH);
61  month = readCMOS(RTC_MONTH);
62  year = readCMOS(RTC_YEAR);
63 
64  // Check if the time values are stored in binary or BCD format
65  if( (readCMOS(RTC_STATUS_B) & RTC_BCD) )
66  {
67  // Convert from binary coded decimal (bcd) to machine numbers
68  sec = bcd2bin(sec);
69  min = bcd2bin(min);
70  hour = bcd2bin(hour);
71  day = bcd2bin(day);
72  month = bcd2bin(month);
73  year = bcd2bin(year);
74  }
75 
76  // Assume that a two-digit year is after 2000
77  if(year < 100)
78  {
79  year += CMOS_YEARS_OFFS;
80  }
81 
82  // Format as an ASCII string
83  time = mktime(year, month, day, hour, min, sec);
84  n = snprintf(tmp, size < sizeof(tmp) ? size : sizeof(tmp), "%u", time);
85  buffer.write(tmp, n);
86 
87  // Done
88  return (Error) size;
89 }
