36 API::Result ProcessCtlHandler(ProcessID procID,
37  ProcessOperation action,
38  Address addr,
39  Address output)
40 {
41  Process *proc = ZERO;
42  ProcessInfo *info = (ProcessInfo *) addr;
43  ProcessManager *procs = Kernel::instance->getProcessManager();
44  Timer *timer;
45  Arch::MemoryMap map;
46 
47  DEBUG("#" << procs->current()->getID() << " " << action << " -> " << procID << " (" << addr << ")");
48 
49  // Does the target process exist?
50  if(action != GetPID && action != Spawn)
51  {
52  if (procID == SELF)
53  proc = procs->current();
54  else if (!(proc = procs->get(procID)))
55  return API::NotFound;
56  }
57  // Handle request
58  switch (action)
59  {
60  case Spawn:
61  proc = procs->create(addr, map);
62  proc->setParent(procs->current()->getID());
63  return proc->getID();
64 
65  case KillPID:
66  procs->remove(proc, addr); // Addr contains the exit status
67  procs->schedule();
68  break;
69 
70  case GetPID:
71  return procs->current()->getID();
72 
73  case GetParent:
74  return procs->current()->getParent();
75 
76  case Schedule:
77  procs->schedule();
78  break;
79 
80  case Resume:
81  // increment wakeup counter and set process ready
82  proc->wakeup();
83  break;
84 
85  case WatchIRQ:
86  Kernel::instance->hookIntVector(IRQ(addr), (InterruptHandler *)interruptNotify, (ulong)proc);
87  break;
88 
89  case EnableIRQ:
90  Kernel::instance->enableIRQ(addr, true);
91  break;
92 
93  case DisableIRQ:
94  Kernel::instance->enableIRQ(addr, false);
95  break;
96 
97  case InfoPID:
98  info->id = proc->getID();
99  info->state = proc->getState();
100  info->userStack = proc->getUserStack();
101  info->kernelStack = proc->getKernelStack();
102  info->pageDirectory = proc->getPageDirectory();
103  info->parent = proc->getParent();
104  break;
105 
106  case WaitPID:
107  procs->current()->setWait(proc->getID());
108  procs->current()->setState(Process::Waiting);
109  procs->schedule();
110  return procs->current()->getWait(); // contains the exit status of the other process
111 
112  case InfoTimer:
113  if (!(timer = Kernel::instance->getTimer()))
114  return API::NotFound;
115 
116  timer->getCurrent((Timer::Info *) addr);
117  break;
118 
119  case WaitTimer:
120  // Process is only allowed to continue execution after the sleep timer expires
121  procs->current()->setSleepTimer((const Timer::Info *)addr);
122  procs->current()->setState(Process::Sleeping);
123  procs->schedule();
124  break;
125 
126  case EnterSleep:
127  // Only sleeps the process if no pending wakeups
128  if (procs->current()->sleep((Timer::Info *)addr) == Process::Success)
129  procs->schedule();
130  break;
131 
132  case SetStack:
133  proc->setUserStack(addr);
134  break;
135  }
136  return API::Success;
137 }
