54 NetCat::Result NetCat::initialize()
55 {
56  DEBUG("");
57 
58  // Create a network client
59  m_client = new NetworkClient(arguments().get("DEVICE"));
60 
61  // Initialize networking client
62  if (m_client->initialize() != NetworkClient::Success)
63  {
64  ERROR("failed to initialize network client for device: "
65  << arguments().get("DEVICE"));
66  return IOError;
67  }
68  // Create an UDP socket
69  if (m_client->createSocket(NetworkClient::UDP, &m_socket) != NetworkClient::Success)
70  {
71  ERROR("failed to create UDP socket");
72  return IOError;
73  }
74  // Convert to IPV4 address
75  if (!(m_host = IPV4::toAddress(arguments().get("HOST"))))
76  {
77  ERROR("failed to convert to IPV4 address: " << arguments().get("HOST"));
78  return IOError;
79  }
80  // Convert to port
81  m_port = atoi(arguments().get("PORT"));
82 
83  // Bind to a local port.
84  if (m_client->bindSocket(m_socket, 0, arguments().get("listen") ? m_port : 0))
85  {
86  ERROR("failed to bind socket");

131 {
132  DEBUG("");
133

136  bool reading = true;
137 
138  // Read a line
139  while (m_lineLen < sizeof(m_lineBuf) - 3 && reading)
140  {
141  // Read a character
142  read(0, &m_lineBuf[m_lineLen], 1);
143 
144  // Process character
145  switch (m_lineBuf[m_lineLen])
146  {
147  case '\r':
148  case '\n':
149  printf("\r\n");
150  reading = false;
151  break;
152 
153  case '\b':
154  if (m_lineLen > 0)
155  {
156  m_lineLen--;
157  printf("\b \b");
158  }
159  break;
160 
161  default:
162  printf("%c", m_lineBuf[m_lineLen]);
163  m_lineLen++;
164  break;
165  }
166  }
167 
168  // Done
169  m_lineBuf[m_lineLen++] = '\r';
170  m_lineBuf[m_lineLen++] = '\n';
171  m_lineBuf[m_lineLen] = ZERO;
172 
173  // Single line dot means quit
174  if (strcmp(m_lineBuf, ".\r\n") == 0)
175  ::exit(0);
176

200 {

203  // Receive UDP datagram
204  int r = recvfrom(m_socket, m_lineBuf, sizeof(m_lineBuf), 0,
205  &addr, sizeof(addr));

209  return IOError;
