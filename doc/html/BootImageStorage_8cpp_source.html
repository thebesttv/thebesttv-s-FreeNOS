29 bool BootImageStorage::load()
30 {
31  SystemInformation info;
32  BootImage *image;
33  BootSymbol *symbol;
34  BootSegment *segment;
35  Memory::Range range;
36  u8 *base;
37 
38  // Request boot image memory
39  range.size = info.bootImageSize;
40  range.access = Memory::User |
41  Memory::Readable;
42  range.virt = ZERO;
43  range.phys = info.bootImageAddress;
44  VMCtl(SELF, Map, &range);
45 
46  // Update our state
47  image = (BootImage *) range.virt;
48  base = (u8 *) image;
49 
50  // Search for the given BootSymbol
51  for (uint i = 0; i < image->symbolTableCount; i++)
52  {
53  symbol = (BootSymbol *) (base + image->symbolTableOffset);
54  symbol += i;
55 
56  if (strcmp(symbol->name, m_name) == 0)
57  {
58  m_size = symbol->segmentsTotalSize;
59  segment = (BootSegment *) (base + image->segmentsTableOffset +
60  (symbol->segmentsOffset * sizeof(BootSegment)));
61  m_data = base + segment->offset;
62  // Success
63  return true;
64  }
65  }
66  // BootSymbol not found
67  return false;
68 }
