61 MemoryChannel::Result MemoryChannel::read(void *buffer)
62 {
63  RingHead head;
64 
65  // Read the current ring head
66  m_data.read(0, sizeof(head), &head);
67 
68  // Check if a message is present
69  if (head.index == m_head.index)
70  return NotFound;
71 
72  // Read one message
73  m_data.read((m_head.index+1) * m_messageSize, m_messageSize, buffer);
74 
75  // Increment head index
76  m_head.index = (m_head.index + 1) % m_maximumMessages;
77 
78  // Update read index
79  m_feedback.write(0, sizeof(m_head), &m_head);
80  return Success;
81 }

83 MemoryChannel::Result MemoryChannel::write(void *buffer)
84 {
85  RingHead reader;
86 
87  // Read current ring head
88  m_feedback.read(0, sizeof(RingHead), &reader);
89 
90  // Check if buffer space is available for the message
91  if (((m_head.index + 1) % m_maximumMessages) == reader.index)
92  return ChannelFull;
93 
94  // write the message
95  m_data.write((m_head.index+1) * m_messageSize, m_messageSize, buffer);
96 
97  // Increment write index
98  m_head.index = (m_head.index + 1) % m_maximumMessages;
99  m_data.write(0, sizeof(m_head), &m_head);
100  return Success;
101 }

103 MemoryChannel::Result MemoryChannel::flush()
104 {
105  // Cannot flush caches in usermode. All usermode code
106  // should memory map without caching.
107  if (!isKernel)
108  return IOError;
109 
110  // Clean both pages from the cache
