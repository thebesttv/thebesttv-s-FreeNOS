44 ProcessList::Result ProcessList::exec()
45 {
46  Arch::MemoryMap map;
47  Memory::Range range = map.range(MemoryMap::UserArgs);
48  ProcessInfo info;
49  String out;
50  char line[256], cmd[PATH_MAX];
51 
52  // Print header
53  out << "ID PARENT USER GROUP STATUS CMD\r\n";
54  memset(&cmd, 0, sizeof(cmd));
55 
56  // Loop processes
57  for (uint i = 0; i < MAX_PROCS; i++)
58  {
59  // Request kernel's process information
60  if (ProcessCtl(i, InfoPID, (Address) &info) != API::NotFound)
61  {
62  DEBUG("PID " << i << " state = " << info.state);
63 
64  // Get the command
65  VMCopy(i, API::Read, (Address) cmd, range.virt, PATH_MAX);
66 
67  // Output a line
68  snprintf(line, sizeof(line),
69  "%3d %7d %4d %5d %10s %32s\r\n",
70  i, info.parent, 0, 0, ProcessStates[info.state], cmd);
71  out << line;
72  }
73  }
74 
75  // Output the table
76  write(1, *out, out.length());
77  return Success;
78 }
