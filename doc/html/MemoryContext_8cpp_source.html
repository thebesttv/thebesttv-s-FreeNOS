80 MemoryContext::Result MemoryContext::findFree(Size size, MemoryMap::Region region, Address *virt) const
81 {
82  Memory::Range r = m_map->range(region);
83  Size currentSize = 0;
84  Address addr = r.virt, currentAddr = r.virt, tmp;
85 
86  while (addr < r.virt+r.size && currentSize < size)
87  {
88  if (lookup(addr, &tmp) == InvalidAddress)
89  {
90  currentSize += PAGESIZE;
91  }
92  else
93  {
94  currentSize = 0;
95  currentAddr = addr + PAGESIZE;
96  }
97  addr += PAGESIZE;
98  }
99 
100  if (currentSize >= size)
101  {
102  *virt = currentAddr;
103  return Success;
104  }
105  else
106  return OutOfMemory;
107 }
