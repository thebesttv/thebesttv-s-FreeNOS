36 NetworkClient::Result NetworkClient::initialize()
37 {
38  // Get a list of mounts
39  refreshMounts(0);
40  FileSystemMount *mounts = ::getMounts();
41  FileSystemMount *match = 0;
42  Size matchLen = 0;
43 
44  // Find closest matching device
45  for (Size i = 0; i < FILESYSTEM_MAXMOUNTS; i++)
46  {
47  if (mounts[i].path[0] && strncmp(mounts[i].path, "/network/", 9) == 0)
48  {
49  Size len = 0;
50 
51  for (Size j = 0; mounts[i].path[j+9] && m_deviceName[j]; j++, len++)
52  {
53  if (mounts[i].path[j+9] != m_deviceName[j])
54  {
55  len = 0;
56  break;
57  }
58  }
59  if (len >= matchLen)
60  {
61  match = &mounts[i];
62  matchLen = len;
63  }
64  }
65  }
66  if (!match)
67  {
68  ERROR("network device not found: " << *m_deviceName);
69  return IOError;
70  }
71  m_deviceName = match->path;
72  return Success;
73 }

115 NetworkClient::Result NetworkClient::writeSocketInfo(
116  int sock,
117  IPV4::Address addr,
118  u16 port,
119  NetworkClient::SocketAction action)
120 {
121  char buf[64];
122 
123  DEBUG("");
124 
125  // Read socket factory. The factory will create
126  // a new socket for us. We need to read the new file path
127  Error r = ::read(sock, buf, sizeof(buf));
128  if (r < 0)
129  return IOError;
130 
131  // Update the file descriptor path
132  FileDescriptor *fd = &getFiles()[sock];
133  MemoryBlock::copy(fd->path, buf, sizeof(buf));
134 
135  // Write address+port+action info to the socket
136  SocketInfo info;
137  info.address = addr;
138  info.port = port;
139  info.action = action;
140 
141  r = ::write(sock, &info, sizeof(info));
142  if (r < 0)
143  return IOError;
144 
145  // Done
146  return Success;
147 }
