23 API::Result VMCtlHandler(ProcessID procID, MemoryOperation op, Memory::Range *range)
24 {
25  ProcessManager *procs = Kernel::instance->getProcessManager();
26  Process *proc = ZERO;
27  Error ret = API::Success;
28 
29  DEBUG("");
30 
31  // Find the given process
32  if (procID == SELF)
33  proc = procs->current();
34  else if (!(proc = procs->get(procID)))
35  {
36  return API::NotFound;
37  }
38 
39  // Retrieve memory context
40  MemoryContext *mem = proc->getMemoryContext();
41 
42  // Perform operation
43  switch (op)
44  {
45  case LookupVirtual:
46  if (mem->lookup(range->virt, &range->phys) != MemoryContext::Success)
47  return API::AccessViolation;
48  break;
49 
50  case Map:
51  if (!range->virt)
52  {
53  mem->findFree(range->size, MemoryMap::UserPrivate, &range->virt);
54  range->virt += range->phys & ~PAGEMASK;
55  }
56  mem->mapRange(range);
57  break;
58 
59  case UnMap:
60  mem->unmapRange(range);
61  break;
62 
63  case Release:
64  mem->releaseRange(range);
65  break;
66 
67  case CacheClean: {
68  Arch::Cache cache;
69  cache.cleanInvalidate(Cache::Data);
70  break;
71  }
72 
73  case Access: {
74  MemoryContext::Result mr = mem->access(range->virt, &range->access);
75  if (mr == MemoryContext::Success)
76  ret = API::Success;
77  else
78  ret = (API::Result) mr;
79  break;
80  }
81 
82  case ReserveMem:
83  for (uint i = 0; i < range->size; i+=PAGESIZE)
84  {
85  if (Kernel::instance->getAllocator()->allocate(range->phys + i) != Allocator::Success)
86  {
87  ERROR("address " << (void *) (range->phys + i) << " already allocated");
88  return API::OutOfMemory;
89  }
90  }
91  break;
92 
93  default:
94  ret = API::InvalidArgument;
95  break;
96  }
97  // Done
98  return ret;
99 }
