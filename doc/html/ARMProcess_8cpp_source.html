33 Process::Result ARMProcess::initialize()
34 {
35  Memory::Range range;
36 
37  // Create MMU context
38  m_memoryContext = new ARMPaging(
39  &m_map,
40  Kernel::instance->getAllocator()
41  );
42  if (!m_memoryContext)
43  return OutOfMemory;
44 
45  // User stack (high memory).
46  range = m_map.range(MemoryMap::UserStack);
47  range.access = Memory::Readable | Memory::Writable | Memory::User;
48  if (Kernel::instance->getAllocator()->allocate(&range.size, &range.phys) != Allocator::Success)
49  return OutOfMemory;
50 
51  if (m_memoryContext->mapRange(&range) != MemoryContext::Success)
52  return MemoryMapError;
53  setUserStack(range.virt + range.size - MEMALIGN8);
54 
55  // Fill usermode program registers
56  MemoryBlock::set(&m_cpuState, 0, sizeof(m_cpuState));
57  m_cpuState.sp = m_userStack; // user stack pointer
58  m_cpuState.pc = m_entry; // user program counter
59  m_cpuState.cpsr = (m_privileged ? SYS_MODE : USR_MODE); // current program status (CPSR)
60 
61  // Finalize with generic initialization
62  return Process::initialize();
63 }

80 void ARMProcess::execute(Process *previous)
81 {

83  m_memoryContext->activate();
84 
85  // First process starts from loadCoreState0
86  if (firstProcess)
87  {
88  firstProcess = false;
89 
90  CPUState *ptr = ((CPUState *) (svcStack)) - 1;
91  MemoryBlock::copy(ptr, &m_cpuState, sizeof(*ptr));
92 
93  // Switch to the actual SVC stack and switch to usermode
94  asm volatile ("ldr sp, =svcStack\n"
95  "sub sp, sp, %0\n"
96  "ldr r0, =loadCoreState0\n"
