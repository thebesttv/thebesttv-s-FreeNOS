57 int CoreServer::runCore()
58 {
59  FileSystemMessage msg;
60 
61  if (m_info.coreId == 0)
62  return run();
63 
64  while (true)
65  {
66  // wait from a message of the master core
67  while (m_fromMaster->read(&msg) != Channel::Success);
68 
69  if (m_ipcHandlers->at(msg.action))
70  {
71  m_sendReply = m_ipcHandlers->at(msg.action)->sendReply;
72  (this->*(m_ipcHandlers->at(msg.action))->exec)(&msg);
73 
74  if (m_sendReply)
75  {
76  while (m_toMaster->write(&msg) != Channel::Success)
77  ;
78  }
79  }
80  }

83 void CoreServer::createProcess(FileSystemMessage *msg)
84 {
85  char cmd[128];
86  Memory::Range range;
87 
88  if (m_info.coreId == 0)
89  {
90  MemoryChannel *ch = (MemoryChannel *) m_toSlave->get(msg->size);
91 
92  if (!ch)
93  {
94  ERROR("invalid coreId=" << msg->size);
95  msg->result = EBADF;
96  return;
97  }
98 
99  range.virt = (Address) msg->buffer;
100  VMCtl(msg->from, LookupVirtual, &range);
101  msg->buffer = (char *) range.phys;
102 
103  range.virt = (Address) msg->path;
104  VMCtl(msg->from, LookupVirtual, &range);
105  msg->path = (char *) range.phys;
106 
107  if (ch->write(msg) != Channel::Success)
108  {
109  ERROR("failed to write channel on core"<<msg->size);
110  msg->result = EBADF;
111  return;
112  }
113  DEBUG("creating program at phys " << (void *) msg->buffer << " on core" << msg->size);
114 
115  ch = (MemoryChannel *) m_fromSlave->get(msg->size);
116  if (!ch)
117  {
118  ERROR("cannot find read channel for core" << msg->size);
119  msg->result = EBADF;
120  return;
121  }
122  while (ch->read(msg) != Channel::Success)
123  ;
124  DEBUG("program created with result " << (int)msg->result << " at core" << msg->size);
125 
126  msg->result = ESUCCESS;
127  ChannelClient::instance->syncSendTo(msg, msg->from);
128  }
129  else
130  {
131  VMCopy(SELF, API::ReadPhys, (Address) cmd, (Address) msg->path, sizeof(cmd));
132 
133  range.phys = (Address) msg->buffer;
134  range.virt = 0;
135  range.access = Memory::Readable | Memory::User;
136  range.size = msg->offset;
137  VMCtl(SELF, Map, &range);
138 
139  pid_t pid = spawn(range.virt, msg->offset, cmd);
140  int status;
141 
142  // reply to master before calling waitpid()
143  msg->result = ESUCCESS;
144  while (m_toMaster->write(msg) != Channel::Success)
145  ;
146 
147  // Wait until the spawned process completes
148  waitpid(pid, &status, 0);
149  }

229 CoreServer::Result CoreServer::loadKernel()
230 {
231  struct stat st;
232  int fd;
233 
234  DEBUG("Opening : " << kernelPath);
235 
236  // Read the program image
237  if (stat(kernelPath, &st) != 0)
238  return IOError;
239 
240  if ((fd = open(kernelPath, O_RDONLY)) < 0)
241  return IOError;
242 
243  m_kernelImage = new u8[st.st_size];
244  if (read(fd, m_kernelImage, st.st_size) != st.st_size)
245  {
246  return IOError;
247  }
248  close(fd);
249 
250  // Attempt to read executable format
251  if (ExecutableFormat::find(m_kernelImage, st.st_size, &m_kernel) != ExecutableFormat::Success)
252  return ExecError;
253 
254  // Retrieve memory regions
255  m_numRegions = 16;
256  if (m_kernel->regions(m_regions, &m_numRegions) != ExecutableFormat::Success)
257  return ExecError;
258 
259  DEBUG("kernel loaded");
260  return Success;
261 }

263 CoreServer::Result CoreServer::bootCore(uint coreId, CoreInfo *info,
264  ExecutableFormat::Region *regions)
265 {
266  SystemInformation sysInfo;
267  DEBUG("Reserving: " << (void *)info->memory.phys << " size=" <<
268  info->memory.size << " available=" << sysInfo.memoryAvail);
269 
270  // Claim the core's memory
271  if (VMCtl(SELF, ReserveMem, &info->memory) != API::Success)
272  {
273  ERROR("failed to reserve memory for core" << coreId <<
274  " at " << (void *)info->memory.phys);
275  return OutOfMemory;
276  }
277 
278  DEBUG("Starting core" << coreId << " with "
279  << info->memory.size / 1024 / 1024 << "MB");
280 
281  // Map the kernel
282  for (Size i = 0; i < m_numRegions; i++)
283  {
284  Memory::Range range;
285  range.phys = info->memory.phys + regions[i].virt;
286  range.virt = 0;
287  range.size = regions[i].size;
288  range.access = Memory::Readable | Memory::Writable |
289  Memory::User;
290 
291  // Map the target kernel's memory for regions[i].size
292  if (VMCtl(SELF, Map, &range) != 0)
293  {
294  return OutOfMemory;
295  }
296  // Copy the kernel to the target core's memory
297 #pragma GCC diagnostic push
298 #pragma GCC diagnostic ignored "-Wsign-compare"
299  API::Result r = VMCopy(SELF, API::Write, (Address) regions[i].data,
300  range.virt,
301  regions[i].size);
302  if ((Size)r != regions[i].size)
303  return MemoryError;
304 
305  // Unmap the target kernel's memory
306  if (VMCtl(SELF, UnMap, &range) != API::Success)
307  {
308  return MemoryError;
309  }
310 
311  DEBUG(kernelPath << "[" << i << "] = " << (void *) m_regions[i].virt);
312  }
313 
314  // Copy the BootImage after the kernel.
315  Memory::Range range;
316  range.phys = info->bootImageAddress;
317  range.virt = 0;
318  range.size = info->bootImageSize;
319  range.access = Memory::Readable | Memory::Writable | Memory::User;
320 
321  // Map BootImage buffer
322  if (VMCtl(SELF, Map, &range) != API::Success)
323  {
324  return OutOfMemory;
325  }
326  // Copy the BootImage
327  Error r = VMCopy(SELF, API::Write, sysInfo.bootImageAddress,
328  range.virt, sysInfo.bootImageSize);
329  if (r != (Error) sysInfo.bootImageSize)
330  return MemoryError;
331 
332  // Unmap the BootImage
333  if (VMCtl(SELF, UnMap, &range) != API::Success)
334  return MemoryError;
335 
336 #ifdef INTEL
337  // Signal the core to boot
338  if (m_mp.boot(info) != IntelMP::Success) {
339  ERROR("failed to boot core" << coreId);
340  return BootError;
341  } else {
342  NOTICE("core" << coreId << " started");
343  }
344 #endif
345  return Success;
346 }

349 CoreServer::Result CoreServer::discover()
350 {
351 #ifdef INTEL
352  SystemInformation sysInfo;
353  Size memPerCore = 0;
354 
355  if (m_acpi.initialize() == IntelACPI::Success &&
356  m_acpi.discover() == IntelACPI::Success)
357  {
358  NOTICE("using ACPI as CoreManager");
359  m_mp.discover();
360  m_cores = &m_acpi;
361  }
362  else if (m_mp.discover() == IntelMP::Success)
363  {
364  NOTICE("using MPTable as CoreManager");
365  m_cores = &m_mp;
366  }
367  else
368  {
369  ERROR("no CoreManager found (ACPI or MPTable)");
370  return NotFound;
371  }
372  List<uint> & cores = m_cores->getCores();
373  if (cores.count() == 0)
374  {
375  ERROR("no cores found");
376  return NotFound;
377  }
378 
379  memPerCore = sysInfo.memorySize / cores.count();
380  memPerCore /= MegaByte(4);
381  memPerCore *= MegaByte(4);
382 
383  NOTICE("found " << cores.count() << " cores -- " <<
384  (memPerCore / 1024 / 1024) << "MB per core");
385 
386  // Allocate CoreInfo for each core
387  m_coreInfo = new Index<CoreInfo>(cores.count());
388 
389  // Boot each core
390  for (ListIterator<uint> i(cores); i.hasCurrent(); i++)
391  {
392  uint coreId = i.current();
393 
394  if (coreId != 0)
395  {
396  CoreInfo *info = new CoreInfo;
397  m_coreInfo->insert(coreId, *info);
398  MemoryBlock::set(info, 0, sizeof(CoreInfo));
399  info->coreId = coreId;
400  info->memory.phys = memPerCore * coreId;
401  info->memory.size = memPerCore - PAGESIZE;
402  info->kernel.phys = info->memory.phys;
403  info->kernel.size = MegaByte(4);
404  info->bootImageAddress = info->kernel.phys + info->kernel.size;
405  info->bootImageSize = sysInfo.bootImageSize;
406  info->coreChannelAddress = info->bootImageAddress + info->bootImageSize;
407  info->coreChannelAddress += PAGESIZE - (info->bootImageSize % PAGESIZE);
408  info->coreChannelSize = PAGESIZE * 4;
409  clearPages(info->coreChannelAddress, info->coreChannelSize);
410 
411  m_kernel->entry(&info->kernelEntry);
412  info->timerCounter = sysInfo.timerCounter;
413  strlcpy(info->kernelCommand, kernelPath, KERNEL_PATHLEN);
414  }
415  }
416 #endif
417  return Success;
418 }
