52 void DeviceServer::registerInterrupt(Device *dev, Size vector)
53 {
54  if (!m_interrupts[vector])
55  {
56  m_interrupts.insert(vector, new List<Device *>);
57  }
58  m_interrupts[vector]->append(dev);
59 
60  // Register to kernel
61  ProcessCtl(SELF, WatchIRQ, vector);
62  ProcessCtl(SELF, EnableIRQ, vector);
63 
64  // Register interrupt handler
65  addIRQHandler(vector, (IRQHandlerFunction) &DeviceServer::interruptHandler);
66 }

68 void DeviceServer::interruptHandler(Size vector)
69 {
70  List<Device *> *lst = m_interrupts.at(vector);
71 
72  // Do we have any Devices with this interrupt vector?
73  if (lst)
74  {
75  // Loop all Devices of interest. Invoke callback.
76  for (ListIterator<Device *> i(lst); i.hasCurrent(); i++)
77  {
78  i.current()->interrupt(vector);
79  }
80  }
81  // Keep retrying any pending requests, if any
82  while (retryRequests());
83 }
