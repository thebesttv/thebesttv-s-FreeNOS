34  proc = procs->current();
35  else if (!(proc = procs->get(procID)))
36  return API::NotFound;

38  MemoryContext *local = procs->current()->getMemoryContext();

40 
41  // Keep on going until all memory is processed
42  while (total < sz)
43  {
44  // Update variables
45  if (how == API::ReadPhys)
46  paddr = theirs & PAGEMASK;
47  else if (remote->lookup(theirs, &paddr) != MemoryContext::Success)
48  return API::AccessViolation;
49

51  pageOff = theirs & ~PAGEMASK;
52  bytes = (PAGESIZE - pageOff) < (sz - total) ?
53  (PAGESIZE - pageOff) : (sz - total);
54 
55  // Valid address?

57

60  return API::RangeError;
