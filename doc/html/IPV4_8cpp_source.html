96 const IPV4::Address IPV4::toAddress(const char *address)
97 {
98  String input = address;
99  List<String> lst = input.split('.');
100  Size shift = 24;
101  IPV4::Address addr = 0;
102 
103  // The address must be 4 bytes
104  if (lst.count() != 4)
105  return ZERO;
106 
107  // Extract bytes in dot format (xxx.xxx.xxx.xxx)
108  for (ListIterator<String> i(lst); i.hasCurrent(); i++)
109  {
110  String & s = i.current();
111  u8 byte = s.toLong();
112  addr |= (byte << shift);
113  shift -= 8;
114  }
115  // Done
116  return addr;
117 }

119 Error IPV4::getTransmitPacket(NetworkQueue::Packet **pkt,
120  IPV4::Address destination,
121  IPV4::Protocol type,
122  Size size)
123 {
124  Ethernet::Address ethAddr;
125  Error r;
126 
127  // Find the ethernet address using ARP first
128  if ((r = m_arp->lookupAddress(&destination, &ethAddr)) != ESUCCESS)
129  return r;
130 
131  // Get a fresh ethernet packet
132  *pkt = m_ether->getTransmitPacket(
133  &ethAddr,
134  Ethernet::IPV4
135  );
136  // Fill IP header
137  Header *hdr = (Header *) ((*pkt)->data + (*pkt)->size);
138  hdr->versionIHL = (sizeof(Header) / sizeof(u32)) | (4 << 4);
139  hdr->typeOfService = 0;
140  hdr->length = cpu_to_be16(size + sizeof(Header));
141  hdr->identification = cpu_to_be16(m_id);
142  hdr->fragmentOffset = cpu_to_be16(0x4000); // dont fragment flag
143  hdr->timeToLive = 64;
144  hdr->protocol = type;
145  hdr->source = cpu_to_be32(m_address);
146  hdr->destination = cpu_to_be32(destination);
147  hdr->checksum = 0;
148  hdr->checksum = checksum(hdr, sizeof(Header));
149  (*pkt)->size += sizeof(Header);
150  m_id++;
151 
152  // Success
153  return ESUCCESS;
154 }

156 const u16 IPV4::checksum(const void *buffer, Size len)
157 {
158  const u16 *ptr = (const u16 *) buffer;
159  uint sum = 0;
160 
161  // Calculate sum of the buffer
162  while (len > 1)
163  {
164  sum += *ptr;
165  ptr++;
166  len -= 2;
167  }
168 
169  // Add left-over byte, if any
170  if (len > 0)
171  sum += be16_to_cpu(*((u8 *)ptr) << 8);
172 
173  // Enforce 16-bit checksum
174  while (sum >> 16)
175  sum = (sum >> 16) + (sum & 0xFFFF);
176 
177  // Convert to one's complement
178  return (~sum);
179 }

182 {
183  DEBUG("");
184 
185  Header *hdr = (Header *) (pkt->data + offset);
186 
187  if (be32_to_cpu(hdr->destination) != m_address)
188  {
189  DEBUG("dropped packet");

192  switch (hdr->protocol)

194  case ICMP:

196  break;
197 
198  case UDP:

204  }
