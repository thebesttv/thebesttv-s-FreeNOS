40 int MPI_Init(int *argc, char ***argv)
41 {
42  SystemInformation info;
43  FileSystemMessage msg;
44  struct stat st;
45  char *programName = (*argv)[0];
46  char programPath[64];
47  u8 *programBuffer;
48  int fd;
49  Memory::Range memChannelBase;
50 
51  // If we are master (node 0):
52  if (info.coreId == 0)
53  {
54  msg.type = ChannelMessage::Request;
55  msg.action = ReadFile;
56  msg.from = SELF;
57  ChannelClient::instance->syncSendReceive(&msg, CORESRV_PID);
58 
59  // provide -n COUNT, --help and other stuff in here too.
60  // to influence the launching of more MPI programs
61  coreCount = msg.size;
62 
63  // Read our own ELF program to a buffer and pass it to CoreServer
64  // for creating new programs on the remote core.
65  if (strncmp(programName, "/bin/", 5) != 0)
66  snprintf(programPath, sizeof(programPath), "/bin/%s", programName);
67  else
68  strlcpy(programPath, programName, sizeof(programPath));
69 
70  if (stat(programPath, &st) != 0)
71  {
72  printf("%s: failed to stat '%s': %s\n",
73  programName, programPath, strerror(errno));
74  return MPI_ERR_BAD_FILE;
75  }
76  programBuffer = new u8[st.st_size];
77  MemoryBlock::set(programBuffer, 0, st.st_size);
78 
79  // Read ELF program
80  if ((fd = open(programPath, O_RDONLY)) == -1)
81  {
82  printf("%s: failed to open '%s': %s\n",
83  programName, programPath, strerror(errno));
84  return MPI_ERR_BAD_FILE;
85  }
86  if (read(fd, programBuffer, st.st_size) != st.st_size)
87  {
88  printf("%s: failed to read '%s': %s\n",
89  programName, programPath, strerror(errno));
90  return MPI_ERR_BAD_FILE;
91  }
92  if (close(fd) != 0)
93  {
94  printf("%s: failed to close '%s': %s\n",
95  programName, programPath, strerror(errno));
96  return MPI_ERR_BAD_FILE;
97  }
98 
99  // Allocate memory space on the local processor for the whole
100  // UniChannel array, NxN communication with MPI.
101  // Then pass the channel offset physical address as an argument -addr 0x.... to spawn()
102  memChannelBase.size = (PAGESIZE * 2) * (msg.size * msg.size);
103  memChannelBase.phys = 0;
104  memChannelBase.virt = 0;
105  memChannelBase.access = Memory::Readable | Memory::Writable | Memory::User;
106  if (VMCtl(SELF, Map, &memChannelBase) != API::Success)
107  {
108  printf("%s: failed to allocate MemoryChannel\n",
109  programName);
110  return MPI_ERR_NO_MEM;
111  }
112  printf("%s: MemoryChannel at physical address %x\n",
113  programName, memChannelBase.phys);
114 
115  // Clear channel pages
116  MemoryBlock::set((void *) memChannelBase.virt, 0, memChannelBase.size);
117 
118  // now create the slaves using coreservers.
119  for (Size i = 1; i < coreCount; i++)
120  {
121  char *cmd = new char[MPI_PROG_CMDLEN];
122  snprintf(cmd, MPI_PROG_CMDLEN, "%s -a %x -c %d",
123  programPath, memChannelBase.phys, coreCount);
124 
125  for (int j = 1; j < *argc; j++)
126  {
127  strcat(cmd, " ");
128  strcat(cmd, (*argv)[j]);
129  }
130 
131  msg.type = ChannelMessage::Request;
132  msg.action = CreateFile;
133  msg.size = i;
134  msg.buffer = (char *) programBuffer;
135  msg.offset = st.st_size;
136  msg.path = cmd;
137  ChannelClient::instance->syncSendReceive(&msg, CORESRV_PID);
138 
139  if (msg.result != ESUCCESS)
140  {
141  printf("%s: failed to create process on core%d\n",
142  programName, i);
143  return MPI_ERR_SPAWN;
144  }
145  }
146  }
147  else
148  {
149  // If we are slave (node N):
150  // read the -addr argument, and map the UniChannels into our address space.
151  for (int i = 1; i < (*argc); i++)
152  {
153  if (!strcmp((*argv)[i], "--addr") ||
154  !strcmp((*argv)[i], "-a"))
155  {
156  if ((*argc) < i+1)
157  return MPI_ERR_ARG;
158 
159  String s = (*argv)[i+1];
160  memChannelBase.phys = s.toLong(Number::Hex);
161  i++;
162  }
163  else if (!strcmp((*argv)[i], "--cores") ||
164  !strcmp((*argv)[i], "-c"))
165  {
166  if ((*argc) < i+1)
167  return MPI_ERR_ARG;
168  coreCount = atoi((*argv)[i+1]);
169  i++;
170  }
171  // Unknown MPI argument. Pass the rest to the user program.
172  else
173  {
174  (*argc) -= (i-1);
175  (*argv)[i-1] = (*argv)[0];
176  (*argv) += (i-1);
177  break;
178  }
179  }
180  }
181 
182  // Create MemoryChannels
183  readChannel = new Index<MemoryChannel>(coreCount);
184  writeChannel = new Index<MemoryChannel>(coreCount);
185 
186  // Fill read channels
187  for (Size i = 0; i < coreCount; i++)
188  {
189  MemoryChannel *ch = new MemoryChannel();
190  ch->setMode(MemoryChannel::Consumer);
191  ch->setMessageSize(sizeof(MPIMessage));
192  ch->setPhysical(MEMBASE(info.coreId) + (PAGESIZE * 2 * i),
193  MEMBASE(info.coreId) + (PAGESIZE * 2 * i) + PAGESIZE);
194  readChannel->insert(i, *ch);
195 
196  if (info.coreId == 0)
197  {
198  printf("%s: read: core%d: data=%x feedback=%x base%d=%x\n", (*argv)[0], i,
199  MEMBASE(info.coreId) + (PAGESIZE * 2 * i),
200  MEMBASE(info.coreId) + (PAGESIZE * 2 * i) + PAGESIZE,
201  i, MEMBASE(i));
202  }
203  }
204 
205  // Fill write channels
206  for (Size i = 0; i < coreCount; i++)
207  {
208  MemoryChannel *ch = new MemoryChannel();
209  ch->setMode(MemoryChannel::Producer);
210  ch->setMessageSize(sizeof(MPIMessage));
211  ch->setPhysical(MEMBASE(i) + (PAGESIZE * 2 * info.coreId),
212  MEMBASE(i) + (PAGESIZE * 2 * info.coreId) + PAGESIZE);
213  writeChannel->insert(i, *ch);
214 
215  if (info.coreId == 0)
216  {
217  printf("%s: write: core%d: data=%x feedback=%x base%d=%x\n", (*argv)[0], i,
218  MEMBASE(i) + (PAGESIZE * 2 * info.coreId),
219  MEMBASE(i) + (PAGESIZE * 2 * info.coreId) + PAGESIZE,
220  i, MEMBASE(i));
221  }
222  }
223 
224  return MPI_SUCCESS;
225 }
