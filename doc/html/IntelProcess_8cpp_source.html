30 Process::Result IntelProcess::initialize()
31 {
32  Address stackSize, stackAddr;
33  Memory::Range range;
34  CPUState *regs;
35  u16 dataSel = m_privileged ? KERNEL_DS_SEL : USER_DS_SEL;
36  u16 codeSel = m_privileged ? KERNEL_CS_SEL : USER_CS_SEL;
37 
38  // Create MMU context
39  m_memoryContext = new IntelPaging(
40  &m_map,
41  Kernel::instance->getAllocator()
42  );
43  if (!m_memoryContext)
44  return OutOfMemory;
45 
46  // User stack (high memory).
47  range = m_map.range(MemoryMap::UserStack);
48  range.access = Memory::Readable | Memory::Writable | Memory::User;
49  if (Kernel::instance->getAllocator()->allocate(&range.size, &range.phys) != Allocator::Success)
50  return OutOfMemory;
51 
52  if (m_memoryContext->mapRange(&range) != MemoryContext::Success)
53  return MemoryMapError;
54  setUserStack(range.virt + range.size - MEMALIGN);
55 
56  // Kernel stack (low memory).
57  stackSize = KernelStackSize;
58  if (Kernel::instance->getAllocator()->allocateLow(stackSize, &stackAddr) != Allocator::Success)
59  return OutOfMemory;
60 
61  stackAddr = (Address) Kernel::instance->getAllocator()->toVirtual(stackAddr);
62  m_kernelStackBase = stackAddr + stackSize;
63  setKernelStack(m_kernelStackBase - sizeof(CPUState)
64  - sizeof(IRQRegs0)
65  - sizeof(CPURegs));
66 
67  // Fill kernel stack with initial (user)registers to restore
68  // loadCoreState: struct CPUState
69  regs = (CPUState *) m_kernelStackBase - 1;
70  MemoryBlock::set(regs, 0, sizeof(CPUState));
71  regs->seg.ss0 = KERNEL_DS_SEL;
72  regs->seg.fs = dataSel;
73  regs->seg.gs = dataSel;
74  regs->seg.es = dataSel;
75  regs->seg.ds = dataSel;
76  regs->regs.ebp = m_userStack;
77  regs->regs.esp0 = m_kernelStack;
78  regs->irq.eip = m_entry;
79  regs->irq.cs = codeSel;
80  regs->irq.eflags = INTEL_EFLAGS_DEFAULT |
81  INTEL_EFLAGS_IRQ;
82  regs->irq.esp3 = m_userStack;
83  regs->irq.ss3 = dataSel;
84 
85  // restoreState: iret
86  IRQRegs0 *irq = (IRQRegs0 *) regs - 1;
87  irq->eip = (Address) loadCoreState;
88  irq->cs = KERNEL_CS_SEL;
89  irq->eflags = INTEL_EFLAGS_DEFAULT;
90 
91  // restoreState: popa
92  CPURegs *pusha = (CPURegs *) irq - 1;
93  MemoryBlock::set(pusha, 0, sizeof(CPURegs));
94  pusha->ebp = m_kernelStackBase - sizeof(CPURegs);
95  pusha->esp0 = pusha->ebp;
96 
97  // Finalize with generic initialization
98  return Process::initialize();
99 }
