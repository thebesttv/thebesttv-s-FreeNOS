50 ChannelClient::Result ChannelClient::connect(ProcessID pid, Size messageSize)
51 {
52  Address prodAddr, consAddr;
53  SystemInformation info;
54 
55  // Allocate consumer
56  MemoryChannel *cons = new MemoryChannel;
57  if (!cons)
58  {
59  return OutOfMemory;
60  }
61  cons->setMessageSize(messageSize);
62  cons->setMode(Channel::Consumer);
63 
64  // Allocate producer
65  MemoryChannel *prod = new MemoryChannel;
66  if (!prod)
67  {
68  delete cons;
69  return OutOfMemory;
70  }
71  prod->setMessageSize(messageSize);
72  prod->setMode(Channel::Producer);
73 
74  // Call VMShare to create shared memory mapping for MemoryChannel.
75  ProcessShares::MemoryShare share;
76  share.pid = pid;
77  share.coreId = info.coreId;
78  share.tagId = 0;
79  share.range.size = PAGESIZE * 4;
80  share.range.virt = 0;
81  share.range.phys = 0;
82  share.range.access = Memory::User | Memory::Readable | Memory::Writable;
83 
84  // Create shared memory mapping
85  Error r = VMShare(pid, API::Create, &share);
86  switch (r)
87  {
88  case API::Success:
89  {
90  prodAddr = share.range.virt;
91  consAddr = share.range.virt + (PAGESIZE * 2);
92  break;
93  }
94  case API::AlreadyExists:
95  {
96  prodAddr = share.range.virt + (PAGESIZE * 2);
97  consAddr = share.range.virt;
98  break;
99  }
100  default:
101  {
102  return IOError;
103  }
104  }
105 
106  // Setup producer memory address
107  if (prod->setVirtual(prodAddr, prodAddr + PAGESIZE) != MemoryChannel::Success)
108  {
109  delete prod;
110  delete cons;
111  return IOError;
112  }
113 
114  // Setup consumer memory address
115  if (cons->setVirtual(consAddr, consAddr + PAGESIZE) != MemoryChannel::Success)
116  {
117  delete prod;
118  delete cons;
119  return IOError;
120  }
121 
122  // Register channels
123  m_registry->registerConsumer(pid, cons);
124  m_registry->registerProducer(pid, prod);
125  return Success;
126 }

141 ChannelClient::Result ChannelClient::sendRequest(ProcessID pid,
142  void *buffer,
143  CallbackFunction *callback)
144 {
145  Request *req = 0;
146  Size identifier = 0;
147  Channel *ch = findProducer(pid);
148  if (!ch)
149  return NotFound;
150 
151  // Find request object
152  for (Size i = 0; i < m_requests.count(); i++)
153  {
154  req = (Request *) m_requests.get(i);
155  if (!req->active)
156  {
157  identifier = i;
158  break;
159  }
160  }
161  // Allocate new request object if none available
162  if (!req || req->active)
163  {
164  req = new Request;
165  req->message = (ChannelMessage *) new u8[ch->getMessageSize()];
166  identifier = m_requests.insert(*req);
167  }
168  // Fill request object
169  MemoryBlock::copy(req->message, buffer, ch->getMessageSize());
170  req->pid = pid;
171  req->message->identifier = identifier;
172  req->message->type = ChannelMessage::Request;
173  req->callback = callback;
174  req->active = true;
175 
176  DEBUG("sending request with id = " << req->message->identifier << " to PID " << pid);
177 
178  // Try to send the message
179  if (ch->write(req->message) != Channel::Success)
180  {
181  req->active = false;
182  return IOError;
183  }
184  // Wakeup the receiver
185  ProcessCtl(pid, Resume, 0);
186  return Success;
187 }

189 ChannelClient::Result ChannelClient::processResponse(ProcessID pid,
190  ChannelMessage *msg)
191 {
192  Size count = m_requests.count();
193 
194  for (Size i = 0; i < count; i++)
195  {
196  Request *req = (Request *) m_requests.get(i);
197 
198  if (req->active &&
199  req->pid == pid &&
200  req->message->identifier == msg->identifier)
201  {
202  req->callback->execute(msg);
203  req->active = false;
204  return Success;
205  }
206  }
207  return NotFound;

251 ChannelClient::Result ChannelClient::syncSendTo(void *buffer, ProcessID pid)
252 {
253  Channel *ch = findProducer(pid);
254  if (!ch)
255  return NotFound;
256 
257  while (true)
258  {
259  switch (ch->write(buffer))
260  {
261  case Channel::Success:
262  ProcessCtl(pid, Resume, 0);
263  return Success;
264 
265  case Channel::ChannelFull:
266  ProcessCtl(pid, Resume, 0);
267  break;
268 
269  default:
270  return IOError;
271  }
272  ProcessCtl(SELF, Schedule, 0);
273  }
274  return IOError;
275 }
