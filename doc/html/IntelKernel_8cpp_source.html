37 IntelKernel::IntelKernel(CoreInfo *info)
38  : Kernel(info)
39 {
40 
41  IntelMap map;
42  IntelCore core;
43  IntelPaging memContext(&map, core.readCR3(), m_alloc);
44 
45  // Refresh MemoryContext::current()
46  memContext.activate();
47 
48  // Install interruptRun() callback
49  interruptRun = ::executeInterrupt;
50 
51  // Setup exception handlers
52  for (int i = 0; i < 17; i++)
53  {
54  hookIntVector(i, exception, 0);
55  }
56  // Setup IRQ handlers
57  for (int i = 17; i < 256; i++)
58  {
59  // Trap gate
60  if (i == 0x90)
61  hookIntVector(0x90, trap, 0);
62 
63  // Hardware Interrupt
64  else
65  hookIntVector(i, interrupt, 0);
66  }
67 
68  // Only core0 uses PIC and PIT.
69  if (info->coreId == 0)
70  {
71  // Set PIT interrupt frequency to 250 hertz
72  m_pit.setFrequency(250);
73 
74  // Configure the master and slave PICs
75  m_pic.initialize();
76  m_intControl = &m_pic;
77  }
78  else
79  m_intControl = 0;
80 
81  // Try to configure the APIC.
82  if (m_apic.initialize() == Timer::Success)
83  {
84  NOTICE("Using APIC timer");
85 
86  // Enable APIC timer interrupt
87  hookIntVector(m_apic.getInterrupt(), clocktick, 0);
88 
89  m_timer = &m_apic;
90 
91  if (m_coreInfo->timerCounter == 0)
92  {
93  m_apic.start(&m_pit);
94  m_coreInfo->timerCounter = m_apic.getCounter();
95  }
96  else
97  m_apic.start(m_coreInfo->timerCounter, m_pit.getFrequency());
98  }
99  // Use PIT as system timer.
100  else
101  {
102  NOTICE("Using PIT timer");
103  m_timer = &m_pit;
104 
105  // Install PIT interrupt vector handler
106  hookIntVector(m_intControl->getBase() +
107  m_pit.getInterrupt(), clocktick, 0);
108 
109  // Enable PIT interrupt
110  enableIRQ(m_pit.getInterrupt(), true);
111  }
112 
113  // Initialize TSS Segment
114  Address tssAddr = (Address) &kernelTss;
115  gdt[KERNEL_TSS].limitLow = sizeof(TSS) + (0xfff / 8);
116  gdt[KERNEL_TSS].baseLow = (tssAddr) & 0xffff;
117  gdt[KERNEL_TSS].baseMid = (tssAddr >> 16) & 0xff;
118  gdt[KERNEL_TSS].type = 9;
119  gdt[KERNEL_TSS].privilege = 0;
120  gdt[KERNEL_TSS].present = 1;
121  gdt[KERNEL_TSS].limitHigh = 0;
122  gdt[KERNEL_TSS].granularity = 8;
123  gdt[KERNEL_TSS].baseHigh = (tssAddr >> 24) & 0xff;
124 
125  // Fill the Task State Segment (TSS).
126  MemoryBlock::set(&kernelTss, 0, sizeof(TSS));
127  kernelTss.ss0 = KERNEL_DS_SEL;
128  kernelTss.esp0 = 0;
129  kernelTss.bitmap = sizeof(TSS);
130  ltr(KERNEL_TSS_SEL);
131 }

170 void IntelKernel::clocktick(CPUState *state, ulong param)
171 {
172  IntelKernel *kern = (IntelKernel *) Kernel::instance;
173  Size irq = kern->m_timer->getInterrupt();
174 
175  kern->enableIRQ(irq, true);
176 
177  // Ensure the APIC timer gets end-of-interrupt
178  if (irq == kern->m_apic.getInterrupt())
179  kern->m_apic.clear(irq);
180 
181  kern->m_timer->tick();
182  kern->getProcessManager()->schedule();
