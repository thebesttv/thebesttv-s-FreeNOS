62 Error Keyboard::read(IOBuffer & buffer, Size size, Size offset)
63 {
64  Error ret = EAGAIN;
65 
66  // Do we have any new key events?
67  if (pending)
68  {
69  pending = false;
70 
71  // Read byte from the keyboard.
72  u8 keycode = ReadByte(PS2_PORT);
73 
74  // Update shift state
75  if (keycode == 0x2a || keycode == 0xaa)
76  {
77  shiftState ^= 1;
78  }
79  // Don't do anything on release
80  else if (!(keycode & PS2_RELEASE) &&
81  (keymap[keycode & 0x7f][shiftState]))
82  {
83  // Write to buffer
84  buffer.write((void *) &keymap[keycode & 0x7f][shiftState], 1);
85  ret = 1;
86  }
87  // Re-enable interrupt
88  ProcessCtl(SELF, EnableIRQ, PS2_IRQ);
89  }
90  return ret;
91 }
