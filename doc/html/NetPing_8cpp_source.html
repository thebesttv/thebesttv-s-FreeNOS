89 NetPing::Result NetPing::arpPing(const char *dev,
90  const char *host)
91 {
92  DEBUG("");
93 
94  NetworkClient client(dev);
95  Ethernet::Address ethAddr;
96  IPV4::Address ipAddr;
97  int sock;
98 
99  // Initialize networking client
100  if (client.initialize() != NetworkClient::Success)
101  {
102  ERROR("failed to initialize network client for device: " << dev);
103  return IOError;
104  }
105 
106  // Create an ARP socket
107  if (client.createSocket(NetworkClient::ARP, &sock) != NetworkClient::Success)
108  {
109  ERROR("failed to create ARP socket");
110  return IOError;
111  }
112 
113  // Convert to IPV4 address
114  if (!(ipAddr = IPV4::toAddress(host)))
115  {
116  ERROR("failed to convert to IPV4 address: " << host);
117  return IOError;
118  }
119  printf("Sending ARP request to %s\n", host);
120 
121  // Send ARP request
122  if (::write(sock, &ipAddr, sizeof(ipAddr)) < 0)
123  {
124  ERROR("failed to send ARP request: " << strerror(errno));
125  return IOError;
126  }
127  // Receive ARP reply, if any
128  if (::read(sock, &ethAddr, sizeof(ethAddr)) < 0)
129  {
130  ERROR("failed to receive ARP response: " << strerror(errno));
131  return IOError;
132  }
133  // Print the MAC address received
134  printf("Received reply for: ");
135 
136  for (Size i = 0; i < sizeof(Ethernet::Address); i++)
137  printf("%x:", ethAddr.addr[i]);
138  printf("\n");
139 
140  // Finished
141  client.close(sock);
142  return Success;
143 }

145 NetPing::Result NetPing::icmpPing(const char *dev, const char *host)
146 {
147  DEBUG("");
148 
149  NetworkClient client(dev);
150  int sock;
151 
152  // Initialize networking client
153  if (client.initialize() != NetworkClient::Success)
154  {
155  ERROR("failed to initialize network client for device: " << dev);
156  return IOError;
157  }
158 
159  // Create an ICMP socket
160  if (client.createSocket(NetworkClient::ICMP, &sock) != NetworkClient::Success)
161  {
162  ERROR("failed to create ARP socket");
163  return IOError;
164  }
165  // Connect socket to the given host
166  if (client.connectSocket(sock, IPV4::toAddress(host)) != NetworkClient::Success)
167  {
168  ERROR("failed to connect ICMP socket");
169  return IOError;
170  }
171  // Send an echo request
172  ICMP::Header msg;
173  msg.type = ICMP::EchoRequest;
174  msg.code = 0;
175  msg.checksum = 0;
176  msg.id = 1;
177  msg.sequence = 1;
178 
179  // Generate checksum
180  msg.checksum = ICMP::checksum(&msg);
181 
182  // Send the packet
183  if (::write(sock, &msg, sizeof(msg)) <= 0)
184  {
185  ERROR("failed to send ICMP request");
186  return IOError;
187  }
188  // Receive echo reply
189  if (::read(sock, &msg, sizeof(msg)) <= 0)
190  {
191  ERROR("failed to receive ICMP response");
192  return IOError;
193  }
194  // Check message type
195  if (msg.type != ICMP::EchoReply)
196  {
197  ERROR("invalid ICMP code in response: " << (int) msg.type);
198  return IOError;
199  }
200  // Print the ICMP address received
201  printf("Received ICMP response with id=%d sequence=%d\n",
202  msg.id, msg.sequence);
203 
204  // Finished
205  ::close(sock);
206  return Success;
207 }
