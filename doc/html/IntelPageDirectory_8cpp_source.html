41 IntelPageTable * IntelPageDirectory::getPageTable(Address virt, SplitAllocator *alloc) const
42 {

45  // Check if the page table is present.

64 MemoryContext::Result IntelPageDirectory::map(Address virt,
65  Address phys,
66  Memory::Access access,
67  SplitAllocator *alloc)
68 {
69  IntelPageTable *table = getPageTable(virt, alloc);
70  Address addr;
71 
72  // Check if the page table is present.
73  if (!table)
74  {
75  // Allocate a new page table
76  if (alloc->allocateLow(sizeof(IntelPageTable), &addr) != Allocator::Success)
77  return MemoryContext::OutOfMemory;
78 
79  MemoryBlock::set(alloc->toVirtual(addr), 0, sizeof(IntelPageTable));
80 
81  // Assign to the page directory
82  m_tables[ DIRENTRY(virt) ] = addr | PAGE_PRESENT | PAGE_WRITE | flags(access);
83  table = getPageTable(virt, alloc);
84  }
85  return table->map(virt, phys, access);
86 }

97 MemoryContext::Result IntelPageDirectory::translate(Address virt,
98  Address *phys,
99  SplitAllocator *alloc) const
100 {
101  IntelPageTable *table = getPageTable(virt, alloc);
102  if (!table)
103  {
104  if (m_tables[DIRENTRY(virt)] & PAGE_SECTION)
105  {
106  *phys = (m_tables[DIRENTRY(virt)] & PAGEMASK) + ((virt % MegaByte(4)) & PAGEMASK);
107  return MemoryContext::Success;
108  }
109  return MemoryContext::InvalidAddress;
110  }
111  else
112  return table->translate(virt, phys);
113 }

136 MemoryContext::Result IntelPageDirectory::releaseRange(Memory::Range range,
137  SplitAllocator *alloc,
138  bool tablesOnly)
139 {
140  Address phys;
141 
142  // Walk the page directory within the specified range
143  for (Size i = 0; i < range.size; i += MegaByte(4))
144  {
145  IntelPageTable *table = getPageTable(range.virt + i, alloc);
146  if (table)
147  {
148  // Release mapped pages
149  if (!tablesOnly)
150  {
151  for (Size j = 0; j < MegaByte(4); j += PAGESIZE)
152  {
153  if (table->translate(range.virt + i + j, &phys) == MemoryContext::Success)
154  {
155  alloc->release(phys);
156  }
157  }
158  }
159  // Release page table
160  alloc->release(m_tables[ DIRENTRY(range.virt + i) ] & PAGEMASK);
161  }
162  }
163  return MemoryContext::Success;
164 }
