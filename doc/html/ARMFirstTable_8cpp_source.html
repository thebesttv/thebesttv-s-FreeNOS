107 ARMSecondTable * ARMFirstTable::getSecondTable(Address virt, SplitAllocator *alloc) const
108 {
109  u32 entry = m_tables[ DIRENTRY(virt) ];
110 
111  // Check if the page table is present.
112  if (!(entry & PAGE1_TABLE))
113  return ZERO;
114  else
115  return (ARMSecondTable *) alloc->toVirtual(entry & PAGEMASK);

118 MemoryContext::Result ARMFirstTable::map(Address virt,
119  Address phys,
120  Memory::Access access,
121  SplitAllocator *alloc)
122 {
123  ARMSecondTable *table = getSecondTable(virt, alloc);
124  Address addr;
125  Arch::Cache cache;
126 
127  // Input addresses must be aligned on pagesize boundary
128  if ((phys & ~PAGEMASK) || (virt & ~PAGEMASK))
129  return MemoryContext::InvalidAddress;
130 
131  // Check if the page table is present.
132  if (!table)
133  {
134  // Reject if already mapped as a (super)section
135  if (m_tables[ DIRENTRY(virt) ] & PAGE1_SECTION)
136  return MemoryContext::AlreadyExists;
137 
138  // Allocate a new page table
139  if (alloc->allocateLow(sizeof(ARMSecondTable), &addr) != Allocator::Success)
140  return MemoryContext::OutOfMemory;
141 
142  MemoryBlock::set(alloc->toVirtual(addr), 0, PAGESIZE);
143 
144  // Assign to the page directory. Do not assign permission flags (only for direct sections).
145  m_tables[ DIRENTRY(virt) ] = addr | PAGE1_TABLE;
146  cache.cleanData(&m_tables[DIRENTRY(virt)]);
147  table = getSecondTable(virt, alloc);
148  }
149  return table->map(virt, phys, access);
150 }

152 MemoryContext::Result ARMFirstTable::mapLarge(Memory::Range range,
153  SplitAllocator *alloc)
154 {
155  Arch::Cache cache;
156 
157  if (range.size & 0xfffff)
158  return MemoryContext::InvalidSize;
159 
160  if ((range.phys & ~PAGEMASK) || (range.virt & ~PAGEMASK))
161  return MemoryContext::InvalidAddress;
162 
163  for (Size i = 0; i < range.size; i += MegaByte(1))
164  {
165  if (m_tables[ DIRENTRY(range.virt + i) ] & (PAGE1_TABLE | PAGE1_SECTION))
166  return MemoryContext::AlreadyExists;
167 
168  m_tables[ DIRENTRY(range.virt + i) ] = (range.phys + i) | PAGE1_SECTION | flags(range.access);
169  cache.cleanData(&m_tables[DIRENTRY(range.virt + i)]);
170  }
171  return MemoryContext::Success;
172 }

216 u32 ARMFirstTable::flags(Memory::Access access) const
217 {
218  u32 f = PAGE1_AP_SYS;
219 
220  // Permissions
221  if (!(access & Memory::Executable)) f |= PAGE1_NOEXEC;
222  if ((access & Memory::User)) f |= PAGE1_AP_USER;
223  if (!(access & Memory::Writable)) f |= PAGE1_APX;
224 
225  // Caching
226  if (access & Memory::Device) f |= PAGE1_DEVICE_SHARED;
227  else if (access & Memory::Uncached) f |= PAGE1_UNCACHED;
228  else f |= PAGE1_CACHE_WRITEBACK;
229 
230  return f;
231 }

233 MemoryContext::Result ARMFirstTable::releaseRange(Memory::Range range,
234  SplitAllocator *alloc,
235  bool tablesOnly)
236 {

238 
239  // Walk the page directory within the specified range
240  for (Size i = 0; i < range.size; i += MegaByte(1))
241  {
242  ARMSecondTable *table = getSecondTable(range.virt + i, alloc);
243  if (table)
244  {
245  // Release mapped pages
246  if (!tablesOnly)
247  {
248  for (Size j = 0; j < MegaByte(1); j += PAGESIZE)
249  {

251  {
252  alloc->release(phys);
253  }
254  }
255  }
256  // Release page table
257  alloc->release(m_tables[ DIRENTRY(range.virt + i) ] & PAGEMASK);
258  }
259  }
260  return MemoryContext::Success;
