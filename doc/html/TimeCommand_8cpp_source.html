44  // If not, try to execute it as a file directly. */

49  // Try to find it on the livecd filesystem. (temporary hardcoded PATH)
50  else if (snprintf(tmp, sizeof(tmp), "/bin/%s", params[0]) &&
51  (pid = forkexec(tmp, (const char **) params)) != -1)

54  }
55  else
