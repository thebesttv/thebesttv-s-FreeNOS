46 void BitArray::set(Size bit, bool value)
47 {
48  assertRead(m_array);
49  assertWrite(m_array);
50 
51  // Check if the bit is inside the array
52  if (bit >= m_size)
53  return;
54 
55  // Check current value
56  bool current = m_array[bit / 8] & (1 << (bit % 8));
57 
58  // Update the bit only if needed (and update administration)
59  if (current != value)
60  {
61  if (value)
62  {
63  m_array[bit / 8] |= 1 << (bit % 8);
64  m_set++;
65  }
66  else
67  {
68  m_array[bit / 8] &= ~(1 << (bit % 8));
69  m_set--;
70  }
71  }
72 }

93 BitArray::Result BitArray::setNext(Size *bit, Size count, Size start, Size boundary)
94 {
95  Size from = 0, found = 0;
96 
97  // Loop BitArray for unset bits
98  for (Size i = start; i < m_size; i++)
99  {
100  if (!isSet(i))
101  {
102  // Remember this bit
103  if (!found)
104  {
105  if (i % boundary)
106  continue;
107  from = i;
108  found = 1;
109  }
110  else
111  found++;
112 
113  // Are there enough contigious bits?
114  if (found >= count)
115  {
116  setRange(from, i);
117  *bit = from;
118  return Success;
119  }
120  }
121  else
122  {
123  from = found = 0;
124  }
125  }
126  // No unset bits left!
