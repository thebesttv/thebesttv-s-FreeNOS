25 BroadcomSPI::Result BroadcomSPI::initialize()
26 {
27  // Map I2C registers
28  if (m_io.map(IO_BASE + SPI0, PAGESIZE,
29  Memory::User|Memory::Readable|Memory::Writable|Memory::Device) != IO::Success)
30  return IOError;
31 
32  // Initialize GPIO controller
33  if (m_gpio.initialize() != BroadcomGPIO::Success)
34  return IOError;
35 
36  // Set GPIO pins to ALT0 function for CLK, MOSI, MISO, CS pins
37  m_gpio.setAltFunction(8, BroadcomGPIO::Function0); // CS0
38  m_gpio.setAltFunction(9, BroadcomGPIO::Function0); // MISO
39  m_gpio.setAltFunction(10, BroadcomGPIO::Function0); // MOSI
40  m_gpio.setAltFunction(11, BroadcomGPIO::Function0); // CLK
41 
42  // Reset registers and clear FIFO
43  m_io.write(Control, 0);
44  m_io.write(Control, ClearFIFO);
45  return Success;
46 }

54 BroadcomSPI::Result BroadcomSPI::transfer(u8 *sendBuf, u8 *recvBuf, Size size)
55 {
56  uint rx = 0, tx = 0;
57  DEBUG("transfer of " << size << " bytes");
58 
59  // Set modes
60  m_io.write(Control, 0);
61 
62  // Clear FIFOs
63  m_io.write(Control, ClearFIFO);
64 
65  // Start transfer
66  m_io.set(Control, TransferActive);
67 
68  // Keep transfering until we reach the length needed
69  while (rx < size || tx < size)
70  {
71  // Append to tx FIFO
72  while (m_io.read(Control) & TxHasSpace && tx < size)
73  {
74  DEBUG("tx[" << tx << "] = " << sendBuf[tx]);
75  m_io.write(FIFO, sendBuf[tx++]);
76  }
77 
78  // Pull from rx FIFO
79  while (m_io.read(Control) & RxHasData && rx < size)
80  {
81  recvBuf[rx] = m_io.read(FIFO);
82  DEBUG("rx[" << rx << "] = " << recvBuf[rx]);
83  rx++;
84  }
85  }
86  DEBUG("waiting for done");
87 
88  // Wait for DONE signal
89  while (m_io.read(Control) & TransferDone);
90 
91  return Success;
92 }
