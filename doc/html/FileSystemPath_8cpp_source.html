64 void FileSystemPath::parse(const char *p, char sep)
65 {
66  const char *cur = p;
67  char *parentStr = ZERO;
68 
69  // Skip heading separators
70  while (*cur && *cur == sep) cur++;
71 
72  // Save parameters
73  p = (char *) cur;
74  m_separator = sep;
75  m_fullLength = strlen((char *)cur);
76  m_fullPath = new String(cur);
77  String str(p);
78 
79  // Split the path into parts
80  List<String> parts = str.split(sep);
81  for (ListIterator<String> i(parts); i.hasCurrent(); i++)
82  m_path.append(new String(i.current()));
83 
84  // Create parent, if any
85  if (m_path.head() && m_path.head()->next)
86  {
87  // Allocate buffer
88  parentStr = new char[strlen(p)];
89  memset(parentStr, 0, strlen(p));
90 
91  // Construct parent path
92  for (List<String *>::Node *l = m_path.head(); l && l->next; l = l->next)
93  {
94  strcat(parentStr, **l->data);
95  if (l->next && l->next->next)
96  strncat(parentStr, &m_separator, 1);
97  }
98  // Save the path, then release buffer
99  m_parentPath = new String(parentStr);
100  delete parentStr;
101  }
102 }
