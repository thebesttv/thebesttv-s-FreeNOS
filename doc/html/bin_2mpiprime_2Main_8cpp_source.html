35 void collect(int n, unsigned *map)
36 {
37  int i, j, sqrt_of_n = sqrt(n), z;
38  MPI_Status status;
39  unsigned *mybuf = (unsigned *) malloc(sizeof(unsigned) * PERNODE(rank, total, sqrt_of_n, n));
40 
41  // Every worker sends it's results to the master
42  if (rank != 0)
43  {
44  // Send mybuf to the master
45  MPI_Send(&map[START(rank, total, sqrt_of_n, n)], PERNODE(rank, total, sqrt_of_n, n), MPI_INT, 0, 0, MPI_COMM_WORLD);
46  }
47  // The master gathers the parts of the list from every worker
48  else
49  {
50  for (i = 1; i < total; i++)
51  {
52  // Receive from worker
53  MPI_Recv(mybuf, PERNODE(rank, total, sqrt_of_n, n), MPI_INT, i, 0, MPI_COMM_WORLD, &status);
54 
55  // Copy inside our buffer
56  for (j = START(i, total, sqrt_of_n, n), z = 0; j < END(i, total, sqrt_of_n, n); j++)
57  {
58  map[j] = mybuf[z++];
59  }
60  }
61  }
62  // Cleanup
63  free(mybuf);
64 }

66 void search_sequential(int k, int n, unsigned *map, int argc, char **argv)
67 {
68  int i, j;

70  // Sequential algorithm
71  // Next is a prime
72  for (i = 2; i < n; i++)
73  {
74  // Prime number?
75  if (map[i])
76  {
77  // Mask off all multiples
78  for (j = i + 1; j < n; j++)
79  {
80  if (!(j % i))
81  map[j] = 0;

87 void search_parallel(int k, int n, unsigned *map, int argc, char **argv)
88 {
89  struct timeval t1, t2;
90  struct timezone tz;
91  gettimeofday(&t1, &tz);
92  int i, last, sqrt_of_n = sqrt(n);
93 
94  // Find all primes below sqrt(n) sequentially
95  search_sequential(k, sqrt_of_n, map, argc, argv);
96 
97  if (rank == 0)
98  {
99  gettimeofday(&t2, &tz);
100  printf("sequential: ");
101  printtimediff(&t1, &t2);
102  printf("\n");
103  gettimeofday(&t1, &tz);
104  }
105 
106  // Every worker calculates all primes k .. sqrt(n) sequentially
107  // and uses the result to mark it's part of the map, concurrently
108  // Note that no communication is needed
109  while (k < sqrt_of_n)
110  {
111  // Prime number?
112  if (!map[k])
113  {
114  k++;
115  continue;
116  }
117 
118  // Mark multiples of k in my range
119  i = START(rank, total, sqrt_of_n, n);
120  last = END(rank, total, sqrt_of_n, n);
121 
122  while (i < last)
123  {
124  // Do we need to unmark this number? (no prime)
125  if (!(i % k))
126  map[i] = 0;
127 
128  i++;
129  }
130  // Look for the next prime
131  k++;
132  }
133  if (rank == 0)
134  {
135  gettimeofday(&t2, &tz);
136  printf("parallel: ");
137  printtimediff(&t1, &t2);
138  printf("\n");
139  }
140 
141  // Collect results of all workers
142  gettimeofday(&t1, &tz);
143  collect(n, map);
144 
145  if (rank == 0)
146  {
147  gettimeofday(&t2, &tz);
148  printf("collect: ");
149  printtimediff(&t1, &t2);
150  printf("\n");
151  }
152 }
