28 void RaspiSerial::init()
29 {
30  // Disable PL011.
31  m_io.write(PL011_CR, 0x00000000);
32 
33  // Setup the GPIO pin 14 && 15.
34  // Disable pull up/down for all GPIO pins & delay for 150 cycles.
35  m_io.write(GPPUD, 0x00000000);
36  delay(150);
37 
38  // Disable pull up/down for pin 14,15 & delay for 150 cycles.
39  m_io.write(GPPUDCLK0, (1 << 14) | (1 << 15));
40  delay(150);
41 
42  // Write 0 to GPPUDCLK0 to make it take effect.
43  m_io.write(GPPUDCLK0, 0x00000000);
44 
45  // Clear pending interrupts.
46  m_io.write(PL011_ICR, 0x7FF);
47 
48  // Set integer & fractional part of baud rate.
49  m_io.write(PL011_IBRD, 26);
50  m_io.write(PL011_FBRD, 3);
51 
52  // Disable FIFO, use 8 bit data transmission, 1 stop bit, no parity
53  m_io.write(PL011_LCRH, PL011_LCRH_WLEN_8BIT);
54 
55  // Mask all interrupts.
56  m_io.write(PL011_IMSC, (1 << 1) | (1 << 4) | (1 << 5) |
57  (1 << 6) | (1 << 7) | (1 << 8) |
58  (1 << 9) | (1 << 10));
59 
60  // Enable PL011, receive & transfer part of UART.
61  m_io.write(PL011_CR, (1 << 0) | (1 << 8) | (1 << 9));
62 }

64 void RaspiSerial::put(u8 byte)
65 {
66  // wait for UART to become ready to transmit
67  while(true)
68  if (!(m_io.read(PL011_FR) & (1 << 5)))
69  break;
