265 bool String::match(const char *mask) const
266 {
267  const char *string = m_string;
268  const char *end = ZERO;
269 
270  while (*string && *mask)
271  {
272  if (Character::isWildcard(*mask))
273  {
274  // Skip extra wildcards
275  while (Character::isWildcard(*mask))
276  mask++;
277 
278  // Find end of the string after the mask, if any.
279  for (end = mask; *end && !Character::isWildcard(*end); end++)
280  ;
281 
282  // If the wildcard was last, its a match.
283  if (mask == end)
284  return true;
285 
286  // Move the string forward until a match
287  for (; *string; string++)
288  {
289  const char *s = string, *m = mask;
290 
291  while (*s && *m && *s == *m)
292  s++, m++;
293 
294  if (m == end)
295  {
296  break;
297  }
298  }
299  }
300  else if (*string != *mask)
301  break;
302 
303  if (*string) string++;
304  if (*mask) mask++;
305  }
306  while (Character::isWildcard(*mask))
307  mask++;
308 
309  return (*string == *mask);
310 }

457 long String::toLong(Number::Base base)
458 {
459  const char *s = m_string;
460  long acc = 0, cutoff;
461  bool negative = false;
462  int cutlim, basenum = 10;
463 
464  // Set the number base
465  switch (base)
466  {
467  case Number::Dec: basenum = 10; break;
468  case Number::Hex: basenum = 16; break;
469  }
470 
471  // Skip whitespace
472  while (Character::isWhitespace(*s))
473  s++;
474 
475  // Negative number?
476  if (s[0] == '-')
477  {
478  negative = true; s++;
479  }
480 
481  // Skip '0x' prefix
482  if (basenum == 16 && s[0] == '0' && s[1] == 'x')
483  s += 2;
484 
485  // Compute output range limits
486  cutoff = negative ? LONG_MIN : LONG_MAX;
487  cutlim = cutoff % basenum;
488  cutoff /= basenum;
489 
490  if (negative) {
491  if (cutlim > 0) {
492  cutlim -= base;
493  cutoff += 1;
494  }
495  cutlim = -cutlim;
496  }
497 
498  while (1)
499  {
500  unsigned char c = (unsigned char) *s++;
501 
502  if (Character::isDigit(c))
503  c -= '0';
504  else if (Character::isAlpha(c))
505  c -= Character::isUpper(c) ? 'A' - 10 : 'a' - 10;
506  else
507  break;
508 
509  if (c >= basenum)
510  break;
511 
512  if (negative)
513  {
514  if (acc < cutoff || (acc == cutoff && c > cutlim))
515  {
516  acc = LONG_MIN;
517  } else {
518  acc *= basenum;
519  acc -= c;
520  }
521  } else {
522  if (acc > cutoff || (acc == cutoff && c > cutlim))
523  {
524  acc = LONG_MAX;
525  } else {
526  acc *= basenum;
527  acc += c;
528  }
529  }
530  }
531  return acc;
532 }

539 Size String::setUnsigned(ulong number, Number::Base base, char *string, bool sign)
540 {
541  char *p, *p1, *p2, *saved, tmp;
542  unsigned long ud = number;
543  int remainder, divisor = 10;
544  Size written = 0;
545 
546  // If needed, make sure enough allocated space is available.
547  if (!string)
548  reserve(STRING_DEFAULT_SIZE - 1);
549 
550  // Set target buffer
551  p = string ? string : m_string;
552 
553  // Set divider according to the number system base.
554  switch (base)
555  {
556  case Number::Dec: divisor = 10; break;
557  case Number::Hex: divisor = 16; break;
558  };
559 
560  // Negative prefix.
561  if (sign && (long)number < 0)
562  {
563  *p++ = '-';
564  ud = -number;
565  written++;
566  }
567  // Add '0x' prefix for hexadecimal numbers
568  if (base == Number::Hex)
569  {
570  *p++ = '0';
571  *p++ = 'x';
572  written += 2;
573  }
574  saved = p;
575 
576  // Divide ud by the divisor, until ud == 0
577  do
578  {
579  remainder = ud % divisor;
580  *p++ = (remainder < 10) ? remainder + '0' : remainder + 'a' - 10;
581  }
582  while (ud /= divisor);
583 
584  // Terminate buffer
585  *p = 0;
586 
587  // Initialize pointers
588  p1 = saved;
589  p2 = p - 1;
590  written += p2-p1+1;
591 
592  // Reverse buf
593  while (p1 < p2)
594  {
595  tmp = *p1;
596  *p1 = *p2;
597  *p2 = tmp;
598  p1++;
599  p2--;
600  }
601  // Update String administration, if needed.
602  if (!string)
603  m_count = written;
604 
605  return written;
606 }
