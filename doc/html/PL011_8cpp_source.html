29 Error PL011::initialize()
30 {
31  // Remap IO base to ensure we have user-level access to the registers.
32  if (m_io.map(IO_BASE + GPIO_BASE, PAGESIZE*2,
33  Memory::User | Memory::Readable | Memory::Writable | Memory::Device)
34  != IO::Success)
35  return EINVAL;
36 
37  // Disable receiving interrupts
38  ProcessCtl(SELF, DisableIRQ, 0);
39 
40  // Disable PL011.
41  m_io.write(PL011_CR, 0x00000000);
42 
43  // Setup the GPIO pin 14 && 15.
44  // Disable pull up/down for all GPIO pins & delay for 150 cycles.
45  m_io.write(GPPUD, 0x00000000);
46  delay(150);
47 
48  // Disable pull up/down for pin 14,15 & delay for 150 cycles.
49  m_io.write(GPPUDCLK0, (1 << 14) | (1 << 15));
50  delay(150);
51 
52  // Write 0 to GPPUDCLK0 to make it take effect.
53  m_io.write(GPPUDCLK0, 0x00000000);
54 
55  // Clear pending interrupts.
56  m_io.write(PL011_ICR, 0x7FF);
57 
58  // Set integer & fractional part of baud rate.
59  m_io.write(PL011_IBRD, 26);
60  m_io.write(PL011_FBRD, 3);
61 
62  // Disable FIFO, use 8 bit data transmission, 1 stop bit, no parity
63  m_io.write(PL011_LCRH, PL011_LCRH_WLEN_8BIT);
64 
65  // Enable Rx/Tx interrupts
66  m_io.write(PL011_IMSC,
67  PL011_IMSC_RXIM);
68 
69  // Enable PL011, receive & transfer part of UART.
70  m_io.write(PL011_CR, (1 << 0) | (1 << 8) | (1 << 9));
71  return 0;
72 }

74 Error PL011::interrupt(u32 vector)
75 {
76  // Clear Receive Interrupts
77  u32 mis = m_io.read(PL011_MIS);
78  if (mis & PL011_MIS_RXMIS)
79  m_io.write(PL011_ICR, PL011_ICR_RXIC);
80 
81  // Clear Transmit Interrupts

83  if (mis & PL011_MIS_TXMIS)
84  m_io.write(PL011_ICR, PL011_ICR_TXIC);
85

92 {

97  if (mis & PL011_MIS_RXMIS)
98  m_io.write(PL011_ICR, PL011_ICR_RXIC);

100  // Read as much bytes as possible
